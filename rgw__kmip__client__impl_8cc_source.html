63static void
64kmip_free_handle_stuff(RGWKmipHandle *kmip)
65{
66 if (kmip->encoding) {
67 kmip_free_buffer(kmip->kmip_ctx,
68 kmip->encoding,
69 kmip->buffer_total_size);
70 kmip_set_buffer(kmip->kmip_ctx, NULL, 0);
71 }
72 if (kmip->need_to_free_kmip)
73 kmip_destroy(kmip->kmip_ctx);
74 if (kmip->bio)
75 BIO_free_all(kmip->bio);
76 if (kmip->ctx)
77 SSL_CTX_free(kmip->ctx);
78}

152RGWKmipHandle *
153RGWKmipHandleBuilder::build() const
154{
155 int failed = 1;
156 RGWKmipHandle *r = new RGWKmipHandle();
157 TextString *up = 0;
158 size_t ns;
159
160 r->ctx = SSL_CTX_new(TLS_client_method());
161
162 if (!clientcert)
163 ;
164 else if (SSL_CTX_use_certificate_file(r->ctx, clientcert, SSL_FILETYPE_PEM) != 1) {
165 lderr(cct) << "ERROR: can't load client cert from "
166 << clientcert << dendl;
167 ERR_print_errors_ceph(cct);
168 goto Done;
169 }
170
171 if (!clientkey)
172 ;
173 else if (SSL_CTX_use_PrivateKey_file(r->ctx, clientkey,
174 SSL_FILETYPE_PEM) != 1) {
175 lderr(cct) << "ERROR: can't load client key from "
176 << clientkey << dendl;
177 ERR_print_errors_ceph(cct);
178 goto Done;
179 }
180
181 if (!capath)
182 ;
183 else if (SSL_CTX_load_verify_locations(r->ctx, capath, NULL) != 1) {
184 lderr(cct) << "ERROR: can't load cacert from "
185 << capath << dendl;
186 ERR_print_errors_ceph(cct);
187 goto Done;
188 }
189 r->bio = BIO_new_ssl_connect(r->ctx);
190 if (!r->bio) {
191 lderr(cct) << "BIO_new_ssl_connect failed" << dendl;
192 goto Done;
193 }
194 BIO_get_ssl(r->bio, &r->ssl);
195 SSL_set_mode(r->ssl, SSL_MODE_AUTO_RETRY);
196
197 BIO_set_conn_hostname(r->bio, host);
198 BIO_set_conn_port(r->bio, portstring);
199 if (BIO_do_connect(r->bio) != 1) {
200 lderr(cct) << "BIO_do_connect failed to " << host
201 << ":" << portstring << dendl;
202 ERR_print_errors_ceph(cct);
203 goto Done;
204 }
205
206 // setup kmip
207
208 kmip_init(r->kmip_ctx, NULL, 0, protocol_version);
209 r->need_to_free_kmip = 1;
210 r->buffer_blocks = 1;
211 r->buffer_block_size = 1024;
212 r->encoding = static_cast<uint8*>(r->kmip_ctx->calloc_func(
213 r->kmip_ctx->state, r->buffer_blocks, r->buffer_block_size));
214 if (!r->encoding) {
215 lderr(cct) << "kmip buffer alloc failed: "
216 << r->buffer_blocks <<
217 " * " << r->buffer_block_size << dendl;
218 goto Done;
219 }
220 ns = r->buffer_blocks * r->buffer_block_size;
221 kmip_set_buffer(r->kmip_ctx, r->encoding, ns);
222 r->buffer_total_size = ns;
223
224 up = r->textstrings;
225 if (username) {
226 memset(r->upc, 0, sizeof *r->upc);
227 up->value = (char *) username;
228 up->size = strlen(username);
229 r->upc->username = up++;
230 if (password) {
231 up->value = (char *) password;
232 up->size = strlen(password);
233 r->upc->password = up++;
234 }
235 r->credential->credential_type = KMIP_CRED_USERNAME_AND_PASSWORD;
236 r->credential->credential_value = r->upc;
237 int i = kmip_add_credential(r->kmip_ctx, r->credential);
238 if (i != KMIP_OK) {
239 fprintf(stderr,"failed to add credential to kmip\n");
240 goto Done;
241 }
242 }
243
244 failed = 0;
245Done:
246 if (!failed)
247 ;
248 else if (!r)
249 ;
250 else {
251 kmip_free_handle_stuff(r);
252 delete r;
253 r = 0;
254 }
255 return r;
256}

278RGWKmipHandle*
279RGWKmipHandles::get_kmip_handle()
280{
281 RGWKmipHandle* kmip = 0;
282 const char *hostaddr = cct->_conf->rgw_crypt_kmip_addr.c_str();
283 {
284 std::lock_guard lock{cleaner_lock};
285 if (!saved_kmip.empty()) {
286 kmip = *saved_kmip.begin();
287 saved_kmip.erase(saved_kmip.begin());
288 }
289 }
290 if (!kmip && hostaddr) {
291 char *hosttemp = strdup(hostaddr);
292 char *port = strchr(hosttemp, ':');
293 if (port)
294 *port++ = 0;
295 kmip = RGWKmipHandleBuilder{cct}
296 .set_clientcert(cct->_conf->rgw_crypt_kmip_client_cert)
297 .set_clientkey(cct->_conf->rgw_crypt_kmip_client_key)
298 .set_capath(cct->_conf->rgw_crypt_kmip_ca_path)
299 .set_host(hosttemp)
300 .set_portstring(port ? port : "5696")
301 .set_username(cct->_conf->rgw_crypt_kmip_username)
302 .set_password(cct->_conf->rgw_crypt_kmip_password)
303 .build();
304 free(hosttemp);
305 }
306 return kmip;
307}

316#define MAXIDLE 5
317void
318RGWKmipHandles::release_kmip_handle(RGWKmipHandle* kmip)
319{
320 if (cleaner_shutdown) {
321 release_kmip_handle_now(kmip);
322 } else {
323 std::lock_guard lock{cleaner_lock};
324 kmip->lastuse = mono_clock::now();
325 saved_kmip.insert(saved_kmip.begin(), 1, kmip);
326 }

329void*
330RGWKmipHandles::entry()
331{
332 RGWKmipHandle* kmip;
333 std::unique_lock lock{cleaner_lock};
334
335 for (;;) {
336 if (cleaner_shutdown) {
337 if (saved_kmip.empty())
338 break;
339 } else {
340 cleaner_cond.wait_for(lock, std::chrono::seconds(MAXIDLE));
341 }
342 mono_time now = mono_clock::now();
343 while (!saved_kmip.empty()) {
344 auto cend = saved_kmip.end();
345 --cend;
346 kmip = *cend;
347 if (!cleaner_shutdown && now - kmip->lastuse
348 < std::chrono::seconds(MAXIDLE))
349 break;
350 saved_kmip.erase(cend);
351 release_kmip_handle_now(kmip);
352 }
353 }
354 return nullptr;
355}

428int
429RGWKmipHandles::do_one_entry(RGWKMIPTransceiver &element)
430{
431 auto h = get_kmip_handle();
432 std::unique_lock l{element.lock};
433 Attribute a[8], *ap;
434 TextString nvalue[1], uvalue[1];
435 Name nattr[1];
436 enum cryptographic_algorithm alg = KMIP_CRYPTOALG_AES;
437 int32 length = 256;
438 int32 mask = KMIP_CRYPTOMASK_ENCRYPT | KMIP_CRYPTOMASK_DECRYPT;
439 size_t ns;
440 ProtocolVersion pv[1];
441 RequestHeader rh[1];
442 RequestMessage rm[1];
443 Authentication auth[1];
444 ResponseMessage resp_m[1];
445 int i;
446 union {
447 CreateRequestPayload create_req[1];
448 LocateRequestPayload locate_req[1];
449 GetRequestPayload get_req[1];
450 GetAttributeListRequestPayload lsattrs_req[1];
451 GetAttributesRequestPayload getattrs_req[1];
452 } u[1];
453 RequestBatchItem rbi[1];
454 TemplateAttribute ta[1];
455 const char *what = "?";
456 int need_to_free_response = 0;
457 char *response = NULL;
458 int response_size = 0;
459 enum result_status rs;
460 ResponseBatchItem *req;
461
462 if (!h) {
463 element.ret = -ERR_SERVICE_UNAVAILABLE;
464 return element.ret;
465 }
466 memset(a, 0, sizeof *a);
467 for (i = 0; i < (int)(sizeof a/sizeof *a); ++i)
468 kmip_init_attribute(a+i);
469 ap = a;
470 switch(element.operation) {
471 case RGWKMIPTransceiver::CREATE:
472 ap->type = KMIP_ATTR_CRYPTOGRAPHIC_ALGORITHM;
473 ap->value = &alg;
474 ++ap;
475 ap->type = KMIP_ATTR_CRYPTOGRAPHIC_LENGTH;
476 ap->value = &length;
477 ++ap;
478 ap->type = KMIP_ATTR_CRYPTOGRAPHIC_USAGE_MASK;
479 ap->value = &mask;
480 ++ap;
481 break;
482 default:
483 break;
484 }
485 if (element.name) {
486 memset(nvalue, 0, sizeof *nvalue);
487 nvalue->value = element.name;
488 nvalue->size = strlen(element.name);
489 memset(nattr, 0, sizeof *nattr);
490 nattr->value = nvalue;
491 nattr->type = KMIP_NAME_UNINTERPRETED_TEXT_STRING;
492 ap->type = KMIP_ATTR_NAME;
493 ap->value = nattr;
494 ++ap;
495 }
496 if (element.unique_id) {
497 memset(uvalue, 0, sizeof *uvalue);
498 uvalue->value = element.unique_id;
499 uvalue->size = strlen(element.unique_id);
500 }
501 memset(pv, 0, sizeof *pv);
502 memset(rh, 0, sizeof *rh);
503 memset(rm, 0, sizeof *rm);
504 memset(auth, 0, sizeof *auth);
505 memset(resp_m, 0, sizeof *resp_m);
506 kmip_init_protocol_version(pv, h->kmip_ctx->version);
507 kmip_init_request_header(rh);
508 rh->protocol_version = pv;
509 rh->maximum_response_size = h->kmip_ctx->max_message_size;
510 rh->time_stamp = time(NULL);
511 rh->batch_count = 1;
512 memset(rbi, 0, sizeof *rbi);
513 kmip_init_request_batch_item(rbi);
514 memset(u, 0, sizeof *u);
515 rbi->request_payload = u;
516 switch(element.operation) {
517 case RGWKMIPTransceiver::CREATE:
518 memset(ta, 0, sizeof *ta);
519 ta->attributes = a;
520 ta->attribute_count = ap-a;
521 u->create_req->object_type = KMIP_OBJTYPE_SYMMETRIC_KEY;
522 u->create_req->template_attribute = ta;
523 rbi->operation = KMIP_OP_CREATE;
524 what = "create";
525 break;
526 case RGWKMIPTransceiver::GET:
527 if (element.unique_id)
528 u->get_req->unique_identifier = uvalue;
529 rbi->operation = KMIP_OP_GET;
530 what = "get";
531 break;
532 case RGWKMIPTransceiver::LOCATE:
533 if (ap > a) {
534 u->locate_req->attributes = a;
535 u->locate_req->attribute_count = ap - a;
536 }
537 rbi->operation = KMIP_OP_LOCATE;
538 what = "locate";
539 break;
540 case RGWKMIPTransceiver::GET_ATTRIBUTES:
541 case RGWKMIPTransceiver::GET_ATTRIBUTE_LIST:
542 case RGWKMIPTransceiver::DESTROY:
543 default:
544 lderr(cct) << "Missing operation logic op=" << element.operation << dendl;
545 element.ret = -EINVAL;
546 goto Done;
547 }
548 rm->request_header = rh;
549 rm->batch_items = rbi;
550 rm->batch_count = 1;
551 if (h->kmip_ctx->credential_list) {
552 LinkedListItem *item = h->kmip_ctx->credential_list->head;
553 if (item) {
554 auth->credential = (Credential *)item->data;
555 rh->authentication = auth;
556 }
557 }
558 for (;;) {
559 i = kmip_encode_request_message(h->kmip_ctx, rm);
560 if (i != KMIP_ERROR_BUFFER_FULL) break;
561 h->kmip_ctx->free_func(h->kmip_ctx->state, h->encoding);
562 h->encoding = 0;
563 ++h->buffer_blocks;
564 h->encoding = static_cast<uint8*>(h->kmip_ctx->calloc_func(h->kmip_ctx->state, h->buffer_blocks, h->buffer_block_size));
565 if (!h->encoding) {
566 lderr(cct) << "kmip buffer alloc failed: "
567 << h->buffer_blocks
568 << " * " << h->buffer_block_size << dendl;
569 element.ret = -ENOMEM;
570 goto Done;
571 }
572 ns = h->buffer_blocks * h->buffer_block_size;
573 kmip_set_buffer(h->kmip_ctx, h->encoding, ns);
574 h->buffer_total_size = ns;
575 }
576 if (i != KMIP_OK) {
577 lderr(cct) << " Failed to encode " << what
578 << " request; err=" << i
579 << " ctx error message " << h->kmip_ctx->error_message
580 << dendl;
581 element.ret = -EINVAL;
582 goto Done;
583 }
584 i = kmip_bio_send_request_encoding(h->kmip_ctx, h->bio,
585 (char*)h->encoding,
586 h->kmip_ctx->index - h->kmip_ctx->buffer,
587 &response, &response_size);
588 if (i < 0) {
589 lderr(cct) << "Problem sending request to " << what << " " << i << " context error message " << h->kmip_ctx->error_message << dendl;
590 element.ret = -EINVAL;
591 goto Done;
592 }
593 kmip_free_buffer(h->kmip_ctx, h->encoding,
594 h->buffer_total_size);
595 h->encoding = 0;
596 kmip_set_buffer(h->kmip_ctx, response, response_size);
597 need_to_free_response = 1;
598 i = kmip_decode_response_message(h->kmip_ctx, resp_m);
599 if (i != KMIP_OK) {
600 lderr(cct) << "Failed to decode " << what << " " << i << " context error message " << h->kmip_ctx->error_message << dendl;
601 element.ret = -EINVAL;
602 goto Done;
603 }
604 if (resp_m->batch_count != 1) {
605 lderr(cct) << "Failed; weird response count doing " << what << " " << resp_m->batch_count << dendl;
606 element.ret = -EINVAL;
607 goto Done;
608 }
609 req = resp_m->batch_items;
610 rs = req->result_status;
611 if (rs != KMIP_STATUS_SUCCESS) {
612 lderr(cct) << "Failed; result status not success " << rs << dendl;
613 element.ret = -EINVAL;
614 goto Done;
615 }
616 if (req->operation != rbi->operation) {
617 lderr(cct) << "Failed; response operation mismatch, got " << req->operation << " expected " << rbi->operation << dendl;
618 element.ret = -EINVAL;
619 goto Done;
620 }
621 switch(req->operation)
622 {
623 case KMIP_OP_CREATE: {
624 CreateResponsePayload *pld = (CreateResponsePayload *)req->response_payload;
625 element.out = static_cast<char *>(malloc(pld->unique_identifier->size+1));
626 memcpy(element.out, pld->unique_identifier->value, pld->unique_identifier->size);
627 element.out[pld->unique_identifier->size] = 0;
628 } break;
629 case KMIP_OP_LOCATE: {
630 LocateResponsePayload *pld = (LocateResponsePayload *)req->response_payload;
631 char **list = static_cast<char **>(malloc(sizeof (char*) * (1 + pld->unique_identifiers_count)));
632 for (i = 0; i < pld->unique_identifiers_count; ++i) {
633 list[i] = static_cast<char *>(malloc(pld->unique_identifiers[i].size+1));
634 memcpy(list[i], pld->unique_identifiers[i].value, pld->unique_identifiers[i].size);
635 list[i][pld->unique_identifiers[i].size] = 0;
636 }
637 list[i] = 0;
638 element.outlist->strings = list;
639 element.outlist->string_count = pld->unique_identifiers_count;
640 } break;
641 case KMIP_OP_GET: {
642 GetResponsePayload *pld = (GetResponsePayload *)req->response_payload;
643 element.out = static_cast<char *>(malloc(pld->unique_identifier->size+1));
644 memcpy(element.out, pld->unique_identifier->value, pld->unique_identifier->size);
645 element.out[pld->unique_identifier->size] = 0;
646 if (pld->object_type != KMIP_OBJTYPE_SYMMETRIC_KEY) {
647 lderr(cct) << "get: expected symmetric key got " << pld->object_type << dendl;
648 element.ret = -EINVAL;
649 goto Done;
650 }
651 KeyBlock *kp = static_cast<SymmetricKey *>(pld->object)->key_block;
652 ByteString *bp;
653 if (kp->key_format_type != KMIP_KEYFORMAT_RAW) {
654 lderr(cct) << "get: expected raw key fromat got " << kp->key_format_type << dendl;
655 element.ret = -EINVAL;
656 goto Done;
657 }
658 KeyValue *kv = static_cast<KeyValue *>(kp->key_value);
659 bp = static_cast<ByteString*>(kv->key_material);
660 element.outkey->data = static_cast<unsigned char *>(malloc(bp->size));
661 element.outkey->keylen = bp->size;
662 memcpy(element.outkey->data, bp->value, bp->size);
663 } break;
664 case KMIP_OP_GET_ATTRIBUTES: {
665 GetAttributesResponsePayload *pld = (GetAttributesResponsePayload *)req->response_payload;
666 element.out = static_cast<char *>(malloc(pld->unique_identifier->size+1));
667 memcpy(element.out, pld->unique_identifier->value, pld->unique_identifier->size);
668 element.out[pld->unique_identifier->size] = 0;
669 } break;
670 case KMIP_OP_GET_ATTRIBUTE_LIST: {
671 GetAttributeListResponsePayload *pld = (GetAttributeListResponsePayload *)req->response_payload;
672 element.out = static_cast<char *>(malloc(pld->unique_identifier->size+1));
673 memcpy(element.out, pld->unique_identifier->value, pld->unique_identifier->size);
674 element.out[pld->unique_identifier->size] = 0;
675 } break;
676 case KMIP_OP_DESTROY: {
677 DestroyResponsePayload *pld = (DestroyResponsePayload *)req->response_payload;
678 element.out = static_cast<char *>(malloc(pld->unique_identifier->size+1));

711 m.requests.erase(iter);

714 entry_lock.lock();

716 for (;;) {

718 auto iter = m.requests.begin();

720 m.requests.erase(iter);
721 element.details.ret = -666;
722 element.details.done = true;
723 element.details.cond.notify_all();
724 }
725 handles.stop();

727 return nullptr;
