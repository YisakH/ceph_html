103/* A converter between std::string_view and null-terminated C-strings.
104 * It copies memory while trying to utilize the local memory instead of
105 * issuing dynamic allocations. */
106template<std::size_t N = 128>
107static inline boost::container::small_vector<char, N>
108sview2cstr(const std::string_view& sv)
109{
110 boost::container::small_vector<char, N> cstr;
111 cstr.reserve(sv.size() + sizeof('\0'));
112
113 cstr.assign(std::begin(sv), std::end(sv));
114 cstr.push_back('\0');
115
116 return cstr;
117}

199template <typename... Args>
200std::string string_cat_reserve(const Args&... args)
201{
202 size_t total_size = detail::sum(string_size(args)...);
203 std::string result;
204 result.reserve(total_size);

211template <typename... Args>
212std::string string_join_reserve(const std::string_view& delim,
213 const Args&... args)
214{
215 size_t delim_size = delim.size() * std::max<ssize_t>(0, sizeof...(args) - 1);
216 size_t total_size = detail::sum(string_size(args)...) + delim_size;
217 std::string result;
218 result.reserve(total_size);
219 detail::join(result, delim, args...);
220 return result;
221}
