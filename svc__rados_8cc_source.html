75int RGWSI_RADOS::pool_iterate(const DoutPrefixProvider *dpp,
76 librados::IoCtx& io_ctx,
77 librados::NObjectIterator& iter,
78 uint32_t num, vector<rgw_bucket_dir_entry>& objs,
79 RGWAccessListFilter *filter,
80 bool *is_truncated)
81{
82 if (iter == io_ctx.nobjects_end())
83 return -ENOENT;
84
85 uint32_t i;
86
87 for (i = 0; i < num && iter != io_ctx.nobjects_end(); ++i, ++iter) {
88 rgw_bucket_dir_entry e;
89
90 string oid = iter->get_oid();
91 ldpp_dout(dpp, 20) << "RGWRados::pool_iterate: got " << oid << dendl;
92
93 // fill it in with initial values; we may correct later
94 if (filter && !filter->filter(oid, oid))
95 continue;
96
97 e.key = oid;
98 objs.push_back(e);
99 }
100
101 if (is_truncated)
102 *is_truncated = (iter != io_ctx.nobjects_end());
103
104 return objs.size();
105}

209int RGWSI_RADOS::Pool::create(const DoutPrefixProvider *dpp, const vector<rgw_pool>& pools, vector<int> *retcodes)
210{
211 vector<librados::PoolAsyncCompletion *> completions;
212 vector<int> rets;
213
214 librados::Rados *rad = rados_svc->get_rados_handle();
215 for (auto iter = pools.begin(); iter != pools.end(); ++iter) {
216 librados::PoolAsyncCompletion *c = librados::Rados::pool_async_create_completion();
217 completions.push_back(c);
218 auto& pool = *iter;
219 int ret = rad->pool_create_async(pool.name.c_str(), c);
220 rets.push_back(ret);
221 }
222
223 vector<int>::iterator riter;
224 vector<librados::PoolAsyncCompletion *>::iterator citer;
225
226 bool error = false;
227 ceph_assert(rets.size() == completions.size());
228 for (riter = rets.begin(), citer = completions.begin(); riter != rets.end(); ++riter, ++citer) {
229 int r = *riter;
230 librados::PoolAsyncCompletion *c = *citer;
231 if (r == 0) {
232 c->wait();
233 r = c->get_return_value();
234 if (r < 0) {
235 ldpp_dout(dpp, 0) << "WARNING: async pool_create returned " << r << dendl;
236 error = true;
237 }
238 }
239 c->release();
240 retcodes->push_back(r);
241 }
242 if (error) {
243 return 0;
244 }
245
246 std::vector<librados::IoCtx> io_ctxs;
247 retcodes->clear();
248 for (auto pool : pools) {
249 io_ctxs.emplace_back();
250 int ret = rad->ioctx_create(pool.name.c_str(), io_ctxs.back());
251 if (ret < 0) {
252 ldpp_dout(dpp, 0) << "WARNING: ioctx_create returned " << ret << dendl;
253 error = true;
254 }
255 retcodes->push_back(ret);
256 }
257 if (error) {
258 return 0;
259 }
260
261 completions.clear();
262 for (auto &io_ctx : io_ctxs) {
263 librados::PoolAsyncCompletion *c =
264 librados::Rados::pool_async_create_completion();
265 completions.push_back(c);
266 int ret = io_ctx.application_enable_async(pg_pool_t::APPLICATION_NAME_RGW,
267 false, c);
268 ceph_assert(ret == 0);
269 }
270
271 retcodes->clear();
272 for (auto c : completions) {
273 c->wait();
274 int ret = c->get_return_value();
275 if (ret == -EOPNOTSUPP) {
276 ret = 0;
277 } else if (ret < 0) {
278 ldpp_dout(dpp, 0) << "WARNING: async application_enable returned " << ret
279 << dendl;
280 error = true;
281 }
282 c->release();
283 retcodes->push_back(ret);
284 }
285 return 0;
286}

304int RGWSI_RADOS::Pool::List::init(const DoutPrefixProvider *dpp, const string& marker, RGWAccessListFilter *filter)
305{
306 if (ctx.initialized) {
307 return -EINVAL;
308 }
309
310 if (!pool) {
311 return -EINVAL;
312 }
313
314 int r = pool->rados_svc->open_pool_ctx(dpp, pool->pool, ctx.ioctx);
315 if (r < 0) {
316 return r;
317 }
318
319 librados::ObjectCursor oc;
320 if (!oc.from_str(marker)) {
321 ldpp_dout(dpp, 10) << "failed to parse cursor: " << marker << dendl;
322 return -EINVAL;
323 }
324
325 ctx.iter = ctx.ioctx.nobjects_begin(oc);
326 ctx.filter = filter;
327 ctx.initialized = true;
328
329 return 0;

399bool RGWSI_RADOS::check_secure_mon_conn(const DoutPrefixProvider *dpp) const
400{
401 AuthRegistry reg(cct);
402
403 reg.refresh_config();
404
405 std::vector<uint32_t> methods;
406 std::vector<uint32_t> modes;
407
408 reg.get_supported_methods(CEPH_ENTITY_TYPE_MON, &methods, &modes);
409 ldpp_dout(dpp, 20) << __func__ << "(): auth registy supported: methods=" << methods << " modes=" << modes << dendl;
410
411 for (auto method : methods) {
412 if (!reg.is_secure_method(method)) {
413 ldpp_dout(dpp, 20) << __func__ << "(): method " << method << " is insecure" << dendl;
414 return false;
415 }
416 }
417
418 for (auto mode : modes) {
419 if (!reg.is_secure_mode(mode)) {
420 ldpp_dout(dpp, 20) << __func__ << "(): mode " << mode << " is insecure" << dendl;
421 return false;
422 }
423 }
424
425 return true;
426}
