45int RGWSI_BucketIndex_RADOS::open_bucket_index_pool(const DoutPrefixProvider *dpp,
46 const RGWBucketInfo& bucket_info,
47 RGWSI_RADOS::Pool *index_pool)
48{
49 const rgw_pool& explicit_pool = bucket_info.bucket.explicit_placement.index_pool;
50
51 if (!explicit_pool.empty()) {
52 return open_pool(dpp, explicit_pool, index_pool, false);
53 }
54
55 auto& zonegroup = svc.zone->get_zonegroup();
56 auto& zone_params = svc.zone->get_zone_params();
57
58 const rgw_placement_rule *rule = &bucket_info.placement_rule;
59 if (rule->empty()) {
60 rule = &zonegroup.default_placement;
61 }
62 auto iter = zone_params.placement_pools.find(rule->name);
63 if (iter == zone_params.placement_pools.end()) {
64 ldpp_dout(dpp, 0) << "could not find placement rule " << *rule << " within zonegroup " << dendl;
65 return -EINVAL;
66 }
67
68 int r = open_pool(dpp, iter->second.index_pool, index_pool, true);
69 if (r < 0)
70 return r;
71
72 return 0;
73}

121static void get_bucket_index_objects(const string& bucket_oid_base,
122 uint32_t num_shards,
123 map<int, string> *_bucket_objects,
124 int shard_id = -1)
125{
126 auto& bucket_objects = *_bucket_objects;
127 if (!num_shards) {
128 bucket_objects[0] = bucket_oid_base;
129 } else {
130 char buf[bucket_oid_base.size() + 32];
131 if (shard_id < 0) {
132 for (uint32_t i = 0; i < num_shards; ++i) {
133 snprintf(buf, sizeof(buf), "%s.%d", bucket_oid_base.c_str(), i);
134 bucket_objects[i] = buf;
135 }
136 } else {
137 if ((uint32_t)shard_id > num_shards) {
138 return;
139 }
140 snprintf(buf, sizeof(buf), "%s.%d", bucket_oid_base.c_str(), shard_id);
141 bucket_objects[shard_id] = buf;
142 }
143 }
144}

146static void get_bucket_instance_ids(const RGWBucketInfo& bucket_info,
147 int shard_id,
148 map<int, string> *result)
149{
150 const rgw_bucket& bucket = bucket_info.bucket;
151 string plain_id = bucket.name + ":" + bucket.bucket_id;
152
153 if (!bucket_info.layout.current_index.layout.normal.num_shards) {
154 (*result)[0] = plain_id;
155 } else {
156 char buf[16];
157 if (shard_id < 0) {
158 for (uint32_t i = 0; i < bucket_info.layout.current_index.layout.normal.num_shards; ++i) {
159 snprintf(buf, sizeof(buf), ":%d", i);
160 (*result)[i] = plain_id + buf;
161 }
162 } else {
163 if ((uint32_t)shard_id > bucket_info.layout.current_index.layout.normal.num_shards) {
164 return;
165 }
166 snprintf(buf, sizeof(buf), ":%d", shard_id);
167 (*result)[shard_id] = plain_id + buf;
168 }
169 }

173 const RGWBucketInfo& bucket_info,

195void RGWSI_BucketIndex_RADOS::get_bucket_index_object(const string& bucket_oid_base,
196 uint32_t num_shards,
197 int shard_id,
198 uint64_t gen_id,
199 string *bucket_obj)
200{
201 if (!num_shards) {
202 // By default with no sharding, we use the bucket oid as itself
203 (*bucket_obj) = bucket_oid_base;
204 } else {
205 char buf[bucket_oid_base.size() + 64];
206 if (gen_id != 0) {
207 snprintf(buf, sizeof(buf), "%s.%" PRIu64 ".%d", bucket_oid_base.c_str(), gen_id, shard_id);
208 (*bucket_obj) = buf;
209 } else {
210 // for backward compatibility, gen_id(0) will not be added in the object name
211 snprintf(buf, sizeof(buf), "%s.%d", bucket_oid_base.c_str(), shard_id);
212 (*bucket_obj) = buf;
213 }
214 }
215}

217int RGWSI_BucketIndex_RADOS::get_bucket_index_object(const string& bucket_oid_base, const string& obj_key,
218 uint32_t num_shards, rgw::BucketHashType hash_type,
219 string *bucket_obj, int *shard_id)
220{
221 int r = 0;
222 switch (hash_type) {
223 case rgw::BucketHashType::Mod:
224 if (!num_shards) {
225 // By default with no sharding, we use the bucket oid as itself
226 (*bucket_obj) = bucket_oid_base;
227 if (shard_id) {
228 *shard_id = -1;
229 }
230 } else {
231 uint32_t sid = bucket_shard_index(obj_key, num_shards);
232 char buf[bucket_oid_base.size() + 32];
233 snprintf(buf, sizeof(buf), "%s.%d", bucket_oid_base.c_str(), sid);
234 (*bucket_obj) = buf;
235 if (shard_id) {
236 *shard_id = (int)sid;
237 }
238 }
239 break;
240 default:
241 r = -ENOTSUP;
242 }
243 return r;
244}

372int RGWSI_BucketIndex_RADOS::read_stats(const DoutPrefixProvider *dpp,
373 const RGWBucketInfo& bucket_info,
374 RGWBucketEnt *result,
375 optional_yield y)
376{
377 vector<rgw_bucket_dir_header> headers;
378
379 result->bucket = bucket_info.bucket;
380 int r = cls_bucket_head(dpp, bucket_info, RGW_NO_SHARD, &headers, nullptr, y);
381 if (r < 0) {
382 return r;
383 }
384
385 result->count = 0;
386 result->size = 0;
387 result->size_rounded = 0;
388
389 auto hiter = headers.begin();
390 for (; hiter != headers.end(); ++hiter) {
391 RGWObjCategory category = RGWObjCategory::Main;
392 auto iter = (hiter->stats).find(category);
393 if (iter != hiter->stats.end()) {
394 struct rgw_bucket_category_stats& stats = iter->second;
395 result->count += stats.num_entries;
396 result->size += stats.total_size;
397 result->size_rounded += stats.total_size_rounded;
398 }
399 }
400
401 result->placement_rule = std::move(bucket_info.placement_rule);
402
403 return 0;
404}

436int RGWSI_BucketIndex_RADOS::handle_overwrite(const DoutPrefixProvider *dpp,
437 const RGWBucketInfo& info,
438 const RGWBucketInfo& orig_info)
439{
440 bool new_sync_enabled = info.datasync_flag_enabled();
441 bool old_sync_enabled = orig_info.datasync_flag_enabled();
442
443 if (old_sync_enabled != new_sync_enabled) {
444 int shards_num = info.layout.current_index.layout.normal.num_shards? info.layout.current_index.layout.normal.num_shards : 1;
445 int shard_id = info.layout.current_index.layout.normal.num_shards? 0 : -1;
446
447 int ret;
448 if (!new_sync_enabled) {
449 ret = svc.bilog->log_stop(dpp, info, -1);
450 } else {
451 ret = svc.bilog->log_start(dpp, info, -1);
452 }
453 if (ret < 0) {
454 ldpp_dout(dpp, -1) << "ERROR: failed writing bilog (bucket=" << info.bucket << "); ret=" << ret << dendl;
455 return ret;
456 }
457
458 for (int i = 0; i < shards_num; ++i, ++shard_id) {
459 ret = svc.datalog_rados->add_entry(dpp, info, shard_id);
460 if (ret < 0) {
461 ldpp_dout(dpp, -1) << "ERROR: failed writing data log (info.bucket=" << info.bucket << ", shard_id=" << shard_id << ")" << dendl;
462 return ret;
463 }
464 }
465 }
466
467 return 0;
