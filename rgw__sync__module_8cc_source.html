42int RGWCallStatRemoteObjCR::operate(const DoutPrefixProvider *dpp) {
43 reenter(this) {
44 yield {
45 call(new RGWStatRemoteObjCR(sync_env->async_rados, sync_env->store,
46 sc->source_zone,
47 src_bucket, key, &mtime, &size, &etag, &attrs, &headers));
48 }
49 if (retcode < 0) {
50 ldpp_dout(dpp, 10) << "RGWStatRemoteObjCR() returned " << retcode << dendl;
51 return set_cr_error(retcode);
52 }
53 ldpp_dout(dpp, 20) << "stat of remote obj: z=" << sc->source_zone
54 << " b=" << src_bucket << " k=" << key
55 << " size=" << size << " mtime=" << mtime << dendl;
56 yield {
57 RGWStatRemoteObjCBCR *cb = allocate_callback();
58 if (cb) {
59 cb->set_result(mtime, size, etag, std::move(attrs), std::move(headers));
60 call(cb);
61 }
62 }
63 if (retcode < 0) {
64 ldpp_dout(dpp, 10) << "RGWStatRemoteObjCR() callback returned " << retcode << dendl;
65 return set_cr_error(retcode);
66 }
67 return set_cr_done();
68 }
69 return 0;
70}
