25int RGWCRHTTPGetDataCB::handle_data(bufferlist& bl, bool *pause) {
26 if (data.length() < GET_DATA_WINDOW_SIZE / 2) {
27 notified = false;
28 }
29
30 {
31 uint64_t bl_len = bl.length();
32
33 std::lock_guard l{lock};
34
35 if (!got_all_extra_data) {
36 uint64_t max = extra_data_len - extra_data.length();
37 if (max > bl_len) {
38 max = bl_len;
39 }
40 bl.splice(0, max, &extra_data);
41 bl_len -= max;
42 got_all_extra_data = extra_data.length() == extra_data_len;
43 }
44
45 data.append(bl);
46 }
47
48 uint64_t data_len = data.length();
49 if (data_len >= GET_DATA_WINDOW_SIZE && !notified) {
50 notified = true;
51 env->manager->io_complete(cr, io_id);
52 }
53 if (data_len >= 2 * GET_DATA_WINDOW_SIZE) {
54 *pause = true;
55 paused = true;
56 }
57 return 0;
58}

60void RGWCRHTTPGetDataCB::claim_data(bufferlist *dest, uint64_t max) {
61 bool need_to_unpause = false;
62
63 {
64 std::lock_guard l{lock};
65
66 if (data.length() == 0) {
67 return;
68 }
69
70 if (data.length() < max) {
71 max = data.length();
72 }
73
74 data.splice(0, max, dest);
75 need_to_unpause = (paused && data.length() <= GET_DATA_WINDOW_SIZE);
76 }
77
78 if (need_to_unpause) {
79 req->unpause_receive();
80 }

141int RGWStreamReadHTTPResourceCRF::read(const DoutPrefixProvider *dpp, bufferlist *out, uint64_t max_size, bool *io_pending)
142{
143 reenter(&read_state) {
144 io_read_mask = req->get_io_id(RGWHTTPClient::HTTPCLIENT_IO_READ | RGWHTTPClient::HTTPCLIENT_IO_CONTROL);
145 while (!req->is_done() ||
146 in_cb->has_data()) {
147 *io_pending = true;
148 if (!in_cb->has_data()) {
149 yield caller->io_block(0, io_read_mask);
150 }
151 got_attrs = true;
152 if (need_extra_data() && !got_extra_data) {
153 if (!in_cb->has_all_extra_data()) {
154 continue;
155 }
156 extra_data.claim_append(in_cb->get_extra_data());
157 map<string, string> attrs;
158 req->get_out_headers(&attrs);
159 int ret = decode_rest_obj(dpp, attrs, extra_data);
160 if (ret < 0) {
161 ldout(cct, 0) << "ERROR: " << __func__ << " decode_rest_obj() returned ret=" << ret << dendl;
162 return ret;
163 }
164 got_extra_data = true;
165 }
166 *io_pending = false;
167 in_cb->claim_data(out, max_size);
168 if (out->length() == 0) {
169 /* this may happen if we just read the prepended extra_data and didn't have any data
170 * after. In that case, retry reading, so that caller doesn't assume it's EOF.
171 */
172 continue;
173 }
174 if (!req->is_done() || out->length() >= max_size) {
175 yield;
176 }
177 }
178 }
179 return 0;
180}

220int RGWStreamWriteHTTPResourceCRF::write(bufferlist& data, bool *io_pending)
221{
222 reenter(&write_state) {
223 while (!req->is_done()) {
224 *io_pending = false;
225 if (req->get_pending_send_size() >= PENDING_WRITES_WINDOW) {
226 *io_pending = true;
227 {
228 lock_guard l(blocked_lock);
229 is_blocked = true;
230
231 /* it's ok to unlock here, even if io_complete() arrives before io_block(), it'll wakeup
232 * correctly */
233 }
234 yield caller->io_block(0, req->get_io_id(RGWHTTPClient::HTTPCLIENT_IO_WRITE | RGWHTTPClient::HTTPCLIENT_IO_CONTROL));
235 }
236 yield req->add_send_data(data);
237 }
238 return req->get_status();
239 }
240 return 0;
241}

243int RGWStreamWriteHTTPResourceCRF::drain_writes(bool *need_retry)
244{
245 reenter(&drain_state) {
246 *need_retry = true;
247 yield req->finish_write();
248 *need_retry = !req->is_done();
249 while (!req->is_done()) {
250 yield caller->io_block(0, req->get_io_id(RGWHTTPClient::HTTPCLIENT_IO_CONTROL));
251 *need_retry = !req->is_done();
252 }
253
254 map<string, string> headers;
255 req->get_out_headers(&headers);
256 handle_headers(headers);
257
258 return req->get_req_retcode();
259 }
260 return 0;
261}

269int RGWStreamSpliceCR::operate(const DoutPrefixProvider *dpp) {
270 reenter(this) {
271 {
272 int ret = in_crf->init(dpp);
273 if (ret < 0) {
274 return set_cr_error(ret);
275 }
276 }
277
278 do {
279
280 bl.clear();
281
282 do {
283 yield {
284 ret = in_crf->read(dpp, &bl, 4 * 1024 * 1024, &need_retry);
285 if (ret < 0) {
286 return set_cr_error(ret);
287 }
288 }
289
290 if (retcode < 0) {
291 ldout(cct, 20) << __func__ << ": in_crf->read() retcode=" << retcode << dendl;
292 return set_cr_error(ret);
293 }
294 } while (need_retry);
295
296 ldout(cct, 20) << "read " << bl.length() << " bytes" << dendl;
297
298 if (!in_crf->has_attrs()) {
299 assert (bl.length() == 0);
300 continue;
301 }
302
303 if (!sent_attrs) {
304 int ret = out_crf->init();
305 if (ret < 0) {
306 return set_cr_error(ret);
307 }
308 out_crf->send_ready(dpp, in_crf->get_rest_obj());
309 ret = out_crf->send();
310 if (ret < 0) {
311 return set_cr_error(ret);
312 }
313 sent_attrs = true;
314 }
315
316 if (bl.length() == 0 && in_crf->is_done()) {
317 break;
318 }
319
320 total_read += bl.length();
321
322 do {
323 yield {
324 ldout(cct, 20) << "writing " << bl.length() << " bytes" << dendl;
325 ret = out_crf->write(bl, &need_retry);
326 if (ret < 0) {
327 return set_cr_error(ret);
328 }
329 }
330
331 if (retcode < 0) {
332 ldout(cct, 20) << __func__ << ": out_crf->write() retcode=" << retcode << dendl;
333 return set_cr_error(ret);
334 }
335 } while (need_retry);
336 } while (true);
337
338 do {
339 yield {
340 int ret = out_crf->drain_writes(&need_retry);
341 if (ret < 0) {
342 return set_cr_error(ret);
343 }
344 }
345 } while (need_retry);
346
347 return set_cr_done();
348 }
349 return 0;
350}
