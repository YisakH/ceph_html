2104class RGWFormPost::SignatureHelper
2105{
2106private:
2107 static constexpr uint32_t output_size =
2108 CEPH_CRYPTO_HMACSHA1_DIGESTSIZE * 2 + 1;
2109
2110 unsigned char dest[CEPH_CRYPTO_HMACSHA1_DIGESTSIZE]; // 20
2111 char dest_str[output_size];
2112
2113public:
2114 SignatureHelper() = default;
2115
2116 const char* calc(const std::string& key,
2117 const std::string_view& path_info,
2118 const std::string_view& redirect,
2119 const std::string_view& max_file_size,
2120 const std::string_view& max_file_count,
2121 const std::string_view& expires) {
2122 using ceph::crypto::HMACSHA1;
2123 using UCHARPTR = const unsigned char*;
2124
2125 HMACSHA1 hmac((UCHARPTR) key.data(), key.size());
2126
2127 hmac.Update((UCHARPTR) path_info.data(), path_info.size());
2128 hmac.Update((UCHARPTR) "\n", 1);
2129
2130 hmac.Update((UCHARPTR) redirect.data(), redirect.size());
2131 hmac.Update((UCHARPTR) "\n", 1);
2132
2133 hmac.Update((UCHARPTR) max_file_size.data(), max_file_size.size());
2134 hmac.Update((UCHARPTR) "\n", 1);
2135
2136 hmac.Update((UCHARPTR) max_file_count.data(), max_file_count.size());
2137 hmac.Update((UCHARPTR) "\n", 1);
2138
2139 hmac.Update((UCHARPTR) expires.data(), expires.size());
2140
2141 hmac.Final(dest);
2142
2143 buf_to_hex((UCHARPTR) dest, sizeof(dest), dest_str);
2144
2145 return dest_str;
2146 }
2147
2148 const char* get_signature() const {
2149 return dest_str;
2150 }
2151
2152 bool is_equal_to(const std::string& rhs) const {
2153 /* never allow out-of-range exception */
2154 if (rhs.size() < (output_size - 1)) {
2155 return false;
2156 }
2157 return rhs.compare(0 /* pos */, output_size, dest_str) == 0;
2158 }
