116int RGWSI_SysObj_Core::raw_stat(const DoutPrefixProvider *dpp, const rgw_raw_obj& obj, uint64_t *psize, real_time *pmtime, uint64_t *epoch,
117 map<string, bufferlist> *attrs, bufferlist *first_chunk,
118 RGWObjVersionTracker *objv_tracker,
119 optional_yield y)
120{
121 RGWSI_RADOS::Obj rados_obj;
122 int r = get_rados_obj(dpp, zone_svc, obj, &rados_obj);
123 if (r < 0) {
124 return r;
125 }
126
127 uint64_t size = 0;
128 struct timespec mtime_ts;
129
130 librados::ObjectReadOperation op;
131 if (objv_tracker) {
132 objv_tracker->prepare_op_for_read(&op);
133 }
134 op.getxattrs(attrs, nullptr);
135 if (psize || pmtime) {
136 op.stat2(&size, &mtime_ts, nullptr);
137 }
138 if (first_chunk) {
139 op.read(0, cct->_conf->rgw_max_chunk_size, first_chunk, nullptr);
140 }
141 bufferlist outbl;
142 r = rados_obj.operate(dpp, &op, &outbl, y);
143
144 if (epoch) {
145 *epoch = rados_obj.get_last_version();
146 }
147
148 if (r < 0)
149 return r;
150
151 if (psize)
152 *psize = size;
153 if (pmtime)
154 *pmtime = ceph::real_clock::from_timespec(mtime_ts);
155
156 return 0;
157}

159int RGWSI_SysObj_Core::stat(RGWSysObjectCtxBase& obj_ctx,
160 RGWSI_SysObj_Obj_GetObjState& _state,
161 const rgw_raw_obj& obj,
162 map<string, bufferlist> *attrs,
163 bool raw_attrs,
164 real_time *lastmod,
165 uint64_t *obj_size,
166 RGWObjVersionTracker *objv_tracker,

173 if (r < 0)
174 return r;
175
176 if (!astate->exists) {
177 return -ENOENT;
178 }
179
180 if (attrs) {
181 if (raw_attrs) {
182 *attrs = astate->attrset;
183 } else {
184 rgw_filter_attrset(astate->attrset, RGW_ATTR_PREFIX, attrs);

186 if (cct->_conf->subsys.should_gather<ceph_subsys_rgw, 20>()) {
187 map<string, bufferlist>::iterator iter;

189 ldpp_dout(dpp, 20) << "Read xattr: " << iter->first << dendl;

192 }
193
194 if (obj_size)
195 *obj_size = astate->size;
196 if (lastmod)
197 *lastmod = astate->mtime;
198
199 return 0;

202int RGWSI_SysObj_Core::read(const DoutPrefixProvider *dpp,
203 RGWSysObjectCtxBase& obj_ctx,
204 RGWSI_SysObj_Obj_GetObjState& _read_state,
205 RGWObjVersionTracker *objv_tracker,
206 const rgw_raw_obj& obj,
207 bufferlist *bl, off_t ofs, off_t end,
208 map<string, bufferlist> *attrs,
209 bool raw_attrs,
210 rgw_cache_entry_info *cache_info,
211 boost::optional<obj_version>,
212 optional_yield y)
213{
214 auto& read_state = static_cast<GetObjState&>(_read_state);
215
216 uint64_t len;
217 librados::ObjectReadOperation op;
218
219 if (end < 0)
220 len = 0;
221 else
222 len = end - ofs + 1;
223
224 if (objv_tracker) {
225 objv_tracker->prepare_op_for_read(&op);
226 }
227
228 ldpp_dout(dpp, 20) << "rados->read ofs=" << ofs << " len=" << len << dendl;
229 op.read(ofs, len, bl, nullptr);
230
231 map<string, bufferlist> unfiltered_attrset;
232
233 if (attrs) {
234 if (raw_attrs) {
235 op.getxattrs(attrs, nullptr);
236 } else {
237 op.getxattrs(&unfiltered_attrset, nullptr);
238 }
239 }
240
241 RGWSI_RADOS::Obj rados_obj;
242 int r = get_rados_obj(dpp, zone_svc, obj, &rados_obj);
243 if (r < 0) {
244 ldpp_dout(dpp, 20) << "get_rados_obj() on obj=" << obj << " returned " << r << dendl;
245 return r;
246 }
247 r = rados_obj.operate(dpp, &op, nullptr, y);
248 if (r < 0) {
249 ldpp_dout(dpp, 20) << "rados_obj.operate() r=" << r << " bl.length=" << bl->length() << dendl;
250 return r;
251 }
252 ldpp_dout(dpp, 20) << "rados_obj.operate() r=" << r << " bl.length=" << bl->length() << dendl;
253
254 uint64_t op_ver = rados_obj.get_last_version();
255
256 if (read_state.last_ver > 0 &&
257 read_state.last_ver != op_ver) {
258 ldpp_dout(dpp, 5) << "raced with an object write, abort" << dendl;
259 return -ECANCELED;
260 }
261
262 if (attrs && !raw_attrs) {
263 rgw_filter_attrset(unfiltered_attrset, RGW_ATTR_PREFIX, attrs);
264 }
265
266 read_state.last_ver = op_ver;
267
268 return bl->length();
269}

303int RGWSI_SysObj_Core::set_attrs(const DoutPrefixProvider *dpp,
304 const rgw_raw_obj& obj,
305 map<string, bufferlist>& attrs,
306 map<string, bufferlist> *rmattrs,
307 RGWObjVersionTracker *objv_tracker,
308 optional_yield y)
309{
310 RGWSI_RADOS::Obj rados_obj;
311 int r = get_rados_obj(dpp, zone_svc, obj, &rados_obj);
312 if (r < 0) {
313 ldpp_dout(dpp, 20) << "get_rados_obj() on obj=" << obj << " returned " << r << dendl;
314 return r;
315 }
316
317 librados::ObjectWriteOperation op;
318
319 if (objv_tracker) {
320 objv_tracker->prepare_op_for_write(&op);
321 }
322
323 map<string, bufferlist>::iterator iter;
324 if (rmattrs) {
325 for (iter = rmattrs->begin(); iter != rmattrs->end(); ++iter) {
326 const string& name = iter->first;
327 op.rmxattr(name.c_str());

332 const string& name = iter->first;
333 bufferlist& bl = iter->second;
334
335 if (!bl.length())
336 continue;
337
338 op.setxattr(name.c_str(), bl);
339 }
340
341 if (!op.size())
342 return 0;
343
344 bufferlist bl;
345
346 r = rados_obj.operate(dpp, &op, y);
347 if (r < 0)
348 return r;
349
350 if (objv_tracker) {
351 objv_tracker->apply_write();
352 }

580 bufferlist acl_bl;
581
582 for (map<string, bufferlist>::iterator iter = attrs.begin(); iter != attrs.end(); ++iter) {
583 const string& name = iter->first;
584 bufferlist& bl = iter->second;
585
586 if (!bl.length())

589 op.setxattr(name.c_str(), bl);

646{
