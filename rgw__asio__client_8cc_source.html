27int ClientIO::init_env(CephContext *cct)
28{
29 env.init(cct);
30
31 perfcounter->inc(l_rgw_qlen);
32 perfcounter->inc(l_rgw_qactive);
33
34 const auto& request = parser.get();
35 const auto& headers = request;
36 for (auto header = headers.begin(); header != headers.end(); ++header) {
37 const auto& field = header->name(); // enum type for known headers
38 const auto& name = header->name_string();
39 const auto& value = header->value();
40
41 if (field == beast::http::field::content_length) {
42 env.set("CONTENT_LENGTH", value.to_string());
43 continue;
44 }
45 if (field == beast::http::field::content_type) {
46 env.set("CONTENT_TYPE", value.to_string());
47 continue;
48 }
49
50 static const std::string_view HTTP_{"HTTP_"};
51
52 char buf[name.size() + HTTP_.size() + 1];
53 auto dest = std::copy(std::begin(HTTP_), std::end(HTTP_), buf);
54 for (auto src = name.begin(); src != name.end(); ++src, ++dest) {
55 if (*src == '-') {
56 *dest = '_';
57 } else if (*src == '_') {
58 *dest = '-';
59 } else {
60 *dest = std::toupper(*src);
61 }
62 }
63 *dest = '\0';
64
65 env.set(buf, value.to_string());
66 }
67
68 int major = request.version() / 10;
69 int minor = request.version() % 10;
70 env.set("HTTP_VERSION", std::to_string(major) + '.' + std::to_string(minor));
71
72 env.set("REQUEST_METHOD", request.method_string().to_string());
73
74 // split uri from query
75 auto uri = request.target();
76 auto pos = uri.find('?');
77 if (pos != uri.npos) {
78 auto query = uri.substr(pos + 1);
79 env.set("QUERY_STRING", query.to_string());
80 uri = uri.substr(0, pos);
81 }
82 env.set("SCRIPT_URI", uri.to_string());
83
84 env.set("REQUEST_URI", request.target().to_string());
85
86 char port_buf[16];
87 snprintf(port_buf, sizeof(port_buf), "%d", local_endpoint.port());
88 env.set("SERVER_PORT", port_buf);
89 if (is_ssl) {
90 env.set("SERVER_PORT_SECURE", port_buf);
91 }
92 env.set("REMOTE_ADDR", remote_endpoint.address().to_string());
93 // TODO: set REMOTE_USER if authenticated
94 return 0;
95}

130static constexpr size_t TIME_BUF_SIZE = 128;
131static size_t dump_date_header(char (&timestr)[TIME_BUF_SIZE])
132{
133 const time_t gtime = time(nullptr);
134 struct tm result;
135 struct tm const * const tmp = gmtime_r(&gtime, &result);
136 if (tmp == nullptr) {
137 return 0;
138 }
139 return strftime(timestr, sizeof(timestr),
140 "Date: %a, %d %b %Y %H:%M:%S %Z\r\n", tmp);
141}

143size_t ClientIO::complete_header()
144{
145 size_t sent = 0;
146
147 char timestr[TIME_BUF_SIZE];
148 if (dump_date_header(timestr)) {
149 sent += txbuf.sputn(timestr, strlen(timestr));
150 }
151
152 if (parser.keep_alive()) {
153 constexpr char CONN_KEEP_ALIVE[] = "Connection: Keep-Alive\r\n";
154 sent += txbuf.sputn(CONN_KEEP_ALIVE, sizeof(CONN_KEEP_ALIVE) - 1);
155 } else {
156 constexpr char CONN_KEEP_CLOSE[] = "Connection: close\r\n";
157 sent += txbuf.sputn(CONN_KEEP_CLOSE, sizeof(CONN_KEEP_CLOSE) - 1);
158 }
159
160 constexpr char HEADER_END[] = "\r\n";
161 sent += txbuf.sputn(HEADER_END, sizeof(HEADER_END) - 1);
162
163 flush();
164 return sent;
165}

167size_t ClientIO::send_header(const std::string_view& name,

169{

171 static constexpr char HEADER_END[] = "\r\n";
172
173 size_t sent = 0;
174
175 sent += txbuf.sputn(name.data(), name.length());
176 sent += txbuf.sputn(HEADER_SEP, sizeof(HEADER_SEP) - 1);
177 sent += txbuf.sputn(value.data(), value.length());
178 sent += txbuf.sputn(HEADER_END, sizeof(HEADER_END) - 1);
179
180 return sent;
