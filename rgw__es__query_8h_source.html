20 void assign(std::list<std::string>& src) {
21 l.swap(src);
22 iter = l.begin();
23 }
24
25 bool peek(std::string *dest) {
26 if (done()) {
27 return false;
28 }
29 *dest = *iter;
30 return true;
31 }
32
33 bool pop(std::string *dest) {
34 bool valid = peek(dest);
35 if (!valid) {
36 return false;
37 }
38 ++iter;
39 return true;
40 }
41
42 bool done() {
43 return (iter == l.end());
44 }
45};
