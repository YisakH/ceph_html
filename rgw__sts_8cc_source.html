45int Credentials::generateCredentials(const DoutPrefixProvider *dpp,
46 CephContext* cct,
47 const uint64_t& duration,
48 const boost::optional<std::string>& policy,
49 const boost::optional<std::string>& roleId,
50 const boost::optional<std::string>& role_session,
51 const boost::optional<std::vector<std::string>>& token_claims,
52 const boost::optional<std::vector<std::pair<std::string,std::string>>>& session_princ_tags,
53 boost::optional<rgw_user> user,
54 rgw::auth::Identity* identity)
55{
56 uuid_d accessKey, secretKey;
57 char accessKeyId_str[MAX_ACCESS_KEY_LEN], secretAccessKey_str[MAX_SECRET_KEY_LEN];
58
59 //AccessKeyId
60 gen_rand_alphanumeric_plain(cct, accessKeyId_str, sizeof(accessKeyId_str));
61 accessKeyId = accessKeyId_str;
62
63 //SecretAccessKey
64 gen_rand_alphanumeric_upper(cct, secretAccessKey_str, sizeof(secretAccessKey_str));
65 secretAccessKey = secretAccessKey_str;
66
67 //Expiration
68 real_clock::time_point t = real_clock::now();
69 real_clock::time_point exp = t + std::chrono::seconds(duration);
70 expiration = ceph::to_iso_8601(exp);
71
72 //Session Token - Encrypt using AES
73 auto* cryptohandler = cct->get_crypto_handler(CEPH_CRYPTO_AES);
74 if (! cryptohandler) {
75 ldpp_dout(dpp, 0) << "ERROR: No AES cryto handler found !" << dendl;
76 return -EINVAL;
77 }
78 string secret_s = cct->_conf->rgw_sts_key;
79 buffer::ptr secret(secret_s.c_str(), secret_s.length());
80 int ret = 0;
81 if (ret = cryptohandler->validate_secret(secret); ret < 0) {
82 ldpp_dout(dpp, 0) << "ERROR: Invalid rgw sts key, please ensure its length is 16" << dendl;
83 return ret;
84 }
85 string error;
86 std::unique_ptr<CryptoKeyHandler> keyhandler(cryptohandler->get_key_handler(secret, error));
87 if (! keyhandler) {
88 ldpp_dout(dpp, 0) << "ERROR: No Key handler found !" << dendl;
89 return -EINVAL;
90 }
91 error.clear();
92 //Storing policy and roleId as part of token, so that they can be extracted
93 // from the token itself for policy evaluation.
94 SessionToken token;
95 //authentication info
96 token.access_key_id = accessKeyId;
97 token.secret_access_key = secretAccessKey;
98 token.expiration = expiration;
99 token.issued_at = ceph::to_iso_8601(t);
100
101 //Authorization info
102 if (policy)
103 token.policy = *policy;
104 else
105 token.policy = {};
106
107 if (roleId)
108 token.roleId = *roleId;
109 else
110 token.roleId = {};
111
112 if (user)
113 token.user = *user;
114 else {
115 rgw_user u({}, {}, {});
116 token.user = u;
117 }
118
119 if (token_claims) {
120 token.token_claims = std::move(*token_claims);
121 }
122
123 if (identity) {
124 token.acct_name = identity->get_acct_name();
125 token.perm_mask = identity->get_perm_mask();
126 token.is_admin = identity->is_admin_of(token.user);
127 token.acct_type = identity->get_identity_type();
128 } else {
129 token.acct_name = {};
130 token.perm_mask = 0;
131 token.is_admin = 0;
132 token.acct_type = TYPE_ROLE;
133 token.role_session = role_session.get();
134 }
135
136 if (session_princ_tags) {
137 token.principal_tags = std::move(*session_princ_tags);
138 }
139 buffer::list input, enc_output;
140 encode(token, input);
141
142 if (ret = keyhandler->encrypt(input, enc_output, &error); ret < 0) {
143 ldpp_dout(dpp, 0) << "ERROR: Encrypting session token returned an error !" << dendl;
144 return ret;
145 }
146
147 bufferlist encoded_op;
148 enc_output.encode_base64(encoded_op);
149 encoded_op.append('\0');
150 sessionToken = encoded_op.c_str();
151
152 return ret;
153}

198int AssumeRoleRequestBase::validate_input(const DoutPrefixProvider *dpp) const
199{
200 if (!err_msg.empty()) {
201 ldpp_dout(dpp, 0) << "ERROR: error message is empty !" << dendl;
202 return -EINVAL;
203 }
204
205 if (duration < MIN_DURATION_IN_SECS ||
206 duration > MAX_DURATION_IN_SECS) {
207 ldpp_dout(dpp, 0) << "ERROR: Incorrect value of duration: " << duration << dendl;
208 return -EINVAL;
209 }
210
211 if (! iamPolicy.empty() &&
212 (iamPolicy.size() < MIN_POLICY_SIZE || iamPolicy.size() > MAX_POLICY_SIZE)) {
213 ldpp_dout(dpp, 0) << "ERROR: Incorrect size of iamPolicy: " << iamPolicy.size() << dendl;
214 return -ERR_PACKED_POLICY_TOO_LARGE;
215 }
216
217 if (! roleArn.empty() &&
218 (roleArn.size() < MIN_ROLE_ARN_SIZE || roleArn.size() > MAX_ROLE_ARN_SIZE)) {
219 ldpp_dout(dpp, 0) << "ERROR: Incorrect size of roleArn: " << roleArn.size() << dendl;
220 return -EINVAL;
221 }
222
223 if (! roleSessionName.empty()) {
224 if (roleSessionName.size() < MIN_ROLE_SESSION_SIZE || roleSessionName.size() > MAX_ROLE_SESSION_SIZE) {
225 ldpp_dout(dpp, 0) << "ERROR: Either role session name is empty or role session size is incorrect: " << roleSessionName.size() << dendl;
226 return -EINVAL;
227 }
228
229 std::regex regex_roleSession("[A-Za-z0-9_=,.@-]+");
230 if (! std::regex_match(roleSessionName, regex_roleSession)) {
231 ldpp_dout(dpp, 0) << "ERROR: Role session name is incorrect: " << roleSessionName << dendl;
232 return -EINVAL;
233 }
234 }
235
236 return 0;
237}

286std::tuple<int, rgw::sal::RGWRole*> STSService::getRoleInfo(const DoutPrefixProvider *dpp,
287 const string& arn,
288 optional_yield y)
289{
290 if (auto r_arn = rgw::ARN::parse(arn); r_arn) {
291 auto pos = r_arn->resource.find_last_of('/');
292 string roleName = r_arn->resource.substr(pos + 1);
293 std::unique_ptr<rgw::sal::RGWRole> role = store->get_role(roleName, r_arn->account);
294 if (int ret = role->get(dpp, y); ret < 0) {
295 if (ret == -ENOENT) {
296 ldpp_dout(dpp, 0) << "Role doesn't exist: " << roleName << dendl;
297 ret = -ERR_NO_ROLE_FOUND;
298 }
299 return make_tuple(ret, nullptr);
300 } else {
301 auto path_pos = r_arn->resource.find('/');
302 string path;
303 if (path_pos == pos) {
304 path = "/";
305 } else {
306 path = r_arn->resource.substr(path_pos, ((pos - path_pos) + 1));
307 }
308 string r_path = role->get_path();
309 if (path != r_path) {
310 ldpp_dout(dpp, 0) << "Invalid Role ARN: Path in ARN does not match with the role path: " << path << " " << r_path << dendl;
311 return make_tuple(-EACCES, nullptr);
312 }
313 this->role = std::move(role);
314 return make_tuple(0, this->role.get());
315 }
316 } else {
317 ldpp_dout(dpp, 0) << "Invalid role arn: " << arn << dendl;
318 return make_tuple(-EINVAL, nullptr);
319 }
320}

339AssumeRoleWithWebIdentityResponse STSService::assumeRoleWithWebIdentity(const DoutPrefixProvider *dpp, AssumeRoleWithWebIdentityRequest& req)
340{
341 AssumeRoleWithWebIdentityResponse response;
342 response.assumeRoleResp.packedPolicySize = 0;
343 std::vector<string> token_claims;
344
345 if (req.getProviderId().empty()) {
346 response.providerId = req.getIss();
347 }
348 response.aud = req.getAud();
349 response.sub = req.getSub();
350
351 token_claims.emplace_back(string("iss") + ":" + req.getIss());
352 token_claims.emplace_back(string("aud") + ":" + req.getAud());
353 token_claims.emplace_back(string("sub") + ":" + req.getSub());
354
355 //Get the role info which is being assumed
356 boost::optional<rgw::ARN> r_arn = rgw::ARN::parse(req.getRoleARN());
357 if (r_arn == boost::none) {
358 ldpp_dout(dpp, 0) << "Error in parsing role arn: " << req.getRoleARN() << dendl;
359 response.assumeRoleResp.retCode = -EINVAL;
360 return response;
361 }
362
363 string roleId = role->get_id();
364 uint64_t roleMaxSessionDuration = role->get_max_session_duration();
365 req.setMaxDuration(roleMaxSessionDuration);
366
367 //Validate input
368 response.assumeRoleResp.retCode = req.validate_input(dpp);
369 if (response.assumeRoleResp.retCode < 0) {
370 return response;
371 }
372
373 //Calculate PackedPolicySize
374 string policy = req.getPolicy();
375 response.assumeRoleResp.packedPolicySize = (policy.size() / req.getMaxPolicySize()) * 100;
376
377 //Generate Assumed Role User
378 response.assumeRoleResp.retCode = response.assumeRoleResp.user.generateAssumedRoleUser(cct,
379 store,
380 roleId,
381 r_arn.get(),
382 req.getRoleSessionName());
383 if (response.assumeRoleResp.retCode < 0) {
384 return response;
385 }
386
387 //Generate Credentials
388 //Role and Policy provide the authorization info, user id and applier info are not needed
389 response.assumeRoleResp.retCode = response.assumeRoleResp.creds.generateCredentials(dpp, cct, req.getDuration(),
390 req.getPolicy(), roleId,
391 req.getRoleSessionName(),
392 token_claims,
393 req.getPrincipalTags(),
394 user_id, nullptr);
395 if (response.assumeRoleResp.retCode < 0) {
396 return response;
397 }
398
399 response.assumeRoleResp.retCode = 0;
400 return response;
401}
