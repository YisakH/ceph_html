42void RGWOp_MDLog_List::execute(optional_yield y) {
43 string period = s->info.args.get("period");
44 string shard = s->info.args.get("id");
45 string max_entries_str = s->info.args.get("max-entries");
46 string marker = s->info.args.get("marker"),
47 err;
48 void *handle;
49 unsigned shard_id, max_entries = LOG_CLASS_LIST_MAX_ENTRIES;
50
51 if (s->info.args.exists("start-time") ||
52 s->info.args.exists("end-time")) {
53 ldpp_dout(this, 5) << "start-time and end-time are no longer accepted" << dendl;
54 op_ret = -EINVAL;
55 return;
56 }
57
58 shard_id = (unsigned)strict_strtol(shard.c_str(), 10, &err);
59 if (!err.empty()) {
60 ldpp_dout(this, 5) << "Error parsing shard_id " << shard << dendl;
61 op_ret = -EINVAL;
62 return;
63 }
64
65 if (!max_entries_str.empty()) {
66 max_entries = (unsigned)strict_strtol(max_entries_str.c_str(), 10, &err);
67 if (!err.empty()) {
68 ldpp_dout(this, 5) << "Error parsing max-entries " << max_entries_str << dendl;
69 op_ret = -EINVAL;
70 return;
71 }
72 if (max_entries > LOG_CLASS_LIST_MAX_ENTRIES) {
73 max_entries = LOG_CLASS_LIST_MAX_ENTRIES;
74 }
75 }
76
77 if (period.empty()) {
78 ldpp_dout(this, 5) << "Missing period id trying to use current" << dendl;
79 period = store->get_zone()->get_current_period_id();
80 if (period.empty()) {
81 ldpp_dout(this, 5) << "Missing period id" << dendl;
82 op_ret = -EINVAL;
83 return;
84 }
85 }
86
87 RGWMetadataLog meta_log{s->cct, static_cast<rgw::sal::RadosStore*>(store)->svc()->zone, static_cast<rgw::sal::RadosStore*>(store)->svc()->cls, period};
88
89 meta_log.init_list_entries(shard_id, {}, {}, marker, &handle);
90
91 op_ret = meta_log.list_entries(this, handle, max_entries, entries,
92 &last_marker, &truncated);
93
94 meta_log.complete_list_entries(handle);
95}

236void RGWOp_MDLog_Lock::execute(optional_yield y) {
237 string period, shard_id_str, duration_str, locker_id, zone_id;
238 unsigned shard_id;
239
240 op_ret = 0;
241
242 period = s->info.args.get("period");
243 shard_id_str = s->info.args.get("id");
244 duration_str = s->info.args.get("length");
245 locker_id = s->info.args.get("locker-id");
246 zone_id = s->info.args.get("zone-id");
247
248 if (period.empty()) {
249 ldpp_dout(this, 5) << "Missing period id trying to use current" << dendl;
250 period = store->get_zone()->get_current_period_id();
251 }
252
253 if (period.empty() ||
254 shard_id_str.empty() ||
255 (duration_str.empty()) ||
256 locker_id.empty() ||
257 zone_id.empty()) {
258 ldpp_dout(this, 5) << "Error invalid parameter list" << dendl;
259 op_ret = -EINVAL;
260 return;
261 }
262
263 string err;
264 shard_id = (unsigned)strict_strtol(shard_id_str.c_str(), 10, &err);
265 if (!err.empty()) {
266 ldpp_dout(this, 5) << "Error parsing shard_id param " << shard_id_str << dendl;
267 op_ret = -EINVAL;
268 return;
269 }
270
271 RGWMetadataLog meta_log{s->cct, static_cast<rgw::sal::RadosStore*>(store)->svc()->zone, static_cast<rgw::sal::RadosStore*>(store)->svc()->cls, period};
272 unsigned dur;
273 dur = (unsigned)strict_strtol(duration_str.c_str(), 10, &err);
274 if (!err.empty() || dur <= 0) {
275 ldpp_dout(this, 5) << "invalid length param " << duration_str << dendl;
276 op_ret = -EINVAL;
277 return;
278 }
279 op_ret = meta_log.lock_exclusive(s, shard_id, make_timespan(dur), zone_id,
280 locker_id);
281 if (op_ret == -EBUSY)
282 op_ret = -ERR_LOCKED;

322void RGWOp_MDLog_Notify::execute(optional_yield y) {
323#define LARGE_ENOUGH_BUF (128 * 1024)
324
325 int r = 0;
326 bufferlist data;
327 std::tie(r, data) = read_all_input(s, LARGE_ENOUGH_BUF);
328 if (r < 0) {
329 op_ret = r;
330 return;
331 }
332
333 char* buf = data.c_str();
334 ldpp_dout(this, 20) << __func__ << "(): read data: " << buf << dendl;
335
336 JSONParser p;
337 r = p.parse(buf, data.length());
338 if (r < 0) {
339 ldpp_dout(this, 0) << "ERROR: failed to parse JSON" << dendl;
340 op_ret = r;
341 return;
342 }
343
344 set<int> updated_shards;
345 try {
346 decode_json_obj(updated_shards, &p);
347 } catch (JSONDecoder::err& err) {
348 ldpp_dout(this, 0) << "ERROR: failed to decode JSON" << dendl;
349 op_ret = -EINVAL;
350 return;
351 }
352
353 if (store->ctx()->_conf->subsys.should_gather<ceph_subsys_rgw, 20>()) {
354 for (set<int>::iterator iter = updated_shards.begin(); iter != updated_shards.end(); ++iter) {
355 ldpp_dout(this, 20) << __func__ << "(): updated shard=" << *iter << dendl;
356 }
357 }
358
359 store->wakeup_meta_sync_shards(updated_shards);
360
361 op_ret = 0;
362}

364void RGWOp_BILog_List::execute(optional_yield y) {
365 string tenant_name = s->info.args.get("tenant"),
366 bucket_name = s->info.args.get("bucket"),
367 marker = s->info.args.get("marker"),
368 max_entries_str = s->info.args.get("max-entries"),
369 bucket_instance = s->info.args.get("bucket-instance");
370 std::unique_ptr<rgw::sal::Bucket> bucket;
371 rgw_bucket b(rgw_bucket_key(tenant_name, bucket_name));
372 unsigned max_entries;
373
374 if (bucket_name.empty() && bucket_instance.empty()) {
375 ldpp_dout(this, 5) << "ERROR: neither bucket nor bucket instance specified" << dendl;
376 op_ret = -EINVAL;
377 return;
378 }
379
380 int shard_id;
381 string bn;
382 op_ret = rgw_bucket_parse_bucket_instance(bucket_instance, &bn, &bucket_instance, &shard_id);
383 if (op_ret < 0) {
384 return;
385 }
386
387 if (!bucket_instance.empty()) {
388 b.name = bn;
389 b.bucket_id = bucket_instance;
390 }
391 op_ret = store->get_bucket(s, nullptr, b, &bucket, y);
392 if (op_ret < 0) {
393 ldpp_dout(this, 5) << "could not get bucket info for bucket=" << bucket_name << dendl;
394 return;
395 }
396
397 bool truncated;
398 unsigned count = 0;
399 string err;
400
401 max_entries = (unsigned)strict_strtol(max_entries_str.c_str(), 10, &err);
402 if (!err.empty())
403 max_entries = LOG_CLASS_LIST_MAX_ENTRIES;
404
405 send_response();
406 do {
407 list<rgw_bi_log_entry> entries;
408 int ret = static_cast<rgw::sal::RadosStore*>(store)->svc()->bilog_rados->log_list(s, bucket->get_info(), shard_id,
409 marker, max_entries - count,
410 entries, &truncated);
411 if (ret < 0) {
412 ldpp_dout(this, 5) << "ERROR: list_bi_log_entries()" << dendl;
413 return;
414 }
415
416 count += entries.size();
417
418 send_response(entries, marker);
419 } while (truncated && count < max_entries);
420
421 send_response_end();
422}

806void RGWOp_BILog_Status::execute(optional_yield y)
807{
808 const auto options = s->info.args.get("options");
809 bool merge = (options == "merge");
810 const auto source_zone = s->info.args.get("source-zone");
811 const auto source_key = s->info.args.get("source-bucket");
812 auto key = s->info.args.get("bucket");
813 if (key.empty()) {
814 key = source_key;
815 }
816 if (key.empty()) {
817 ldpp_dout(this, 4) << "no 'bucket' provided" << dendl;
818 op_ret = -EINVAL;
819 return;
820 }
821
822 rgw_bucket b;
823 int shard_id{-1}; // unused
824 op_ret = rgw_bucket_parse_bucket_key(s->cct, key, &b, &shard_id);
825 if (op_ret < 0) {
826 ldpp_dout(this, 4) << "invalid 'bucket' provided" << dendl;
827 op_ret = -EINVAL;
828 return;
829 }
830
831 // read the bucket instance info for num_shards
832 std::unique_ptr<rgw::sal::Bucket> bucket;
833 op_ret = store->get_bucket(s, nullptr, b, &bucket, y);
834 if (op_ret < 0) {
835 ldpp_dout(this, 4) << "failed to read bucket info: " << cpp_strerror(op_ret) << dendl;
836 return;
837 }
838
839 rgw_bucket source_bucket;
840
841 if (source_key.empty() ||
842 source_key == key) {
843 source_bucket = bucket->get_key();
844 } else {
845 op_ret = rgw_bucket_parse_bucket_key(s->cct, source_key, &source_bucket, nullptr);
846 if (op_ret < 0) {
847 ldpp_dout(this, 4) << "invalid 'source-bucket' provided (key=" << source_key << ")" << dendl;
848 return;
849 }
850 }
851
852 const auto& local_zone_id = store->get_zone()->get_id();
853
854 if (!merge) {
855 rgw_sync_bucket_pipe pipe;
856 pipe.source.zone = source_zone;
857 pipe.source.bucket = source_bucket;
858 pipe.dest.zone = local_zone_id;
859 pipe.dest.bucket = bucket->get_key();
860
861 ldpp_dout(this, 20) << "RGWOp_BILog_Status::execute(optional_yield y): getting sync status for pipe=" << pipe << dendl;
862
863 op_ret = rgw_bucket_sync_status(this, static_cast<rgw::sal::RadosStore*>(store), pipe, bucket->get_info(), nullptr, &status);
864
865 if (op_ret < 0) {
866 ldpp_dout(this, -1) << "ERROR: rgw_bucket_sync_status() on pipe=" << pipe << " returned ret=" << op_ret << dendl;
867 }
868 return;
869 }
870
871 rgw_zone_id source_zone_id(source_zone);
872
873 RGWBucketSyncPolicyHandlerRef source_handler;
874 op_ret = store->get_sync_policy_handler(s, source_zone_id, source_bucket, &source_handler, y);
875 if (op_ret < 0) {
876 ldpp_dout(this, -1) << "could not get bucket sync policy handler (r=" << op_ret << ")" << dendl;
877 return;
878 }
879
880 auto local_dests = source_handler->get_all_dests_in_zone(local_zone_id);
881
882 std::vector<rgw_bucket_shard_sync_info> current_status;
883 for (auto& entry : local_dests) {
884 auto pipe = entry.second;
885
886 ldpp_dout(this, 20) << "RGWOp_BILog_Status::execute(optional_yield y): getting sync status for pipe=" << pipe << dendl;
887
888 RGWBucketInfo *pinfo = &bucket->get_info();
889 std::optional<RGWBucketInfo> opt_dest_info;
890
891 if (!pipe.dest.bucket) {
892 /* Uh oh, something went wrong */
893 ldpp_dout(this, 20) << "ERROR: RGWOp_BILog_Status::execute(optional_yield y): BUG: pipe.dest.bucket was not initialized" << pipe << dendl;
894 op_ret = -EIO;
895 return;
896 }
897
898 if (*pipe.dest.bucket != pinfo->bucket) {
899 opt_dest_info.emplace();
900 std::unique_ptr<rgw::sal::Bucket> dest_bucket;
901 op_ret = store->get_bucket(s, nullptr, *pipe.dest.bucket, &dest_bucket, y);
902 if (op_ret < 0) {
903 ldpp_dout(this, 4) << "failed to read target bucket info (bucket=: " << cpp_strerror(op_ret) << dendl;
904 return;
905 }
906
907 *opt_dest_info = dest_bucket->get_info();
908 pinfo = &(*opt_dest_info);
909 pipe.dest.bucket = pinfo->bucket;
910 }
911
912 int r = rgw_bucket_sync_status(this, static_cast<rgw::sal::RadosStore*>(store), pipe, *pinfo, &bucket->get_info(), &current_status);
913 if (r < 0) {
914 ldpp_dout(this, -1) << "ERROR: rgw_bucket_sync_status() on pipe=" << pipe << " returned ret=" << r << dendl;
915 op_ret = r;
916 return;
917 }
918
919 if (status.empty()) {
920 status = std::move(current_status);
921 } else {
922 if (current_status.size() !=
923 status.size()) {
924 op_ret = -EINVAL;
925 ldpp_dout(this, -1) << "ERROR: different number of shards for sync status of buckets syncing from the same source: status.size()= " << status.size() << " current_status.size()=" << current_status.size() << dendl;
926 return;
927 }
928 auto m = status.begin();
929 for (auto& cur_shard_status : current_status) {
930 auto& result_shard_status = *m++;
931 // always take the first marker, or any later marker that's smaller
932 if (cur_shard_status.inc_marker.position < result_shard_status.inc_marker.position) {
933 result_shard_status = std::move(cur_shard_status);
934 }
935 }
936 }
937 }
938}
