42 static enum token_type to_type(const string& s) {
43 if (boost::iequals(s, "ad"))
44 return TOKEN_AD;
45 if (boost::iequals(s, "ldap"))
46 return TOKEN_LDAP;
47 if (boost::iequals(s, "keystone"))
48 return TOKEN_KEYSTONE;
49 return TOKEN_NONE;
50 }

52 static const char* from_type(enum token_type type) {
53 switch (type) {
54 case TOKEN_AD:
55 return "ad";
56 case TOKEN_LDAP:
57 return "ldap";
58 case TOKEN_KEYSTONE:
59 return "keystone";
60 default:
61 return "none";
62 };
63 }

97 void encode(bufferlist& bl) const {
98 uint32_t ver = version();
99 string typestr{from_type(type)};
100 ENCODE_START(1, 1, bl);
101 encode(type_name, bl);
102 encode(ver, bl);
103 encode(typestr, bl);
104 encode(id, bl);
105 encode(key, bl);
106 ENCODE_FINISH(bl);
107 }

109 void decode(bufferlist::const_iterator& bl) {
110 string name;
111 string typestr;
112 uint32_t version;
113 DECODE_START(1, bl);
114 decode(name, bl);
115 decode(version, bl);
116 decode(typestr, bl);
117 type = to_type(typestr);
118 decode(id, bl);
119 decode(key, bl);
120 DECODE_FINISH(bl);
121 }

130 void encode_json(Formatter* f) {
131 RGWToken& token = *this;
132 f->open_object_section(type_name);
133 ::encode_json(type_name, token, f);
134 f->close_section();
135 }

137 void decode_json(JSONObj* obj) {
138 uint32_t version;
139 string type_name;
140 string typestr;
141 JSONDecoder::decode_json("version", version, obj);
142 JSONDecoder::decode_json("type", typestr, obj);
143 type = to_type(typestr);
144 JSONDecoder::decode_json("id", id, obj);
145 JSONDecoder::decode_json("key", key, obj);
146 }

148 std::string encode_json_base64(Formatter* f) {
149 encode_json(f);
150 std::ostringstream os;
151 f->flush(os);
152 return to_base64(std::move(os.str()));
153 }
