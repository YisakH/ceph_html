47 void init(const std::string& _oid, const std::string& _upload_id, const std::string& part_unique_str) {
48 if (_oid.empty()) {
49 clear();
50 return;
51 }

53 upload_id = _upload_id;
54 prefix = oid + ".";
55 meta = prefix + upload_id + MP_META_SUFFIX;
56 prefix.append(part_unique_str);
57 }

67 std::string s = prefix;
68 s.append(".");
69 s.append(part);

76 return oid;
77 }
78 bool from_meta(const std::string& meta) {
79 int end_pos = meta.rfind('.'); // search for ".meta"
80 if (end_pos < 0)
81 return false;
82 int mid_pos = meta.rfind('.', end_pos - 1); // <key>.<upload_id>
83 if (mid_pos < 0)
84 return false;
85 oid = meta.substr(0, mid_pos);
86 upload_id = meta.substr(mid_pos + 1, end_pos - mid_pos - 1);
87 init(oid, upload_id, upload_id);
88 return true;
89 }
90 void clear() {
91 oid = "";
92 prefix = "";
93 meta = "";
94 upload_id = "";
95 }

98 ", meta=" << std::quoted(obj.meta) << " }";
99 }
