69static inline void prepend_bucket_marker(const rgw_bucket& bucket, const std::string& orig_oid, std::string& oid)
70{
71 if (bucket.marker.empty() || orig_oid.empty()) {
72 oid = orig_oid;
73 } else {
74 oid = bucket.marker;
75 oid.append("_");
76 oid.append(orig_oid);
77 }
78}

80static inline void get_obj_bucket_and_oid_loc(const rgw_obj& obj, std::string& oid, std::string& locator)
81{
82 const rgw_bucket& bucket = obj.bucket;
83 prepend_bucket_marker(bucket, obj.get_oid(), oid);
84 const std::string& loc = obj.key.get_loc();
85 if (!loc.empty()) {
86 prepend_bucket_marker(bucket, loc, locator);
87 } else {
88 locator.clear();
89 }
90}
