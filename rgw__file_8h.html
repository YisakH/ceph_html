1085 /* find or create an RGWFileHandle */
1086 LookupFHResult lookup_fh(RGWFileHandle* parent, const char *name,
1087 const uint32_t flags = RGWFileHandle::FLAG_NONE) {
1088 using std::get;
1089
1090 // cast int32_t(RGWFileHandle::FLAG_NONE) due to strictness of Clang
1091 // the cast transfers a lvalue into a rvalue in the ctor
1092 // check the commit message for the full details
1093 LookupFHResult fhr { nullptr, uint32_t(RGWFileHandle::FLAG_NONE) };
1094
1095 /* mount is stale? */
1096 if (state.flags & FLAG_CLOSED)
1097 return fhr;
1098
1099 RGWFileHandle::FHCache::Latch lat;
1100 bool fh_locked = flags & RGWFileHandle::FLAG_LOCKED;
1101
1102 std::string obj_name{name};
1103 std::string key_name{parent->make_key_name(name)};
1104 fh_key fhk = parent->make_fhk(obj_name);
1105
1106 lsubdout(get_context(), rgw, 10)
1107 << __func__ << " called on "
1108 << parent->object_name() << " for " << key_name
1109 << " (" << obj_name << ")"
1110 << " -> " << fhk
1111 << dendl;
1112
1113 retry:
1114 RGWFileHandle* fh =
1115 fh_cache.find_latch(fhk.fh_hk.object /* partition selector*/,
1116 fhk /* key */, lat /* serializer */,
1117 RGWFileHandle::FHCache::FLAG_LOCK);
1118 /* LATCHED */
1119 if (fh) {
1120 if (likely(! fh_locked))
1121 fh->mtx.lock(); // XXX !RAII because may-return-LOCKED
1122 if (fh->flags & RGWFileHandle::FLAG_DELETED) {
1123 /* for now, delay briefly and retry */
1124 lat.lock->unlock();
1125 if (likely(! fh_locked))
1126 fh->mtx.unlock();
1127 std::this_thread::sleep_for(std::chrono::milliseconds(20));
1128 goto retry; /* !LATCHED */
1129 }
1130 /* need initial ref from LRU (fast path) */
1131 if (! fh_lru.ref(fh, cohort::lru::FLAG_INITIAL)) {
1132 lat.lock->unlock();
1133 if (likely(! fh_locked))
1134 fh->mtx.unlock();
1135 goto retry; /* !LATCHED */
1136 }
1137 /* LATCHED, LOCKED */
1138 if (! (flags & RGWFileHandle::FLAG_LOCK))
1139 if (likely(! fh_locked))
1140 fh->mtx.unlock(); /* ! LOCKED */
1141 } else {
1142 /* make or re-use handle */
1143 RGWFileHandle::Factory prototype(this, parent, fhk,
1144 obj_name, CREATE_FLAGS(flags));
1145 uint32_t iflags{cohort::lru::FLAG_INITIAL};
1146 fh = static_cast<RGWFileHandle*>(
1147 fh_lru.insert(&prototype,
1148 cohort::lru::Edge::MRU,
1149 iflags));
1150 if (fh) {
1151 /* lock fh (LATCHED) */
1152 if (flags & RGWFileHandle::FLAG_LOCK)
1153 fh->mtx.lock();
1154 if (likely(! (iflags & cohort::lru::FLAG_RECYCLE))) {
1155 /* inserts at cached insert iterator, releasing latch */
1156 fh_cache.insert_latched(
1157 fh, lat, RGWFileHandle::FHCache::FLAG_UNLOCK);
1158 } else {
1159 /* recycle step invalidates Latch */
1160 fh_cache.insert(
1161 fhk.fh_hk.object, fh, RGWFileHandle::FHCache::FLAG_NONE);
1162 lat.lock->unlock(); /* !LATCHED */
1163 }
1164 get<1>(fhr) |= RGWFileHandle::FLAG_CREATE;
1165 /* ref parent (non-initial ref cannot fail on valid object) */
1166 if (! parent->is_mount()) {
1167 (void) fh_lru.ref(parent, cohort::lru::FLAG_NONE);
1168 }
1169 goto out; /* !LATCHED */
1170 } else {
1171 lat.lock->unlock();
1172 goto retry; /* !LATCHED */
1173 }
1174 }
1175 lat.lock->unlock(); /* !LATCHED */
1176 out:
1177 get<0>(fhr) = fh;
1178 if (fh) {
1179 lsubdout(get_context(), rgw, 17)
1180 << __func__ << " 2 " << *fh
1181 << dendl;
1182 }
1183 return fhr;
1184 } /* lookup_fh(RGWFileHandle*, const char *, const uint32_t) */
