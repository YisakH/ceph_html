122template <typename CompletionToken>
123auto AsyncScheduler::async_request(const client_id& client,
124 const ReqParams& params,
125 const Time& time, Cost cost,
126 CompletionToken&& token)
127{
128 boost::asio::async_completion<CompletionToken, Signature> init(token);
129
130 auto ex1 = get_executor();
131 auto& handler = init.completion_handler;
132
133 // allocate the Request and add it to the queue
134 auto completion = Completion::create(ex1, std::move(handler),
135 Request{client, time, cost});
136 // cast to unique_ptr<Request>
137 auto req = RequestRef{std::move(completion)};
138 int r = queue.add_request(std::move(req), client, params, time, cost);
139 if (r == 0) {
140 // schedule an immediate call to process() on the executor
141 schedule(crimson::dmclock::TimeZero);
142 if (auto c = counters(client)) {
143 c->inc(queue_counters::l_qlen);
144 c->inc(queue_counters::l_cost, cost);
145 }
146 } else {
147 // post the error code
148 boost::system::error_code ec(r, boost::system::system_category());
149 // cast back to Completion
150 auto completion = static_cast<Completion*>(req.release());
151 async::post(std::unique_ptr<Completion>{completion},
152 ec, PhaseType::priority);
153 if (auto c = counters(client)) {
154 c->inc(queue_counters::l_limit);
155 c->inc(queue_counters::l_limit_cost, cost);
156 }
157 }
158
159 return init.result.get();
160}

162class SimpleThrottler : public md_config_obs_t, public dmclock::Scheduler {
163public:
164 SimpleThrottler(CephContext *cct) :
165 max_requests(cct->_conf.get_val<int64_t>("rgw_max_concurrent_requests")),
166 counters(cct, "simple-throttler")
167 {
168 if (max_requests <= 0) {
169 max_requests = std::numeric_limits<int64_t>::max();
170 }
171 cct->_conf.add_observer(this);
172 }
173
174 const char** get_tracked_conf_keys() const override {
175 static const char* keys[] = { "rgw_max_concurrent_requests", nullptr };
176 return keys;
177 }
178
179 void handle_conf_change(const ConfigProxy& conf,
180 const std::set<std::string>& changed) override
181 {
182 if (changed.count("rgw_max_concurrent_requests")) {
183 auto new_max = conf.get_val<int64_t>("rgw_max_concurrent_requests");
184 max_requests = new_max > 0 ? new_max : std::numeric_limits<int64_t>::max();
185 }
186 }
187
188 void request_complete() override {
189 --outstanding_requests;
190 if (auto c = counters();
191 c != nullptr) {
192 c->inc(throttle_counters::l_outstanding, -1);
193 }
194
195 }
196
197private:
198 int schedule_request_impl(const client_id&, const ReqParams&,
199 const Time&, const Cost&,
200 optional_yield) override {
201 if (outstanding_requests++ >= max_requests) {
202 if (auto c = counters();
203 c != nullptr) {
204 c->inc(throttle_counters::l_outstanding);
205 c->inc(throttle_counters::l_throttle);
206 }
207 return -EAGAIN;
208 }
209
210 return 0 ;
211 }
212
213 std::atomic<int64_t> max_requests;
214 std::atomic<int64_t> outstanding_requests = 0;
215 ThrottleCounters counters;
216};
