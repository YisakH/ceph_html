59std::string rgw_bucket::get_key(char tenant_delim, char id_delim, size_t reserve) const
60{
61 const size_t max_len = tenant.size() + sizeof(tenant_delim) +
62 name.size() + sizeof(id_delim) + bucket_id.size() + reserve;
63
64 std::string key;
65 key.reserve(max_len);
66 if (!tenant.empty() && tenant_delim) {
67 key.append(tenant);
68 key.append(1, tenant_delim);
69 }
70 key.append(name);
71 if (!bucket_id.empty() && id_delim) {
72 key.append(1, id_delim);
73 key.append(bucket_id);
74 }
75 return key;
76}
