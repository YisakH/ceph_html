62int RGWRestRole::parse_tags()
63{
64 vector<string> keys, vals;
65 auto val_map = s->info.args.get_params();
66 const regex pattern_key("Tags.member.([0-9]+).Key");
67 const regex pattern_value("Tags.member.([0-9]+).Value");
68 for (auto& v : val_map) {
69 string key_index="", value_index="";
70 for(sregex_iterator it = sregex_iterator(
71 v.first.begin(), v.first.end(), pattern_key);
72 it != sregex_iterator(); it++) {
73 smatch match;
74 match = *it;
75 key_index = match.str(1);
76 ldout(s->cct, 20) << "Key index: " << match.str(1) << dendl;
77 if (!key_index.empty()) {
78 int index = stoi(key_index);
79 auto pos = keys.begin() + (index-1);
80 keys.insert(pos, v.second);
81 }
82 }
83 for(sregex_iterator it = sregex_iterator(
84 v.first.begin(), v.first.end(), pattern_value);
85 it != sregex_iterator(); it++) {
86 smatch match;
87 match = *it;
88 value_index = match.str(1);
89 ldout(s->cct, 20) << "Value index: " << match.str(1) << dendl;
90 if (!value_index.empty()) {
91 int index = stoi(value_index);
92 auto pos = vals.begin() + (index-1);
93 vals.insert(pos, v.second);
94 }
95 }
96 }
97 if (keys.size() != vals.size()) {
98 ldout(s->cct, 0) << "No. of keys doesn't match with no. of values in tags" << dendl;
99 return -EINVAL;
100 }
101 for (size_t i = 0; i < keys.size(); i++) {
102 tags.emplace(keys[i], vals[i]);
103 ldout(s->cct, 0) << "Tag Key: " << keys[i] << " Tag Value is: " << vals[i] << dendl;
104 }
105 return 0;
106}

187void RGWCreateRole::execute(optional_yield y)
188{
189 op_ret = get_params();
190 if (op_ret < 0) {
191 return;
192 }
193 std::string user_tenant = s->user->get_tenant();
194 std::unique_ptr<rgw::sal::RGWRole> role = store->get_role(role_name,
195 user_tenant,
196 role_path,
197 trust_policy,
198 max_session_duration,
199 tags);
200 if (!user_tenant.empty() && role->get_tenant() != user_tenant) {
201 ldpp_dout(this, 20) << "ERROR: the tenant provided in the role name does not match with the tenant of the user creating the role"
202 << dendl;
203 op_ret = -EINVAL;
204 return;
205 }
206
207 std::string role_id;
208
209 if (!store->is_meta_master()) {
210 RGWXMLDecoder::XMLParser parser;
211 if (!parser.init()) {
212 ldpp_dout(this, 0) << "ERROR: failed to initialize xml parser" << dendl;
213 op_ret = -EINVAL;
214 return;
215 }
216
217 bufferlist data;
218 s->info.args.remove("RoleName");
219 s->info.args.remove("Path");
220 s->info.args.remove("AssumeRolePolicyDocument");
221 s->info.args.remove("MaxSessionDuration");
222 s->info.args.remove("Action");
223 s->info.args.remove("Version");
224 auto& val_map = s->info.args.get_params();
225 for (auto it = val_map.begin(); it!= val_map.end(); it++) {
226 if (it->first.find("Tags.member.") == 0) {
227 val_map.erase(it);
228 }
229 }
230
231 RGWUserInfo info = s->user->get_info();
232 const auto& it = info.access_keys.begin();
233 RGWAccessKey key;
234 if (it != info.access_keys.end()) {
235 key.id = it->first;
236 RGWAccessKey cred = it->second;
237 key.key = cred.key;
238 }
239 op_ret = store->forward_iam_request_to_master(s, key, nullptr, bl_post_body, &parser, s->info, y);
240 if (op_ret < 0) {
241 ldpp_dout(this, 20) << "ERROR: forward_iam_request_to_master failed with error code: " << op_ret << dendl;
242 return;
243 }
244
245 XMLObj* create_role_resp_obj = parser.find_first("CreateRoleResponse");;
246 if (!create_role_resp_obj) {
247 ldpp_dout(this, 5) << "ERROR: unexpected xml: CreateRoleResponse" << dendl;
248 op_ret = -EINVAL;
249 return;
250 }
251
252 XMLObj* create_role_res_obj = create_role_resp_obj->find_first("CreateRoleResult");
253 if (!create_role_res_obj) {
254 ldpp_dout(this, 5) << "ERROR: unexpected xml: CreateRoleResult" << dendl;
255 op_ret = -EINVAL;
256 return;
257 }
258
259 XMLObj* role_obj = nullptr;
260 if (create_role_res_obj) {
261 role_obj = create_role_res_obj->find_first("Role");
262 }
263 if (!role_obj) {
264 ldpp_dout(this, 5) << "ERROR: unexpected xml: Role" << dendl;
265 op_ret = -EINVAL;
266 return;
267 }
268
269 try {
270 if (role_obj) {
271 RGWXMLDecoder::decode_xml("RoleId", role_id, role_obj, true);
272 }
273 } catch (RGWXMLDecoder::err& err) {
274 ldpp_dout(this, 5) << "ERROR: unexpected xml: RoleId" << dendl;
275 op_ret = -EINVAL;
276 return;
277 }
278 ldpp_dout(this, 0) << "role_id decoded from master zonegroup response is" << role_id << dendl;
279 }
280
281 op_ret = role->create(s, true, role_id, y);
282 if (op_ret == -EEXIST) {
283 op_ret = -ERR_ROLE_EXISTS;
284 return;
285 }
286
287 if (op_ret == 0) {
288 s->formatter->open_object_section("CreateRoleResponse");
289 s->formatter->open_object_section("CreateRoleResult");
290 s->formatter->open_object_section("Role");
291 role->dump(s->formatter);
292 s->formatter->close_section();
293 s->formatter->close_section();
294 s->formatter->open_object_section("ResponseMetadata");
295 s->formatter->dump_string("RequestId", s->trans_id);
296 s->formatter->close_section();
297 s->formatter->close_section();
298 }
