148template <size_t N>
149constexpr std::bitset<N> make_bitmask(size_t s) {
150 // unfortunately none of the shift/logic operators of std::bitset have a constexpr variation
151 return s < 64 ? std::bitset<N> ((1ULL << s) - 1) :
152 std::bitset<N>((1ULL << 63) - 1) | make_bitmask<N> (s - 63) << 63;
153}
154
155template <size_t N>
156constexpr std::bitset<N> set_cont_bits(size_t start, size_t end)
157{
158 return (make_bitmask<N>(end - start)) << start;
159}

274inline bool operator ==(const MaskedIP& l, const MaskedIP& r) {
275 auto shift = std::max((l.v6 ? 128 : 32) - ((int) l.prefix),

336 static boost::optional<bool> as_bool(const std::string& s) {
337 std::size_t p = 0;
338
339 if (s.empty() || boost::iequals(s, "false")) {
340 return false;
341 }
342
343 try {
344 double d = std::stod(s, &p);
345 if (p == s.length()) {
346 return !((d == +0.0) || (d == -0.0) || std::isnan(d));
347 }
348 } catch (const std::logic_error& e) {
349 // Fallthrough
350 }
351
352 return true;
353 }

397 using unordered_multimap_it_pair = std::pair <std::unordered_multimap<std::string,std::string>::const_iterator, std::unordered_multimap<std::string,std::string>::const_iterator>;
398
399 template<typename F>
400 static bool andible(F&& f, const unordered_multimap_it_pair& it,
401 const std::vector<std::string>& v) {
402 for (auto itr = it.first; itr != it.second; itr++) {
403 bool matched = false;
404 for (const auto& d : v) {
405 if (std::forward<F>(f)(itr->second, d)) {
406 matched = true;
407 }
408 }
409 if (!matched)
410 return false;
411 }
412 return true;
413 }

416 static bool orrible(F&& f, const unordered_multimap_it_pair& it,
417 const std::vector<std::string>& v) {
418 for (auto itr = it.first; itr != it.second; itr++) {
419 for (const auto& d : v) {
420 if (std::forward<F>(f)(itr->second, d)) {
421 return true;
422 }
423 }
424 }

428 template<typename F, typename X>
429 static bool shortible(F&& f, X& x, const std::string& c,
430 const std::vector<std::string>& v) {
431 auto xc = std::forward<X>(x)(c);
432 if (!xc) {
433 return false;
434 }
435
436 for (const auto& d : v) {
437 auto xd = std::forward<X>(x)(d);
438 if (!xd) {
439 continue;
440 }
441
442 if (std::forward<F>(f)(*xc, *xd)) {
443 return true;
444 }
445 }
446 return false;
447 }
