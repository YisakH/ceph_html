98int RGWBackoffControlCR::operate(const DoutPrefixProvider *dpp) {
99 reenter(this) {
100 // retry the operation until it succeeds
101 while (true) {
102 yield {
103 std::lock_guard l{lock};
104 cr = alloc_cr();
105 cr->get();
106 call(cr);
107 }
108 {
109 std::lock_guard l{lock};
110 cr->put();
111 cr = NULL;
112 }
113 if (retcode >= 0) {
114 break;
115 }
116 if (retcode != -EBUSY && retcode != -EAGAIN) {
117 ldout(cct, 0) << "ERROR: RGWBackoffControlCR called coroutine returned " << retcode << dendl;
118 if (exit_on_error) {
119 return set_cr_error(retcode);
120 }
121 }
122 if (reset_backoff) {
123 backoff.reset();
124 }
125 yield backoff.backoff(this);
126 }
127
128 // run an optional finisher
129 yield call(alloc_finisher_cr());
130 if (retcode < 0) {
131 ldout(cct, 0) << "ERROR: call to finisher_cr() failed: retcode=" << retcode << dendl;
132 return set_cr_error(retcode);
133 }
134 return set_cr_done();
135 }
136 return 0;
137}

161int RGWShardCollectCR::operate(const DoutPrefixProvider *dpp) {
162 reenter(this) {
163 while (spawn_next()) {
164 current_running++;
165
166 while (current_running >= max_concurrent) {
167 int child_ret;
168 yield wait_for_child();
169 if (collect_next(&child_ret)) {
170 current_running--;
171 if (child_ret < 0 && child_ret != -ENOENT) {
172 ldout(cct, 10) << __func__ << ": failed to fetch log status, ret=" << child_ret << dendl;
173 status = child_ret;
174 }
175 }
176 }
177 }
178 while (current_running > 0) {
179 int child_ret;
180 yield wait_for_child();
181 if (collect_next(&child_ret)) {
182 current_running--;
183 if (child_ret < 0 && child_ret != -ENOENT) {
184 ldout(cct, 10) << __func__ << ": failed to fetch log status, ret=" << child_ret << dendl;
185 status = child_ret;
186 }
187 }
188 }
189 if (status < 0) {
190 return set_cr_error(status);
191 }
192 return set_cr_done();
193 }
194 return 0;
195}

650 int ret;
651 reenter(this) {
652 yield {
653 set_status("acquiring sync lock");
654 uint32_t lock_duration = cct->_conf->rgw_sync_lease_period;

665 set_status("lease lock failed, early abort");

667 }

669 yield;
670 }

684 }

689 spawn(new RGWReadRemoteMDLogShardInfoCR(sync_env, status.period, i,
690 &shards_info[i]), false);
691 }
692 }
693
694 drain_all_but_stack(lease_stack.get()); /* the lease cr still needs to run */
695
696 yield {
697 set_status("updating sync status");
698 for (int i = 0; i < (int)status.num_shards; i++) {
699 rgw_meta_sync_marker marker;

706 store->svc()->sysobj,

711 yield {
712 set_status("changing sync state: build full sync maps");
713 status.state = rgw_meta_sync_info::StateBuildingFullSyncMaps;
714 rgw::sal::RadosStore* store = sync_env->store;
715 call(new RGWSimpleRadosWriteCR<rgw_meta_sync_info>(dpp, sync_env->async_rados, store->svc()->sysobj,
716 rgw_raw_obj(store->svc()->zone->get_zone_params().log_pool, sync_env->status_oid()),
717 status));
718 }
719 set_status("drop lock lease");
720 yield lease_cr->go_down();
721 while (collect(&ret, NULL)) {
722 if (ret < 0) {
723 return set_cr_error(ret);
724 }
725 yield;
726 }
727 drain_all();
728 return set_cr_done();
729 }

887
888 reenter(this) {
889 yield {
890 set_status(string("acquiring lock (") + sync_env->status_oid() + ")");
891 uint32_t lock_duration = cct->_conf->rgw_sync_lease_period;

906 yield;
907 }

910 mdlog_sync_full_sync_index_prefix));

914 }
915 if (get_ret_status() < 0) {
916 ldpp_dout(dpp, 0) << "ERROR: failed to fetch metadata sections" << dendl;
917 yield entries_index->finish();
918 yield lease_cr->go_down();
919 drain_all();
920 return set_cr_error(get_ret_status());

923 sections_iter = sections.begin();
924 for (; sections_iter != sections.end(); ++sections_iter) {
925 do {
926 yield {
927#define META_FULL_SYNC_CHUNK_SIZE "1000"
928 string entrypoint = string("/admin/metadata/") + *sections_iter;
929 rgw_http_param_pair pairs[] = { { "max-entries", META_FULL_SYNC_CHUNK_SIZE },

939 ret_status = 0;
940 }
941 if (ret_status < 0) {
942 tn->log(0, SSTR("ERROR: failed to fetch metadata section: " << *sections_iter));
943 yield entries_index->finish();
944 yield lease_cr->go_down();
945 drain_all();
946 return set_cr_error(ret_status);
947 }

950 if (!lease_cr->is_locked()) {

952 break;
953 }
954 yield; // allow entries_index consumer to make progress
955
956 tn->log(20, SSTR("list metadata: section=" << *sections_iter << " key=" << *iter));
957 string s = *sections_iter + ":" + *iter;
958 int shard_id;
959 rgw::sal::RadosStore* store = sync_env->store;
960 int ret = store->ctl()->meta.mgr->get_shard_id(*sections_iter, *iter, &shard_id);
961 if (ret < 0) {
962 tn->log(0, SSTR("ERROR: could not determine shard id for " << *sections_iter << ":" << *iter));
963 ret_status = ret;
964 break;
965 }
966 if (!entries_index->append(s, shard_id)) {
967 break;
968 }
969 }
970 } while (result.truncated);
971 }
972 yield {
973 if (!entries_index->finish()) {
974 failed = true;
975 }
976 }
977 if (!failed) {
978 for (map<uint32_t, rgw_meta_sync_marker>::iterator iter = markers.begin(); iter != markers.end(); ++iter) {
979 int shard_id = (int)iter->first;
980 rgw_meta_sync_marker& marker = iter->second;
981 marker.total_entries = entries_index->get_total_entries(shard_id);
982 spawn(new RGWSimpleRadosWriteCR<rgw_meta_sync_marker>(dpp, sync_env->async_rados, sync_env->store->svc()->sysobj,
983 rgw_raw_obj(sync_env->store->svc()->zone->get_zone_params().log_pool, sync_env->shard_obj_name(shard_id)),
984 marker), true);
985 }
986 }
987
988 drain_all_but_stack(lease_stack.get()); /* the lease cr still needs to run */
989
990 yield lease_cr->go_down();
991
992 int ret;
993 while (collect(&ret, NULL)) {
994 if (ret < 0) {
995 return set_cr_error(ret);
996 }
997 yield;
998 }
999 drain_all();
1000 if (failed) {
1001 yield return set_cr_error(-EIO);
1002 }
1003 if (lost_lock) {
1004 yield return set_cr_error(-EBUSY);
1005 }
1006
1007 if (ret_status < 0) {
1008 yield return set_cr_error(ret_status);
1009 }
1010
1011 yield return set_cr_done();
1012 }
1013 return 0;
1014 }

1285 for (tries = 0; tries < NUM_TRANSIENT_ERROR_RETRIES; tries++) {

1287 pos = raw_key.find(':');
1288 section = raw_key.substr(0, pos);
1289 key = raw_key.substr(pos + 1);
1290 tn->log(10, SSTR("fetching remote metadata entry" << (tries == 0 ? "" : " (retry)")));
1291 call(new RGWReadRemoteMetadataCR(sync_env, section, key, &md_bl, tn));
1292 }
1293
1294 sync_status = retcode;
1295
1296 if (sync_status == -ENOENT) {
1297 /* FIXME: do we need to remove the entry from the local zone? */
1298 break;
1299 }
1300
1301 if (sync_status < 0) {
1302 if (tries < NUM_TRANSIENT_ERROR_RETRIES - 1) {
1303 ldpp_dout(dpp, 20) << *this << ": failed to fetch remote metadata: " << section << ":" << key << ", will retry" << dendl;
1304 continue;
1305 }
1306
1307 tn->log(10, SSTR("failed to read remote metadata entry: section=" << section << " key=" << key << " status=" << sync_status));
1308 log_error() << "failed to read remote metadata entry: section=" << section << " key=" << key << " status=" << sync_status << std::endl;
1309 yield call(sync_env->error_logger->log_error_cr(dpp, sync_env->conn->get_remote_id(), section, key, -sync_status,
1310 string("failed to read remote metadata entry: ") + cpp_strerror(-sync_status)));
1311 return set_cr_error(sync_status);
1312 }
1313
1314 break;
1315 }
1316
1317 retcode = 0;
1318 for (tries = 0; tries < NUM_TRANSIENT_ERROR_RETRIES; tries++) {
1319 if (sync_status != -ENOENT) {
1320 tn->log(10, SSTR("storing local metadata entry"));
1321 yield call(new RGWMetaStoreEntryCR(sync_env, raw_key, md_bl));
1322 } else {
1323 tn->log(10, SSTR("removing local metadata entry"));
1324 yield call(new RGWMetaRemoveEntryCR(sync_env, raw_key));
1325 }
1326 if ((retcode < 0) && (tries < NUM_TRANSIENT_ERROR_RETRIES - 1)) {
1327 ldpp_dout(dpp, 20) << *this << ": failed to store metadata: " << section << ":" << key << ", got retcode=" << retcode << dendl;
1328 continue;
1329 }
1330 break;
1331 }
1332
1333 sync_status = retcode;
1334
1335 if (sync_status == 0 && marker_tracker) {
1336 /* update marker */
1337 yield call(marker_tracker->finish(entry_marker));
1338 sync_status = retcode;
1339 }
1340 if (sync_status < 0) {
1341 tn->log(10, SSTR("failed, status=" << sync_status));
1342 return set_cr_error(sync_status);
1343 }
1344 tn->log(10, "success");
1345 return set_cr_done();
1346 }
1347 return 0;
1348}

1507 int child_ret;
1508 RGWCoroutinesStack *child;
1509 while (collect_next(&child_ret, &child)) {
1510 auto iter = stack_to_pos.find(child);
1511 if (iter == stack_to_pos.end()) {
1512 /* some other stack that we don't care about */
1513 continue;
1514 }
1515
1516 string& pos = iter->second;
1517
1518 if (child_ret < 0) {
1519 ldpp_dout(sync_env->dpp, 0) << *this << ": child operation stack=" << child << " entry=" << pos << " returned " << child_ret << dendl;
1520 // on any error code from RGWMetaSyncSingleEntryCR, we do not advance
1521 // the sync status marker past this entry, and set
1522 // can_adjust_marker=false to exit out of RGWMetaSyncShardCR.
1523 // RGWMetaSyncShardControlCR will rerun RGWMetaSyncShardCR from the
1524 // previous marker and retry
1525 can_adjust_marker = false;

1527

1529 ceph_assert(prev_iter != pos_to_prev.end());
1530
1531 if (pos_to_prev.size() == 1) {
1532 if (can_adjust_marker) {
1533 sync_marker.marker = pos;
1534 }
1535 pos_to_prev.erase(prev_iter);
1536 } else {
1537 ceph_assert(pos_to_prev.size() > 1);
1538 pos_to_prev.erase(prev_iter);
1539 prev_iter = pos_to_prev.begin();
1540 if (can_adjust_marker) {
1541 sync_marker.marker = prev_iter->second;
1542 }
1543 }

1550 int full_sync() {
1551#define OMAP_GET_MAX_ENTRIES 100
1552 int max_entries = OMAP_GET_MAX_ENTRIES;
1553 reenter(&full_cr) {
1554 set_status("full_sync");
1555 tn->log(10, "start full sync");
1556 oid = full_sync_index_shard_oid(shard_id);
1557 can_adjust_marker = true;
1558 /* grab lock */
1559 yield {
1560 uint32_t lock_duration = cct->_conf->rgw_sync_lease_period;
1561 string lock_name = "sync_lock";
1562 rgw::sal::RadosStore* store = sync_env->store;
1563 lease_cr.reset(new RGWContinuousLeaseCR(sync_env->async_rados, store,
1564 rgw_raw_obj(pool, sync_env->shard_obj_name(shard_id)),
1565 lock_name, lock_duration, this));
1566 lease_stack.reset(spawn(lease_cr.get(), false));
1567 lost_lock = false;
1568 }
1569 while (!lease_cr->is_locked()) {
1570 if (lease_cr->is_done()) {
1571 drain_all();
1572 tn->log(5, "failed to take lease");
1573 return lease_cr->get_ret_status();
1574 }
1575 set_sleeping(true);
1576 yield;
1577 }
1578 tn->log(10, "took lease");
1579
1580 /* lock succeeded, a retry now should avoid previous backoff status */
1581 *reset_backoff = true;
1582
1583 /* prepare marker tracker */
1584 set_marker_tracker(new RGWMetaSyncShardMarkerTrack(sync_env,
1585 sync_env->shard_obj_name(shard_id),
1586 sync_marker, tn));
1587
1588 marker = sync_marker.marker;
1589
1590 total_entries = sync_marker.pos;
1591
1592 /* sync! */
1593 do {
1594 if (!lease_cr->is_locked()) {
1595 tn->log(10, "lost lease");
1596 lost_lock = true;
1597 break;
1598 }
1599 omapkeys = std::make_shared<RGWRadosGetOmapKeysCR::Result>();
1600 yield call(new RGWRadosGetOmapKeysCR(sync_env->store, rgw_raw_obj(pool, oid),
1601 marker, max_entries, omapkeys));
1602 if (retcode < 0) {
1603 ldpp_dout(sync_env->dpp, 0) << "ERROR: " << __func__ << "(): RGWRadosGetOmapKeysCR() returned ret=" << retcode << dendl;
1604 tn->log(0, SSTR("ERROR: failed to list omap keys, status=" << retcode));
1605 yield lease_cr->go_down();
1606 drain_all();
1607 return retcode;
1608 }
1609 entries = std::move(omapkeys->entries);
1610 tn->log(20, SSTR("retrieved " << entries.size() << " entries to sync"));
1611 if (entries.size() > 0) {
1612 tn->set_flag(RGW_SNS_FLAG_ACTIVE); /* actually have entries to sync */
1613 }
1614 iter = entries.begin();
1615 for (; iter != entries.end(); ++iter) {
1616 marker = *iter;
1617 tn->log(20, SSTR("full sync: " << marker));
1618 total_entries++;
1619 if (!marker_tracker->start(marker, total_entries, real_time())) {
1620 tn->log(0, SSTR("ERROR: cannot start syncing " << marker << ". Duplicate entry?"));
1621 } else {
1622 // fetch remote and write locally
1623 yield {
1624 RGWCoroutinesStack *stack = spawn(new RGWMetaSyncSingleEntryCR(sync_env, marker, marker, MDLOG_STATUS_COMPLETE, marker_tracker, tn), false);
1625 // stack_to_pos holds a reference to the stack
1626 stack_to_pos[stack] = marker;
1627 pos_to_prev[marker] = marker;
1628 }
1629 // limit spawn window
1630 while (num_spawned() > META_SYNC_SPAWN_WINDOW) {
1631 yield wait_for_child();
1632 collect_children();
1633 }
1634 }
1635 }
1636 collect_children();
1637 } while (omapkeys->more && can_adjust_marker);
1638
1639 tn->unset_flag(RGW_SNS_FLAG_ACTIVE); /* actually have entries to sync */
1640
1641 while (num_spawned() > 1) {
1642 yield wait_for_child();
1643 collect_children();
1644 }
1645
1646 if (!lost_lock) {
1647 /* update marker to reflect we're done with full sync */
1648 if (can_adjust_marker) {
1649 // apply updates to a temporary marker, or operate() will send us
1650 // to incremental_sync() after we yield
1651 temp_marker = sync_marker;
1652 temp_marker->state = rgw_meta_sync_marker::IncrementalSync;
1653 temp_marker->marker = std::move(temp_marker->next_step_marker);
1654 temp_marker->next_step_marker.clear();
1655 temp_marker->realm_epoch = realm_epoch;
1656 ldpp_dout(sync_env->dpp, 4) << *this << ": saving marker pos=" << temp_marker->marker << " realm_epoch=" << realm_epoch << dendl;
1657
1658 using WriteMarkerCR = RGWSimpleRadosWriteCR<rgw_meta_sync_marker>;
1659 yield call(new WriteMarkerCR(sync_env->dpp, sync_env->async_rados, sync_env->store->svc()->sysobj,
1660 rgw_raw_obj(pool, sync_env->shard_obj_name(shard_id)),
1661 *temp_marker));
1662 }
1663
1664 if (retcode < 0) {
1665 ldpp_dout(sync_env->dpp, 0) << "ERROR: failed to set sync marker: retcode=" << retcode << dendl;
1666 yield lease_cr->go_down();
1667 drain_all();
1668 return retcode;
1669 }
1670 // clean up full sync index
1671 yield {
1672 auto oid = full_sync_index_shard_oid(shard_id);
1673 call(new RGWRadosRemoveCR(sync_env->store, {pool, oid}));
1674 }
1675 }
1676
1677 /*
1678 * if we reached here, it means that lost_lock is true, otherwise the state
1679 * change in the previous block will prevent us from reaching here
1680 */
1681
1682 yield lease_cr->go_down();
1683
1684 lease_cr.reset();
1685
1686 drain_all();
1687
1688 if (!can_adjust_marker) {
1689 return -EAGAIN;
1690 }
1691
1692 if (lost_lock) {
1693 return -EBUSY;
1694 }
1695
1696 tn->log(10, "full sync complete");
1697
1698 // apply the sync marker update
1699 ceph_assert(temp_marker);
1700 sync_marker = std::move(*temp_marker);
1701 temp_marker = boost::none;
1702 // must not yield after this point!
1703 }
1704 return 0;
1705 }

1708 int incremental_sync() {
1709 reenter(&incremental_cr) {
1710 set_status("incremental_sync");
1711 tn->log(10, "start incremental sync");
1712 can_adjust_marker = true;
1713 /* grab lock */
1714 if (!lease_cr) { /* could have had a lease_cr lock from previous state */
1715 yield {
1716 uint32_t lock_duration = cct->_conf->rgw_sync_lease_period;
1717 string lock_name = "sync_lock";
1718 rgw::sal::RadosStore* store = sync_env->store;
1719 lease_cr.reset( new RGWContinuousLeaseCR(sync_env->async_rados, store,
1720 rgw_raw_obj(pool, sync_env->shard_obj_name(shard_id)),
1721 lock_name, lock_duration, this));
1722 lease_stack.reset(spawn(lease_cr.get(), false));
1723 lost_lock = false;
1724 }
1725 while (!lease_cr->is_locked()) {
1726 if (lease_cr->is_done()) {
1727 drain_all();
1728 tn->log(10, "failed to take lease");
1729 return lease_cr->get_ret_status();
1730 }
1731 set_sleeping(true);
1732 yield;
1733 }
1734 }
1735 tn->log(10, "took lease");
1736 // if the period has advanced, we can't use the existing marker
1737 if (sync_marker.realm_epoch < realm_epoch) {
1738 ldpp_dout(sync_env->dpp, 4) << "clearing marker=" << sync_marker.marker
1739 << " from old realm_epoch=" << sync_marker.realm_epoch
1740 << " (now " << realm_epoch << ')' << dendl;
1741 sync_marker.realm_epoch = realm_epoch;
1742 sync_marker.marker.clear();
1743 }
1744 mdlog_marker = sync_marker.marker;
1745 set_marker_tracker(new RGWMetaSyncShardMarkerTrack(sync_env,
1746 sync_env->shard_obj_name(shard_id),
1747 sync_marker, tn));
1748
1749 /*
1750 * mdlog_marker: the remote sync marker positiion
1751 * sync_marker: the local sync marker position
1752 * max_marker: the max mdlog position that we fetched
1753 * marker: the current position we try to sync
1754 * period_marker: the last marker before the next period begins (optional)
1755 */
1756 marker = max_marker = sync_marker.marker;
1757 /* inc sync */
1758 do {
1759 if (!lease_cr->is_locked()) {
1760 lost_lock = true;
1761 tn->log(10, "lost lease");
1762 break;
1763 }
1764#define INCREMENTAL_MAX_ENTRIES 100
1765 ldpp_dout(sync_env->dpp, 20) << __func__ << ":" << __LINE__ << ": shard_id=" << shard_id << " mdlog_marker=" << mdlog_marker << " sync_marker.marker=" << sync_marker.marker << " period_marker=" << period_marker << dendl;
1766 if (!period_marker.empty() && period_marker <= mdlog_marker) {
1767 tn->log(10, SSTR("finished syncing current period: mdlog_marker=" << mdlog_marker << " sync_marker=" << sync_marker.marker << " period_marker=" << period_marker));
1768 done_with_period = true;
1769 break;
1770 }
1771 if (mdlog_marker <= max_marker) {
1772 /* we're at the tip, try to bring more entries */
1773 ldpp_dout(sync_env->dpp, 20) << __func__ << ":" << __LINE__ << ": shard_id=" << shard_id << " syncing mdlog for shard_id=" << shard_id << dendl;
1774 yield call(new RGWCloneMetaLogCoroutine(sync_env, mdlog,
1775 period, shard_id,
1776 mdlog_marker, &mdlog_marker));
1777 }
1778 if (retcode < 0) {
1779 tn->log(10, SSTR(*this << ": failed to fetch more log entries, retcode=" << retcode));
1780 yield lease_cr->go_down();
1781 drain_all();
1782 *reset_backoff = false; // back off and try again later
1783 return retcode;
1784 }
1785 *reset_backoff = true; /* if we got to this point, all systems function */
1786 if (mdlog_marker > max_marker) {
1787 tn->set_flag(RGW_SNS_FLAG_ACTIVE); /* actually have entries to sync */
1788 tn->log(20, SSTR("mdlog_marker=" << mdlog_marker << " sync_marker=" << sync_marker.marker));
1789 marker = max_marker;
1790 yield call(new RGWReadMDLogEntriesCR(sync_env, mdlog, shard_id,
1791 &max_marker, INCREMENTAL_MAX_ENTRIES,
1792 &log_entries, &truncated));
1793 if (retcode < 0) {
1794 tn->log(10, SSTR("failed to list mdlog entries, retcode=" << retcode));
1795 yield lease_cr->go_down();
1796 drain_all();
1797 *reset_backoff = false; // back off and try again later
1798 return retcode;
1799 }
1800 for (log_iter = log_entries.begin(); log_iter != log_entries.end() && !done_with_period; ++log_iter) {
1801 if (!period_marker.empty() && period_marker <= log_iter->id) {
1802 done_with_period = true;
1803 if (period_marker < log_iter->id) {
1804 tn->log(10, SSTR("found key=" << log_iter->id
1805 << " past period_marker=" << period_marker));
1806 break;
1807 }
1808 ldpp_dout(sync_env->dpp, 10) << "found key at period_marker=" << period_marker << dendl;
1809 // sync this entry, then return control to RGWMetaSyncCR
1810 }
1811 if (!mdlog_entry.convert_from(*log_iter)) {
1812 tn->log(0, SSTR("ERROR: failed to convert mdlog entry, shard_id=" << shard_id << " log_entry: " << log_iter->id << ":" << log_iter->section << ":" << log_iter->name << ":" << log_iter->timestamp << " ... skipping entry"));
1813 continue;
1814 }
1815 tn->log(20, SSTR("log_entry: " << log_iter->id << ":" << log_iter->section << ":" << log_iter->name << ":" << log_iter->timestamp));
1816 if (!marker_tracker->start(log_iter->id, 0, log_iter->timestamp.to_real_time())) {
1817 ldpp_dout(sync_env->dpp, 0) << "ERROR: cannot start syncing " << log_iter->id << ". Duplicate entry?" << dendl;
1818 } else {
1819 raw_key = log_iter->section + ":" + log_iter->name;
1820 yield {
1821 RGWCoroutinesStack *stack = spawn(new RGWMetaSyncSingleEntryCR(sync_env, raw_key, log_iter->id, mdlog_entry.log_data.status, marker_tracker, tn), false);
1822 ceph_assert(stack);
1823 // stack_to_pos holds a reference to the stack
1824 stack_to_pos[stack] = log_iter->id;
1825 pos_to_prev[log_iter->id] = marker;
1826 }
1827 // limit spawn window
1828 while (num_spawned() > META_SYNC_SPAWN_WINDOW) {
1829 yield wait_for_child();
1830 collect_children();
1831 }
1832 }
1833 marker = log_iter->id;
1834 }
1835 }
1836 collect_children();
1837 ldpp_dout(sync_env->dpp, 20) << __func__ << ":" << __LINE__ << ": shard_id=" << shard_id << " mdlog_marker=" << mdlog_marker << " max_marker=" << max_marker << " sync_marker.marker=" << sync_marker.marker << " period_marker=" << period_marker << dendl;
1838 if (done_with_period) {
1839 // return control to RGWMetaSyncCR and advance to the next period
1840 tn->log(10, SSTR(*this << ": done with period"));
1841 break;
1842 }
1843 if (mdlog_marker == max_marker && can_adjust_marker) {
1844 tn->unset_flag(RGW_SNS_FLAG_ACTIVE);
1845#define INCREMENTAL_INTERVAL 20
1846 yield wait(utime_t(INCREMENTAL_INTERVAL, 0));
1847 }
1848 } while (can_adjust_marker);
1849
1850 tn->unset_flag(RGW_SNS_FLAG_ACTIVE);
1851
1852 while (num_spawned() > 1) {
1853 yield wait_for_child();
1854 collect_children();
1855 }
1856
1857 yield lease_cr->go_down();
1858
1859 drain_all();
1860
1861 if (lost_lock) {
1862 return -EBUSY;
1863 }
1864
1865 if (!can_adjust_marker) {
1866 return -EAGAIN;
1867 }
1868
1869 return set_cr_done();
1870 }

1919class RGWMetaSyncCR : public RGWCoroutine {
1920 RGWMetaSyncEnv *sync_env;
1921 const rgw_pool& pool;
1922 RGWPeriodHistory::Cursor cursor; //< sync position in period history
1923 RGWPeriodHistory::Cursor next; //< next period in history
1924 rgw_meta_sync_status sync_status;
1925 RGWSyncTraceNodeRef tn;
1926
1927 std::mutex mutex; //< protect access to shard_crs
1928
1929 // TODO: it should be enough to hold a reference on the stack only, as calling
1930 // RGWCoroutinesStack::wakeup() doesn't refer to the RGWCoroutine if it has
1931 // already completed
1932 using ControlCRRef = boost::intrusive_ptr<RGWMetaSyncShardControlCR>;
1933 using StackRef = boost::intrusive_ptr<RGWCoroutinesStack>;
1934 using RefPair = std::pair<ControlCRRef, StackRef>;
1935 map<int, RefPair> shard_crs;
1936 int ret{0};
1937
1938public:
1939 RGWMetaSyncCR(RGWMetaSyncEnv *_sync_env, const RGWPeriodHistory::Cursor &cursor,
1940 const rgw_meta_sync_status& _sync_status, RGWSyncTraceNodeRef& _tn)
1941 : RGWCoroutine(_sync_env->cct), sync_env(_sync_env),
1942 pool(sync_env->store->svc()->zone->get_zone_params().log_pool),
1943 cursor(cursor), sync_status(_sync_status), tn(_tn) {}
1944
1945 ~RGWMetaSyncCR() {
1946 }
1947
1948 int operate(const DoutPrefixProvider *dpp) override {
1949 reenter(this) {
1950 // loop through one period at a time
1951 tn->log(1, "start");
1952 for (;;) {
1953 if (cursor == sync_env->store->svc()->mdlog->get_period_history()->get_current()) {
1954 next = RGWPeriodHistory::Cursor{};
1955 if (cursor) {
1956 ldpp_dout(dpp, 10) << "RGWMetaSyncCR on current period="
1957 << cursor.get_period().get_id() << dendl;
1958 } else {
1959 ldpp_dout(dpp, 10) << "RGWMetaSyncCR with no period" << dendl;
1960 }
1961 } else {
1962 next = cursor;
1963 next.next();
1964 ldpp_dout(dpp, 10) << "RGWMetaSyncCR on period="
1965 << cursor.get_period().get_id() << ", next="
1966 << next.get_period().get_id() << dendl;
1967 }
1968
1969 yield {
1970 // get the mdlog for the current period (may be empty)
1971 auto& period_id = sync_status.sync_info.period;
1972 auto realm_epoch = sync_status.sync_info.realm_epoch;
1973 auto mdlog = sync_env->store->svc()->mdlog->get_log(period_id);
1974
1975 tn->log(1, SSTR("realm epoch=" << realm_epoch << " period id=" << period_id));
1976
1977 // prevent wakeup() from accessing shard_crs while we're spawning them
1978 std::lock_guard<std::mutex> lock(mutex);
1979
1980 // sync this period on each shard
1981 for (const auto& m : sync_status.sync_markers) {
1982 uint32_t shard_id = m.first;
1983 auto& marker = m.second;
1984
1985 std::string period_marker;
1986 if (next) {
1987 // read the maximum marker from the next period's sync status
1988 period_marker = next.get_period().get_sync_status()[shard_id];
1989 if (period_marker.empty()) {
1990 // no metadata changes have occurred on this shard, skip it
1991 ldpp_dout(dpp, 10) << "RGWMetaSyncCR: skipping shard " << shard_id
1992 << " with empty period marker" << dendl;
1993 continue;
1994 }
1995 }
1996
1997 using ShardCR = RGWMetaSyncShardControlCR;
1998 auto cr = new ShardCR(sync_env, pool, period_id, realm_epoch,
1999 mdlog, shard_id, marker,
2000 std::move(period_marker), tn);
2001 auto stack = spawn(cr, false);
2002 shard_crs[shard_id] = RefPair{cr, stack};
2003 }
2004 }
2005 // wait for each shard to complete
2006 while (ret == 0 && num_spawned() > 0) {
2007 yield wait_for_child();
2008 collect(&ret, nullptr);
2009 }
2010 drain_all();
2011 {
2012 // drop shard cr refs under lock
2013 std::lock_guard<std::mutex> lock(mutex);
2014 shard_crs.clear();
2015 }
2016 if (ret < 0) {
2017 return set_cr_error(ret);
2018 }
2019 // advance to the next period
2020 ceph_assert(next);
2021 cursor = next;
2022
2023 // write the updated sync info
2024 sync_status.sync_info.period = cursor.get_period().get_id();
2025 sync_status.sync_info.realm_epoch = cursor.get_epoch();
2026 yield call(new RGWSimpleRadosWriteCR<rgw_meta_sync_info>(dpp, sync_env->async_rados,
2027 sync_env->store->svc()->sysobj,
2028 rgw_raw_obj(pool, sync_env->status_oid()),
2029 sync_status.sync_info));
2030 }
2031 }
2032 return 0;
2033 }

2037 auto iter = shard_crs.find(shard_id);
2038 if (iter == shard_crs.end()) {
2039 return;
2040 }
2041 iter->second.first->wakeup();
2042 }

2156 }

2161 rgw_mdlog_info mdlog_info;

2240 }
2241
2242 RGWPeriodHistory::Cursor cursor;
