113int RGWSI_Zone::do_start(optional_yield y, const DoutPrefixProvider *dpp)
114{
115 int ret = sysobj_svc->start(y, dpp);
116 if (ret < 0) {
117 return ret;
118 }
119
120 assert(sysobj_svc->is_started()); /* if not then there's ordering issue */
121
122 ret = rados_svc->start(y, dpp);
123 if (ret < 0) {
124 return ret;
125 }
126
127 ret = realm->init(dpp, cct, sysobj_svc, y);
128 if (ret < 0 && ret != -ENOENT) {
129 ldpp_dout(dpp, 0) << "failed reading realm info: ret "<< ret << " " << cpp_strerror(-ret) << dendl;
130 return ret;
131 }
132
133 ldpp_dout(dpp, 20) << "realm " << realm->get_name() << " " << realm->get_id() << dendl;
134 ret = current_period->init(dpp, cct, sysobj_svc, realm->get_id(), y,
135 realm->get_name());
136 if (ret < 0 && ret != -ENOENT) {
137 ldpp_dout(dpp, 0) << "failed reading current period info: " << " " << cpp_strerror(-ret) << dendl;
138 return ret;
139 }
140
141 ret = zone_params->init(dpp, cct, sysobj_svc, y);
142 bool found_zone = (ret == 0);
143 if (ret < 0 && ret != -ENOENT) {
144 lderr(cct) << "failed reading zone info: ret "<< ret << " " << cpp_strerror(-ret) << dendl;
145 return ret;
146 }
147
148 cur_zone_id = rgw_zone_id(zone_params->get_id());
149
150 bool found_period_conf = false;
151
152 /* try to find zone in period config (if we have one) */
153 if (found_zone &&
154 !current_period->get_id().empty()) {
155 found_period_conf = current_period->find_zone(dpp,
156 cur_zone_id,
157 zonegroup,
158 y);
159 if (ret < 0) {
160 ldpp_dout(dpp, 0) << "ERROR: current_period->find_zone() returned ret=" << ret << dendl;
161 return ret;
162 }
163 if (!found_period_conf) {
164 ldpp_dout(dpp, 0) << "period (" << current_period->get_id() << " does not have zone " << cur_zone_id << " configured" << dendl;
165 }
166 }
167
168 RGWRealm search_realm;
169
170 if (found_zone &&
171 !found_period_conf) {
172 ldpp_dout(dpp, 20) << "searching for the correct realm" << dendl;
173 ret = search_realm_with_zone(dpp,
174 cur_zone_id,
175 realm,
176 current_period,
177 zonegroup,
178 &found_period_conf,
179 y);
180 if (ret < 0) {
181 ldpp_dout(dpp, 0) << "ERROR: search_realm_conf() failed: ret="<< ret << dendl;
182 return ret;
183 }
184 }
185 bool zg_initialized = found_period_conf;
186
187 if (!zg_initialized) {
188 /* couldn't find a proper period config, use local zonegroup */
189 ret = zonegroup->init(dpp, cct, sysobj_svc, y);
190 zg_initialized = (ret == 0);
191 if (ret < 0 && ret != -ENOENT) {
192 ldpp_dout(dpp, 0) << "failed reading zonegroup info: " << cpp_strerror(-ret) << dendl;
193 return ret;
194 }
195 }
196
197 auto& zonegroup_param = cct->_conf->rgw_zonegroup;
198 bool init_from_period = found_period_conf;
199 bool explicit_zg = !zonegroup_param.empty();
200
201 if (!zg_initialized &&
202 (!explicit_zg || zonegroup_param == default_zonegroup_name)) {
203 /* we couldn't initialize any zonegroup,
204 falling back to a non-multisite config with default zonegroup */
205 ret = create_default_zg(dpp, y);
206 if (ret < 0) {
207 return ret;
208 }
209 zg_initialized = true;
210 }
211
212 if (!zg_initialized) {
213 ldpp_dout(dpp, 0) << "ERROR: could not find zonegroup (" << zonegroup_param << ")" << dendl;
214 return -ENOENT;
215 }
216
217 /* we have zonegroup now */
218
219 if (explicit_zg &&
220 zonegroup->get_name() != zonegroup_param) {
221 ldpp_dout(dpp, 0) << "ERROR: incorrect zonegroup: " << zonegroup_param << " (got: " << zonegroup_param << ", expected: " << zonegroup->get_name() << ")" << dendl;
222 return -EINVAL;
223 }
224
225 auto& zone_param = cct->_conf->rgw_zone;
226 bool explicit_zone = !zone_param.empty();
227
228 if (!found_zone) {
229 if ((!explicit_zone || zone_param == default_zone_name) &&
230 zonegroup->get_name() == default_zonegroup_name) {
231 ret = init_default_zone(dpp, y);
232 if (ret < 0 && ret != -ENOENT) {
233 return ret;
234 }
235 cur_zone_id = zone_params->get_id();
236 } else {
237 ldpp_dout(dpp, 0) << "ERROR: could not find zone (" << zone_param << ")" << dendl;
238 return -ENOENT;
239 }
240 }
241
242 /* we have zone now */
243
244 ret = replace_region_with_zonegroup(dpp, y);
245 if (ret < 0) {
246 ldpp_dout(dpp, -1) << "failed converting region to zonegroup : ret "<< ret << " " << cpp_strerror(-ret) << dendl;
247 return ret;
248 }
249
250 ret = convert_regionmap(dpp, y);
251 if (ret < 0) {
252 ldpp_dout(dpp, -1) << "failed converting regionmap: " << cpp_strerror(-ret) << dendl;
253 return ret;
254 }
255
256 auto zone_iter = zonegroup->zones.find(zone_params->get_id());
257 if (zone_iter == zonegroup->zones.end()) {
258 /* shouldn't happen if relying on period config */
259 if (!init_from_period) {
260 ldpp_dout(dpp, -1) << "Cannot find zone id=" << zone_params->get_id() << " (name=" << zone_params->get_name() << ")" << dendl;
261 return -EINVAL;
262 }
263 ldpp_dout(dpp, 1) << "Cannot find zone id=" << zone_params->get_id() << " (name=" << zone_params->get_name() << "), switching to local zonegroup configuration" << dendl;
264 init_from_period = false;
265 zone_iter = zonegroup->zones.find(zone_params->get_id());
266 }
267 if (zone_iter == zonegroup->zones.end()) {
268 ldpp_dout(dpp, -1) << "Cannot find zone id=" << zone_params->get_id() << " (name=" << zone_params->get_name() << ")" << dendl;
269 return -EINVAL;
270 }
271 *zone_public_config = zone_iter->second;
272 ldout(cct, 20) << "zone " << zone_params->get_name() << " found" << dendl;
273
274 ldpp_dout(dpp, 4) << "Realm: " << std::left << setw(20) << realm->get_name() << " (" << realm->get_id() << ")" << dendl;
275 ldpp_dout(dpp, 4) << "ZoneGroup: " << std::left << setw(20) << zonegroup->get_name() << " (" << zonegroup->get_id() << ")" << dendl;
276 ldpp_dout(dpp, 4) << "Zone: " << std::left << setw(20) << zone_params->get_name() << " (" << zone_params->get_id() << ")" << dendl;
277
278 if (init_from_period) {
279 ldpp_dout(dpp, 4) << "using period configuration: " << current_period->get_id() << ":" << current_period->get_epoch() << dendl;
280 ret = init_zg_from_period(dpp, y);
281 if (ret < 0) {
282 return ret;
283 }
284 } else {
285 ldout(cct, 10) << "cannot find current period zonegroup using local zonegroup configuration" << dendl;
286 ret = init_zg_from_local(dpp, y);
287 if (ret < 0) {
288 return ret;
289 }
290 // read period_config into current_period
291 auto& period_config = current_period->get_config();
292 ret = period_config.read(dpp, sysobj_svc, zonegroup->realm_id, y);
293 if (ret < 0 && ret != -ENOENT) {
294 ldout(cct, 0) << "ERROR: failed to read period config: "
295 << cpp_strerror(ret) << dendl;
296 return ret;
297 }
298 }
299
300 zone_short_id = current_period->get_map().get_zone_short_id(zone_params->get_id());
301
302 for (auto ziter : zonegroup->zones) {
303 auto zone_handler = std::make_shared<RGWBucketSyncPolicyHandler>(this, sync_modules_svc, bucket_sync_svc, ziter.second.id);
304 ret = zone_handler->init(dpp, y);
305 if (ret < 0) {
306 ldpp_dout(dpp, -1) << "ERROR: could not initialize zone policy handler for zone=" << ziter.second.name << dendl;
307 return ret;
308 }
309 sync_policy_handlers[ziter.second.id] = zone_handler;
310 }
311
312 sync_policy_handler = sync_policy_handlers[zone_id()]; /* we made sure earlier that zonegroup->zones has our zone */
313
314 set<rgw_zone_id> source_zones;
315 set<rgw_zone_id> target_zones;
316
317 sync_policy_handler->reflect(dpp, nullptr, nullptr,
318 nullptr, nullptr,
319 &source_zones,
320 &target_zones,
321 false); /* relaxed: also get all zones that we allow to sync to/from */
322
323 ret = sync_modules_svc->start(y, dpp);
324 if (ret < 0) {
325 return ret;
326 }
327
328 auto sync_modules = sync_modules_svc->get_manager();
329 RGWSyncModuleRef sm;
330 if (!sync_modules->get_module(zone_public_config->tier_type, &sm)) {
331 ldpp_dout(dpp, -1) << "ERROR: tier type not found: " << zone_public_config->tier_type << dendl;
332 return -EINVAL;
333 }
334
335 writeable_zone = sm->supports_writes();
336 exports_data = sm->supports_data_export();
337
338 /* first build all zones index */
339 for (auto ziter : zonegroup->zones) {
340 const rgw_zone_id& id = ziter.first;
341 RGWZone& z = ziter.second;
342 zone_id_by_name[z.name] = id;
343 zone_by_id[id] = z;
344 }
345
346 if (zone_by_id.find(zone_id()) == zone_by_id.end()) {
347 ldpp_dout(dpp, 0) << "WARNING: could not find zone config in zonegroup for local zone (" << zone_id() << "), will use defaults" << dendl;
348 }
349
350 for (const auto& ziter : zonegroup->zones) {
351 const rgw_zone_id& id = ziter.first;
352 const RGWZone& z = ziter.second;
353 if (id == zone_id()) {
354 continue;
355 }
356 if (z.endpoints.empty()) {
357 ldpp_dout(dpp, 0) << "WARNING: can't generate connection for zone " << z.id << " id " << z.name << ": no endpoints defined" << dendl;
358 continue;
359 }
360 ldpp_dout(dpp, 20) << "generating connection object for zone " << z.name << " id " << z.id << dendl;
361 RGWRESTConn *conn = new RGWRESTConn(cct, this, z.id, z.endpoints, zonegroup->api_name);
362 zone_conn_map[id] = conn;
363
364 bool zone_is_source = source_zones.find(z.id) != source_zones.end();
365 bool zone_is_target = target_zones.find(z.id) != target_zones.end();
366
367 if (zone_is_source || zone_is_target) {
368 if (zone_is_source && sync_modules->supports_data_export(z.tier_type)) {
369 data_sync_source_zones.push_back(&z);
370 }
371 if (zone_is_target) {
372 zone_data_notify_to_map[id] = conn;
373 }
374 } else {
375 ldpp_dout(dpp, 20) << "NOTICE: not syncing to/from zone " << z.name << " id " << z.id << dendl;
376 }
377 }
378
379 ldpp_dout(dpp, 20) << "started zone id=" << zone_params->get_id() << " (name=" << zone_params->get_name() <<
380 ") with tier type = " << zone_public_config->tier_type << dendl;
381
382 return 0;
383}

477int RGWSI_Zone::replace_region_with_zonegroup(const DoutPrefixProvider *dpp, optional_yield y)
478{
479 /* copy default region */
480 /* convert default region to default zonegroup */
481 string default_oid = cct->_conf->rgw_default_region_info_oid;
482 if (default_oid.empty()) {
483 default_oid = default_region_info_oid;
484 }
485
486 RGWZoneGroup default_zonegroup;
487 rgw_pool pool{default_zonegroup.get_pool(cct)};
488 string oid = "converted";
489 bufferlist bl;
490
491 RGWSysObjectCtx obj_ctx = sysobj_svc->init_obj_ctx();
492 RGWSysObj sysobj = sysobj_svc->get_obj(obj_ctx, rgw_raw_obj(pool, oid));
493
494 int ret = sysobj.rop().read(dpp, &bl, y);
495 if (ret < 0 && ret != -ENOENT) {
496 ldpp_dout(dpp, 0) << __func__ << " failed to read converted: ret "<< ret << " " << cpp_strerror(-ret)
497 << dendl;
498 return ret;
499 } else if (ret != -ENOENT) {
500 ldpp_dout(dpp, 20) << "System already converted " << dendl;
501 return 0;
502 }
503
504 string default_region;
505 ret = default_zonegroup.init(dpp, cct, sysobj_svc, y, false, true);
506 if (ret < 0) {
507 ldpp_dout(dpp, 0) << __func__ << " failed init default region: ret "<< ret << " " << cpp_strerror(-ret) << dendl;
508 return ret;
509 }
510 ret = default_zonegroup.read_default_id(dpp, default_region, y, true);
511 if (ret < 0 && ret != -ENOENT) {
512 ldpp_dout(dpp, 0) << __func__ << " failed reading old default region: ret "<< ret << " " << cpp_strerror(-ret) << dendl;
513 return ret;
514 }
515
516 /* convert regions to zonegroups */
517 list<string> regions;
518 ret = list_regions(dpp, regions);
519 if (ret < 0 && ret != -ENOENT) {
520 ldpp_dout(dpp, 0) << __func__ << " failed to list regions: ret "<< ret << " " << cpp_strerror(-ret) << dendl;
521 return ret;
522 } else if (ret == -ENOENT || regions.empty()) {
523 RGWZoneParams zoneparams(default_zone_name);
524 int ret = zoneparams.init(dpp, cct, sysobj_svc, y);
525 if (ret < 0 && ret != -ENOENT) {
526 ldpp_dout(dpp, 0) << __func__ << ": error initializing default zone params: " << cpp_strerror(-ret) << dendl;
527 return ret;
528 }
529 /* update master zone */
530 RGWZoneGroup default_zg(default_zonegroup_name);
531 ret = default_zg.init(dpp, cct, sysobj_svc, y);
532 if (ret < 0 && ret != -ENOENT) {
533 ldpp_dout(dpp, 0) << __func__ << ": error in initializing default zonegroup: " << cpp_strerror(-ret) << dendl;
534 return ret;
535 }
536 if (ret != -ENOENT && default_zg.master_zone.empty()) {
537 default_zg.master_zone = zoneparams.get_id();
538 return default_zg.update(dpp, y);
539 }
540 return 0;
541 }
542
543 string master_region;
544 rgw_zone_id master_zone;
545 for (list<string>::iterator iter = regions.begin(); iter != regions.end(); ++iter) {
546 if (*iter != default_zonegroup_name){
547 RGWZoneGroup region(*iter);
548 int ret = region.init(dpp, cct, sysobj_svc, y, true, true);
549 if (ret < 0) {
550 ldpp_dout(dpp, 0) << __func__ << " failed init region "<< *iter << ": " << cpp_strerror(-ret) << dendl;
551 return ret;
552 }
553 if (region.is_master_zonegroup()) {
554 master_region = region.get_id();
555 master_zone = region.master_zone;
556 }
557 }
558 }
559
560 /* create realm if there is none.
561 The realm name will be the region and zone concatenated
562 realm id will be mds of its name */
563 if (realm->get_id().empty() && !master_region.empty() && !master_zone.empty()) {
564 string new_realm_name = master_region + "." + master_zone.id;
565 unsigned char md5[CEPH_CRYPTO_MD5_DIGESTSIZE];
566 char md5_str[CEPH_CRYPTO_MD5_DIGESTSIZE * 2 + 1];
567 MD5 hash;
568 // Allow use of MD5 digest in FIPS mode for non-cryptographic purposes
569 hash.SetFlags(EVP_MD_CTX_FLAG_NON_FIPS_ALLOW);
570 hash.Update((const unsigned char *)new_realm_name.c_str(), new_realm_name.length());
571 hash.Final(md5);
572 buf_to_hex(md5, CEPH_CRYPTO_MD5_DIGESTSIZE, md5_str);
573 string new_realm_id(md5_str);
574 RGWRealm new_realm(new_realm_id,new_realm_name);
575 ret = new_realm.init(dpp, cct, sysobj_svc, y, false);
576 if (ret < 0) {
577 ldpp_dout(dpp, 0) << __func__ << " Error initing new realm: " << cpp_strerror(-ret) << dendl;
578 return ret;
579 }
580 ret = new_realm.create(dpp, y);
581 if (ret < 0 && ret != -EEXIST) {
582 ldpp_dout(dpp, 0) << __func__ << " Error creating new realm: " << cpp_strerror(-ret) << dendl;
583 return ret;
584 }
585 ret = new_realm.set_as_default(dpp, y);
586 if (ret < 0) {
587 ldpp_dout(dpp, 0) << __func__ << " Error setting realm as default: " << cpp_strerror(-ret) << dendl;
588 return ret;
589 }
590 ret = realm->init(dpp, cct, sysobj_svc, y);
591 if (ret < 0) {
592 ldpp_dout(dpp, 0) << __func__ << " Error initing realm: " << cpp_strerror(-ret) << dendl;
593 return ret;
594 }
595 ret = current_period->init(dpp, cct, sysobj_svc, realm->get_id(), y,
596 realm->get_name());
597 if (ret < 0) {
598 ldpp_dout(dpp, 0) << __func__ << " Error initing current period: " << cpp_strerror(-ret) << dendl;
599 return ret;
600 }
601 }
602
603 list<string>::iterator iter;
604 /* create zonegroups */
605 for (iter = regions.begin(); iter != regions.end(); ++iter)
606 {
607 ldpp_dout(dpp, 0) << __func__ << " Converting " << *iter << dendl;
608 /* check to see if we don't have already a zonegroup with this name */
609 RGWZoneGroup new_zonegroup(*iter);
610 ret = new_zonegroup.init(dpp, cct , sysobj_svc, y);
611 if (ret == 0 && new_zonegroup.get_id() != *iter) {
612 ldpp_dout(dpp, 0) << __func__ << " zonegroup "<< *iter << " already exists id " << new_zonegroup.get_id () <<
613 " skipping conversion " << dendl;
614 continue;
615 }
616 RGWZoneGroup zonegroup(*iter);
617 zonegroup.set_id(*iter);
618 int ret = zonegroup.init(dpp, cct, sysobj_svc, y, true, true);
619 if (ret < 0) {
620 ldpp_dout(dpp, 0) << __func__ << " failed init zonegroup: ret "<< ret << " " << cpp_strerror(-ret) << dendl;
621 return ret;
622 }
623 zonegroup.realm_id = realm->get_id();
624 /* fix default region master zone */
625 if (*iter == default_zonegroup_name && zonegroup.master_zone.empty()) {
626 ldpp_dout(dpp, 0) << __func__ << " Setting default zone as master for default region" << dendl;
627 zonegroup.master_zone = default_zone_name;
628 }
629 ret = zonegroup.update(dpp, y);
630 if (ret < 0 && ret != -EEXIST) {
631 ldpp_dout(dpp, 0) << __func__ << " failed to update zonegroup " << *iter << ": ret "<< ret << " " << cpp_strerror(-ret)
632 << dendl;
633 return ret;
634 }
635 ret = zonegroup.update_name(dpp, y);
636 if (ret < 0 && ret != -EEXIST) {
637 ldpp_dout(dpp, 0) << __func__ << " failed to update_name for zonegroup " << *iter << ": ret "<< ret << " " << cpp_strerror(-ret)
638 << dendl;
639 return ret;
640 }
641 if (zonegroup.get_name() == default_region) {
642 ret = zonegroup.set_as_default(dpp, y);
643 if (ret < 0) {
644 ldpp_dout(dpp, 0) << __func__ << " failed to set_as_default " << *iter << ": ret "<< ret << " " << cpp_strerror(-ret)
645 << dendl;
646 return ret;
647 }
648 }
649 for (auto iter = zonegroup.zones.begin(); iter != zonegroup.zones.end();
650 ++iter) {
651 ldpp_dout(dpp, 0) << __func__ << " Converting zone" << iter->first << dendl;
652 RGWZoneParams zoneparams(iter->first, iter->second.name);
653 zoneparams.set_id(iter->first.id);
654 zoneparams.realm_id = realm->get_id();
655 ret = zoneparams.init(dpp, cct, sysobj_svc, y);
656 if (ret < 0 && ret != -ENOENT) {
657 ldpp_dout(dpp, 0) << __func__ << " failed to init zoneparams " << iter->first << ": " << cpp_strerror(-ret) << dendl;
658 return ret;
659 } else if (ret == -ENOENT) {
660 ldpp_dout(dpp, 0) << __func__ << " zone is part of another cluster " << iter->first << " skipping " << dendl;
661 continue;
662 }
663 zonegroup.realm_id = realm->get_id();
664 ret = zoneparams.update(dpp, y);
665 if (ret < 0 && ret != -EEXIST) {
666 ldpp_dout(dpp, 0) << __func__ << " failed to update zoneparams " << iter->first << ": " << cpp_strerror(-ret) << dendl;
667 return ret;
668 }
669 ret = zoneparams.update_name(dpp, y);
670 if (ret < 0 && ret != -EEXIST) {
671 ldpp_dout(dpp, 0) << __func__ << " failed to init zoneparams " << iter->first << ": " << cpp_strerror(-ret) << dendl;
672 return ret;
673 }
674 }
675
676 if (!current_period->get_id().empty()) {
677 ret = current_period->add_zonegroup(dpp, zonegroup, y);
678 if (ret < 0) {
679 ldpp_dout(dpp, 0) << __func__ << " failed to add zonegroup to current_period: " << cpp_strerror(-ret) << dendl;
680 return ret;
681 }
682 }
683 }
684
685 if (!current_period->get_id().empty()) {
686 ret = current_period->update(dpp, y);
687 if (ret < 0) {
688 ldpp_dout(dpp, 0) << __func__ << " failed to update new period: " << cpp_strerror(-ret) << dendl;
689 return ret;
690 }
691 ret = current_period->store_info(dpp, false, y);
692 if (ret < 0) {
693 ldpp_dout(dpp, 0) << __func__ << " failed to store new period: " << cpp_strerror(-ret) << dendl;
694 return ret;
695 }
696 ret = current_period->reflect(dpp, y);
697 if (ret < 0) {
698 ldpp_dout(dpp, 0) << __func__ << " failed to update local objects: " << cpp_strerror(-ret) << dendl;
699 return ret;
700 }
701 }
702
703 for (auto const& iter : regions) {
704 RGWZoneGroup zonegroup(iter);
705 int ret = zonegroup.init(dpp, cct, sysobj_svc, y, true, true);
706 if (ret < 0) {
707 ldpp_dout(dpp, 0) << __func__ << " failed init zonegroup" << iter << ": ret "<< ret << " " << cpp_strerror(-ret) << dendl;
708 return ret;
709 }
710 ret = zonegroup.delete_obj(dpp, y, true);
711 if (ret < 0 && ret != -ENOENT) {
712 ldpp_dout(dpp, 0) << __func__ << " failed to delete region " << iter << ": ret "<< ret << " " << cpp_strerror(-ret)
713 << dendl;
714 return ret;
715 }
716 }
717
718 /* mark as converted */
719 ret = sysobj.wop()
720 .set_exclusive(true)
721 .write(dpp, bl, y);
722 if (ret < 0 ) {
723 ldpp_dout(dpp, 0) << __func__ << " failed to mark cluster as converted: ret "<< ret << " " << cpp_strerror(-ret)
724 << dendl;
725 return ret;
726 }
727
728 return 0;
729}

753
754 map<string, RGWZoneGroup>::const_iterator iter =
755 current_period->get_map().zonegroups.find(zonegroup->get_id());
756
757 if (iter != current_period->get_map().zonegroups.end()) {

773 if (zg.zones.size() == 0)

783 if (ret < 0) {
784 ldpp_dout(dpp, 0) << "error updating zonegroup : " << cpp_strerror(-ret) << dendl;

787 } else {
788 RGWZoneGroup fixed_zg(zg.get_id(),zg.get_name());

790 if (ret < 0) {
791 ldpp_dout(dpp, 0) << "error initializing zonegroup : " << cpp_strerror(-ret) << dendl;
792 return ret;
793 }
794 fixed_zg.master_zone = master->second.id;
795 ret = fixed_zg.update(dpp, y);
796 if (ret < 0) {
797 ldpp_dout(dpp, 0) << "error initializing zonegroup : " << cpp_strerror(-ret) << dendl;
798 return ret;
799 }
800 }
801 } else {
802 ldpp_dout(dpp, 0) << "zonegroup " << zg.get_name() << " missing zone for master_zone=" <<

813 }
814
815 return 0;

881int RGWSI_Zone::convert_regionmap(const DoutPrefixProvider *dpp, optional_yield y)
882{
883 RGWZoneGroupMap zonegroupmap;
884
885 string pool_name = cct->_conf->rgw_zone_root_pool;
886 if (pool_name.empty()) {
887 pool_name = RGW_DEFAULT_ZONE_ROOT_POOL;
888 }
889 string oid = region_map_oid;
890
891 rgw_pool pool(pool_name);
892 bufferlist bl;
893
894 RGWSysObjectCtx obj_ctx = sysobj_svc->init_obj_ctx();
895 RGWSysObj sysobj = sysobj_svc->get_obj(obj_ctx, rgw_raw_obj(pool, oid));
896
897 int ret = sysobj.rop().read(dpp, &bl, y);
898 if (ret < 0 && ret != -ENOENT) {
899 return ret;
900 } else if (ret == -ENOENT) {
901 return 0;
902 }
903
904 try {
905 auto iter = bl.cbegin();
906 decode(zonegroupmap, iter);
907 } catch (buffer::error& err) {
908 ldpp_dout(dpp, 0) << "error decoding regionmap from " << pool << ":" << oid << dendl;
909 return -EIO;
910 }
911
912 for (map<string, RGWZoneGroup>::iterator iter = zonegroupmap.zonegroups.begin();
913 iter != zonegroupmap.zonegroups.end(); ++iter) {
914 RGWZoneGroup& zonegroup = iter->second;
915 ret = zonegroup.init(dpp, cct, sysobj_svc, y, false);
916 ret = zonegroup.update(dpp, y);
917 if (ret < 0 && ret != -ENOENT) {
918 ldpp_dout(dpp, 0) << "Error could not update zonegroup " << zonegroup.get_name() << ": " <<
919 cpp_strerror(-ret) << dendl;
920 return ret;
921 } else if (ret == -ENOENT) {
922 ret = zonegroup.create(dpp, y);
923 if (ret < 0) {
924 ldpp_dout(dpp, 0) << "Error could not create " << zonegroup.get_name() << ": " <<
925 cpp_strerror(-ret) << dendl;
926 return ret;
927 }
928 }
929 }
930
931 current_period->set_user_quota(zonegroupmap.user_quota);
932 current_period->set_bucket_quota(zonegroupmap.bucket_quota);
933
934 // remove the region_map so we don't try to convert again
935 ret = sysobj.wop().remove(dpp, y);
936 if (ret < 0) {
937 ldpp_dout(dpp, 0) << "Error could not remove " << sysobj.get_obj()
938 << " after upgrading to zonegroup map: " << cpp_strerror(ret) << dendl;
939 return ret;
940 }
941
942 return 0;

1115int RGWSI_Zone::select_new_bucket_location(const DoutPrefixProvider *dpp, const RGWUserInfo& user_info, const string& zonegroup_id,
1116 const rgw_placement_rule& request_rule,
1117 rgw_placement_rule *pselected_rule_name, RGWZonePlacementInfo *rule_info,
1118 optional_yield y)
1119{
1120 /* first check that zonegroup exists within current period. */
1121 RGWZoneGroup zonegroup;
1122 int ret = get_zonegroup(zonegroup_id, zonegroup);
1123 if (ret < 0) {
1124 ldpp_dout(dpp, 0) << "could not find zonegroup " << zonegroup_id << " in current period" << dendl;
1125 return ret;
1126 }
1127
1128 const rgw_placement_rule *used_rule;
1129
1130 /* find placement rule. Hierarchy: request rule > user default rule > zonegroup default rule */
1131 std::map<std::string, RGWZoneGroupPlacementTarget>::const_iterator titer;
1132
1133 if (!request_rule.name.empty()) {
1134 used_rule = &request_rule;
1135 titer = zonegroup.placement_targets.find(request_rule.name);
1136 if (titer == zonegroup.placement_targets.end()) {
1137 ldpp_dout(dpp, 0) << "could not find requested placement id " << request_rule
1138 << " within zonegroup " << dendl;
1139 return -ERR_INVALID_LOCATION_CONSTRAINT;
1140 }
1141 } else if (!user_info.default_placement.name.empty()) {
1142 used_rule = &user_info.default_placement;
1143 titer = zonegroup.placement_targets.find(user_info.default_placement.name);
1144 if (titer == zonegroup.placement_targets.end()) {
1145 ldpp_dout(dpp, 0) << "could not find user default placement id " << user_info.default_placement
1146 << " within zonegroup " << dendl;
1147 return -ERR_INVALID_LOCATION_CONSTRAINT;
1148 }
1149 } else {
1150 if (zonegroup.default_placement.name.empty()) { // zonegroup default rule as fallback, it should not be empty.
1151 ldpp_dout(dpp, 0) << "misconfiguration, zonegroup default placement id should not be empty." << dendl;
1152 return -ERR_ZONEGROUP_DEFAULT_PLACEMENT_MISCONFIGURATION;
1153 } else {
1154 used_rule = &zonegroup.default_placement;
1155 titer = zonegroup.placement_targets.find(zonegroup.default_placement.name);
1156 if (titer == zonegroup.placement_targets.end()) {
1157 ldpp_dout(dpp, 0) << "could not find zonegroup default placement id " << zonegroup.default_placement
1158 << " within zonegroup " << dendl;
1159 return -ERR_INVALID_LOCATION_CONSTRAINT;
1160 }
1161 }
1162 }
1163
1164 /* now check tag for the rule, whether user is permitted to use rule */
1165 const auto& target_rule = titer->second;
1166 if (!target_rule.user_permitted(user_info.placement_tags)) {
1167 ldpp_dout(dpp, 0) << "user not permitted to use placement rule " << titer->first << dendl;
1168 return -EPERM;
1169 }
1170
1171 const string *storage_class = &request_rule.storage_class;
1172
1173 if (storage_class->empty()) {
1174 storage_class = &used_rule->storage_class;
1175 }
1176
1177 rgw_placement_rule rule(titer->first, *storage_class);
1178
1179 if (pselected_rule_name) {
1180 *pselected_rule_name = rule;
1181 }
1182
1183 return select_bucket_location_by_rule(dpp, rule, rule_info, y);
1184}

1186int RGWSI_Zone::select_bucket_location_by_rule(const DoutPrefixProvider *dpp, const rgw_placement_rule& location_rule, RGWZonePlacementInfo *rule_info, optional_yield y)
1187{
1188 if (location_rule.name.empty()) {
1189 /* we can only reach here if we're trying to set a bucket location from a bucket
1190 * created on a different zone, using a legacy / default pool configuration
1191 */

1249int RGWSI_Zone::select_legacy_bucket_placement(const DoutPrefixProvider *dpp, RGWZonePlacementInfo *rule_info,
1250 optional_yield y)
1251{
1252 bufferlist map_bl;
1253 map<string, bufferlist> m;
1254 string pool_name;
1255 bool write_map = false;
1256
1257 rgw_raw_obj obj(zone_params->domain_root, avail_pools);
1258
1259 auto obj_ctx = sysobj_svc->init_obj_ctx();
1260 auto sysobj = obj_ctx.get_obj(obj);
1261
1262 int ret = sysobj.rop().read(dpp, &map_bl, y);
1263 if (ret < 0) {
1264 goto read_omap;
1265 }
1266
1267 try {
1268 auto iter = map_bl.cbegin();
1269 decode(m, iter);
1270 } catch (buffer::error& err) {
1271 ldpp_dout(dpp, 0) << "ERROR: couldn't decode avail_pools" << dendl;
1272 }
1273
1274read_omap:
1275 if (m.empty()) {
1276 ret = sysobj.omap().get_all(dpp, &m, y);
1277
1278 write_map = true;
1279 }
1280
1281 if (ret < 0 || m.empty()) {
1282 vector<rgw_pool> pools;
1283 string s = string("default.") + default_storage_pool_suffix;
1284 pools.push_back(rgw_pool(s));
1285 vector<int> retcodes;
1286 bufferlist bl;
1287 ret = rados_svc->pool().create(dpp, pools, &retcodes);
1288 if (ret < 0)
1289 return ret;
1290 ret = sysobj.omap().set(dpp, s, bl, y);
1291 if (ret < 0)
1292 return ret;
1293 m[s] = bl;
1294 }
1295
1296 if (write_map) {
1297 bufferlist new_bl;
1298 encode(m, new_bl);
1299 ret = sysobj.wop().write(dpp, new_bl, y);
1300 if (ret < 0) {
1301 ldpp_dout(dpp, 0) << "WARNING: could not save avail pools map info ret=" << ret << dendl;
1302 }
1303 }
1304
1305 auto miter = m.begin();
1306 if (m.size() > 1) {
1307 // choose a pool at random
1308 auto r = ceph::util::generate_random_number<size_t>(0, m.size() - 1);
1309 std::advance(miter, r);
1310 }
1311 pool_name = miter->first;
1312
1313 rgw_pool pool = pool_name;
1314
1315 rule_info->storage_classes.set_storage_class(RGW_STORAGE_CLASS_STANDARD, &pool, nullptr);
1316 rule_info->data_extra_pool = pool_name;
1317 rule_info->index_pool = pool_name;
1318 rule_info->index_type = rgw::BucketIndexType::Normal;
1319
1320 return 0;
1321}
