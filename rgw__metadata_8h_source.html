199 static bool check_versions(bool exists,
200 const obj_version& ondisk, const real_time& ondisk_time,
201 const obj_version& incoming, const real_time& incoming_time,
202 RGWMDLogSyncType sync_mode) {
203 switch (sync_mode) {
204 case APPLY_UPDATES:
205 if ((ondisk.tag != incoming.tag) ||
206 (ondisk.ver >= incoming.ver))
207 return false;
208 break;
209 case APPLY_NEWER:
210 if (ondisk_time >= incoming_time)
211 return false;
212 break;
213 case APPLY_EXCLUSIVE:
214 if (exists)
215 return false;
216 break;
217 case APPLY_ALWAYS: //deliberate fall-thru -- we always apply!
218 default: break;
219 }
220 return true;
221 }
