144 void decode(bufferlist::const_iterator &p) {
145 DECODE_START_LEGACY_COMPAT_LEN(14, 5, 5, p);
146 decode(object_owner.id, p);
147 if (struct_v > 3)
148 decode(bucket_owner.id, p);
149 decode(bucket, p);
150 decode(time, p);
151 decode(remote_addr, p);
152 decode(user, p);
153 decode(obj.name, p);
154 decode(op, p);
155 decode(uri, p);
156 decode(http_status, p);
157 decode(error_code, p);
158 decode(bytes_sent, p);
159 decode(obj_size, p);
160 decode(total_time, p);
161 decode(user_agent, p);
162 decode(referrer, p);
163 if (struct_v >= 2)
164 decode(bytes_received, p);
165 else
166 bytes_received = 0;
167
168 if (struct_v >= 3) {
169 if (struct_v <= 5) {
170 uint64_t id;
171 decode(id, p);
172 char buf[32];
173 snprintf(buf, sizeof(buf), "%" PRIu64, id);
174 bucket_id = buf;
175 } else {
176 decode(bucket_id, p);
177 }
178 } else {
179 bucket_id = "";
180 }
181 if (struct_v >= 7) {
182 decode(obj, p);
183 }
184 if (struct_v >= 8) {
185 decode(object_owner, p);
186 decode(bucket_owner, p);
187 }
188 if (struct_v >= 9) {
189 decode(x_headers, p);
190 }
191 if (struct_v >= 10) {
192 decode(trans_id, p);
193 }
194 if (struct_v >= 11) {
195 decode(token_claims, p);
196 }
197 if (struct_v >= 12) {
198 decode(identity_type, p);
199 }
200 if (struct_v >= 13) {
201 decode(access_key_id, p);
202 decode(subuser, p);
203 decode(temp_url, p);
204 }
205 if (struct_v >= 14) {
206 decode(delete_multi_obj_meta, p);
207 }
208 DECODE_FINISH(p);
209 }
