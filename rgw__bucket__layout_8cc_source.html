139void decode(BucketLayout& l, bufferlist::const_iterator& bl)
140{
141 DECODE_START(2, bl);
142 decode(l.resharding, bl);
143 decode(l.current_index, bl);
144 decode(l.target_index, bl);
145 if (struct_v < 2) {
146 l.logs.clear();
147 // initialize the log layout to match the current index layout
148 if (l.current_index.layout.type == BucketIndexType::Normal) {
149 const auto gen = l.current_index.gen;
150 const auto& index = l.current_index.layout.normal;
151 l.logs.push_back(log_layout_from_index(gen, index));
152 }
153 } else {
154 decode(l.logs, bl);
155 }
156 DECODE_FINISH(bl);
157}
