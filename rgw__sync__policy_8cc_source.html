16bool rgw_sync_pipe_filter_tag::from_str(const string& s)
17{
18 if (s.empty()) {
19 return false;
20 }
21
22 auto pos = s.find('=');
23 if (pos == string::npos) {
24 key = s;
25 return true;
26 }
27
28 key = s.substr(0, pos);
29 if (pos < s.size() - 1) {
30 value = s.substr(pos + 1);
31 }
32
33 return true;
34}
35
36bool rgw_sync_pipe_filter_tag::operator==(const string& s) const
37{
38 if (s.empty()) {
39 return false;
40 }
41
42 auto pos = s.find('=');
43 if (pos == string::npos) {
44 return value.empty() && (s == key);
45 }
46
47 return s.compare(0, pos, s) == 0 &&
48 s.compare(pos + 1, s.size() - pos - 1, value) == 0;
49}

95bool rgw_sync_pipe_filter::is_subset_of(const rgw_sync_pipe_filter& f) const
96{
97 if (f.prefix) {
98 if (!prefix) {
99 return false;
100 }
101 /* f.prefix exists, and this->prefix is either equal or bigger,
102 * therefore this->prefix also set */
103
104 if (!boost::starts_with(*prefix, *f.prefix)) {
105 return false;
106 }
107 }
108
109 /* prefix is subset, now check tags. All our tags should exist in f.tags */
110
111 for (auto& t : tags) {
112 if (f.tags.find(t) == f.tags.end()) {
113 return false;
114 }
115 }
116
117 return true;
118}

379void rgw_sync_data_flow_group::remove_symmetrical(const string& flow_id, std::optional<std::vector<rgw_zone_id> > zones)
380{
381 if (symmetrical.empty()) {
382 return;
383 }
384
385 auto& groups = symmetrical;
386
387 auto iter = groups.begin();
388
389 for (; iter != groups.end(); ++iter) {
390 if (iter->id == flow_id) {
391 if (!zones) {
392 groups.erase(iter);
393 if (groups.empty()) {
394 symmetrical.clear();
395 }
396 return;
397 }
398 break;
399 }
400 }
401
402 if (iter == groups.end()) {
403 return;
404 }
405
406 auto& flow_group = *iter;
407
408 for (auto& z : *zones) {
409 flow_group.zones.erase(z);
410 }
411
412 if (flow_group.zones.empty()) {
413 groups.erase(iter);
414 }
415 if (groups.empty()) {
416 symmetrical.clear();
417 }
