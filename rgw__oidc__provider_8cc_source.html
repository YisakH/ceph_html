34int RGWOIDCProvider::get_tenant_url_from_arn(string& tenant, string& url)
35{
36 auto provider_arn = rgw::ARN::parse(arn);
37 if (!provider_arn) {
38 return -EINVAL;
39 }
40 url = provider_arn->resource;
41 tenant = provider_arn->account;
42 auto pos = url.find("oidc-provider/");
43 if (pos != std::string::npos) {
44 url.erase(pos, 14);
45 }
46 return 0;
47}

49int RGWOIDCProvider::create(const DoutPrefixProvider *dpp, bool exclusive, optional_yield y)
50{
51 int ret;
52
53 if (! validate_input(dpp)) {
54 return -EINVAL;
55 }
56
57 string idp_url = url_remove_prefix(provider_url);
58
59 /* check to see the name is not used */
60 ret = read_url(dpp, idp_url, tenant);
61 if (exclusive && ret == 0) {
62 ldpp_dout(dpp, 0) << "ERROR: url " << provider_url << " already in use"
63 << id << dendl;
64 return -EEXIST;
65 } else if ( ret < 0 && ret != -ENOENT) {
66 ldpp_dout(dpp, 0) << "failed reading provider url " << provider_url << ": "
67 << cpp_strerror(-ret) << dendl;
68 return ret;
69 }
70
71 //arn
72 arn = oidc_arn_prefix + tenant + ":oidc-provider/" + idp_url;
73
74 // Creation time
75 real_clock::time_point t = real_clock::now();
76
77 struct timeval tv;
78 real_clock::to_timeval(t, tv);
79
80 char buf[30];
81 struct tm result;
82 gmtime_r(&tv.tv_sec, &result);
83 strftime(buf,30,"%Y-%m-%dT%H:%M:%S", &result);
84 sprintf(buf + strlen(buf),".%dZ",(int)tv.tv_usec/1000);
85 creation_date.assign(buf, strlen(buf));
86
87 ret = store_url(dpp, idp_url, exclusive, y);
88 if (ret < 0) {
89 ldpp_dout(dpp, 0) << "ERROR: storing role info in OIDC pool: "
90 << provider_url << ": " << cpp_strerror(-ret) << dendl;
91 return ret;
92 }
93
94 return 0;
95}

97int RGWOIDCProvider::get(const DoutPrefixProvider *dpp)
98{
99 string url, tenant;
100 auto ret = get_tenant_url_from_arn(tenant, url);
101 if (ret < 0) {
102 ldpp_dout(dpp, 0) << "ERROR: failed to parse arn" << dendl;
103 return -EINVAL;
104 }
105
106 if (this->tenant != tenant) {
107 ldpp_dout(dpp, 0) << "ERROR: tenant in arn doesn't match that of user " << this->tenant << ", "
108 << tenant << ": " << dendl;
109 return -EINVAL;
110 }
111
112 ret = read_url(dpp, url, tenant);
113 if (ret < 0) {
114 return ret;
115 }
116
117 return 0;
118}

146bool RGWOIDCProvider::validate_input(const DoutPrefixProvider *dpp)
147{
148 if (provider_url.length() > MAX_OIDC_URL_LEN) {
149 ldpp_dout(dpp, 0) << "ERROR: Invalid length of url " << dendl;
150 return false;
151 }
152 if (client_ids.size() > MAX_OIDC_NUM_CLIENT_IDS) {
153 ldpp_dout(dpp, 0) << "ERROR: Invalid number of client ids " << dendl;
154 return false;
155 }
156
157 for (auto& it : client_ids) {
158 if (it.length() > MAX_OIDC_CLIENT_ID_LEN) {
159 return false;
160 }
161 }
162
163 if (thumbprints.size() > MAX_OIDC_NUM_THUMBPRINTS) {
164 ldpp_dout(dpp, 0) << "ERROR: Invalid number of thumbprints " << thumbprints.size() << dendl;
165 return false;
166 }
167
168 for (auto& it : thumbprints) {
169 if (it.length() > MAX_OIDC_THUMBPRINT_LEN) {
170 return false;
171 }
172 }
173
174 return true;
175}
