100/*
101 The manifest defines a set of rules for structuring the object parts.
102 There are a few terms to note:
103 - head: the head part of the object, which is the part that contains
104 the first chunk of data. An object might not have a head (as in the
105 case of multipart-part objects).
106 - stripe: data portion of a single rgw object that resides on a single
107 rados object.
108 - part: a collection of stripes that make a contiguous part of an
109 object. A regular object will only have one part (although might have
110 many stripes), a multipart object might have many parts. Each part
111 has a fixed stripe size, although the last stripe of a part might
112 be smaller than that. Consecutive parts may be merged if their stripe
113 value is the same.
114*/
115
116struct RGWObjManifestRule {
117 uint32_t start_part_num;
118 uint64_t start_ofs;
119 uint64_t part_size; /* each part size, 0 if there's no part size, meaning it's unlimited */
120 uint64_t stripe_max_size; /* underlying obj max size */
121 std::string override_prefix;
122
123 RGWObjManifestRule() : start_part_num(0), start_ofs(0), part_size(0), stripe_max_size(0) {}
124 RGWObjManifestRule(uint32_t _start_part_num, uint64_t _start_ofs, uint64_t _part_size, uint64_t _stripe_max_size) :
125 start_part_num(_start_part_num), start_ofs(_start_ofs), part_size(_part_size), stripe_max_size(_stripe_max_size) {}
126
127 void encode(bufferlist& bl) const {
128 ENCODE_START(2, 1, bl);
129 encode(start_part_num, bl);
130 encode(start_ofs, bl);
131 encode(part_size, bl);
132 encode(stripe_max_size, bl);
133 encode(override_prefix, bl);
134 ENCODE_FINISH(bl);
135 }
136
137 void decode(bufferlist::const_iterator& bl) {
138 DECODE_START(2, bl);
139 decode(start_part_num, bl);
140 decode(start_ofs, bl);
141 decode(part_size, bl);
142 decode(stripe_max_size, bl);
143 if (struct_v >= 2)
144 decode(override_prefix, bl);
145 DECODE_FINISH(bl);
146 }
147 void dump(Formatter *f) const;
148};

252 void encode(bufferlist& bl) const {
253 ENCODE_START(8, 6, bl);
254 encode(obj_size, bl);
255 encode(objs, bl);
256 encode(explicit_objs, bl);
257 encode(obj, bl);
258 encode(head_size, bl);
259 encode(max_head_size, bl);
260 encode(prefix, bl);
261 encode(rules, bl);
262 bool encode_tail_bucket = !(tail_placement.bucket == obj.bucket);
263 encode(encode_tail_bucket, bl);
264 if (encode_tail_bucket) {
265 encode(tail_placement.bucket, bl);
266 }
267 bool encode_tail_instance = (tail_instance != obj.key.instance);
268 encode(encode_tail_instance, bl);
269 if (encode_tail_instance) {
270 encode(tail_instance, bl);
271 }
272 encode(head_placement_rule, bl);
273 encode(tail_placement.placement_rule, bl);

279 void decode(bufferlist::const_iterator& bl) {
280 DECODE_START_LEGACY_COMPAT_LEN_32(7, 2, 2, bl);
281 decode(obj_size, bl);
282 decode(objs, bl);
283 if (struct_v >= 3) {
284 decode(explicit_objs, bl);
285 decode(obj, bl);
286 decode(head_size, bl);
287 decode(max_head_size, bl);
288 decode(prefix, bl);
289 decode(rules, bl);
290 } else {
291 explicit_objs = true;
292 if (!objs.empty()) {
293 std::map<uint64_t, RGWObjManifestPart>::iterator iter = objs.begin();
294 obj = iter->second.loc;
295 head_size = iter->second.size;
296 max_head_size = head_size;
297 }
298 }
299
300 if (explicit_objs && head_size > 0 && !objs.empty()) {
301 /* patch up manifest due to issue 16435:
302 * the first object in the explicit objs list might not be the one we need to access, use the
303 * head object instead if set. This would happen if we had an old object that was created
304 * when the explicit objs manifest was around, and it got copied.
305 */
306 rgw_obj& obj_0 = objs[0].loc;
307 if (!obj_0.get_oid().empty() && obj_0.key.ns.empty()) {
308 objs[0].loc = obj;
309 objs[0].size = head_size;
310 }
311 }
312
313 if (struct_v >= 4) {
314 if (struct_v < 6) {
315 decode(tail_placement.bucket, bl);
316 } else {
317 bool need_to_decode;
318 decode(need_to_decode, bl);
319 if (need_to_decode) {
320 decode(tail_placement.bucket, bl);
321 } else {
322 tail_placement.bucket = obj.bucket;
323 }
324 }
325 }
326
327 if (struct_v >= 5) {
328 if (struct_v < 6) {
329 decode(tail_instance, bl);
330 } else {
331 bool need_to_decode;
332 decode(need_to_decode, bl);
333 if (need_to_decode) {
334 decode(tail_instance, bl);
335 } else {
336 tail_instance = obj.key.instance;
337 }
338 }
339 } else { // old object created before 'tail_instance' field added to manifest
340 tail_instance = obj.key.instance;
341 }
342
343 if (struct_v >= 7) {
344 decode(head_placement_rule, bl);
345 decode(tail_placement.placement_rule, bl);
346 }
347
348 if (struct_v >= 8) {
349 decode(tier_type, bl);
350 decode(tier_config, bl);
351 }
352
353 DECODE_FINISH(bl);
354 }
