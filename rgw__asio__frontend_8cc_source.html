68template <typename Stream>
69class StreamIO : public rgw::asio::ClientIO {
70 CephContext* const cct;
71 Stream& stream;
72 timeout_timer& timeout;
73 yield_context yield;
74 parse_buffer& buffer;
75 boost::system::error_code fatal_ec;
76 public:
77 StreamIO(CephContext *cct, Stream& stream, timeout_timer& timeout,
78 rgw::asio::parser_type& parser, yield_context yield,
79 parse_buffer& buffer, bool is_ssl,
80 const tcp::endpoint& local_endpoint,
81 const tcp::endpoint& remote_endpoint)
82 : ClientIO(parser, is_ssl, local_endpoint, remote_endpoint),
83 cct(cct), stream(stream), timeout(timeout), yield(yield),
84 buffer(buffer)
85 {}
86
87 boost::system::error_code get_fatal_error_code() const { return fatal_ec; }
88
89 size_t write_data(const char* buf, size_t len) override {
90 boost::system::error_code ec;
91 timeout.start();
92 auto bytes = boost::asio::async_write(stream, boost::asio::buffer(buf, len),
93 yield[ec]);
94 timeout.cancel();
95 if (ec) {
96 ldout(cct, 4) << "write_data failed: " << ec.message() << dendl;
97 if (ec == boost::asio::error::broken_pipe) {
98 boost::system::error_code ec_ignored;
99 stream.lowest_layer().shutdown(tcp_socket::shutdown_both, ec_ignored);
100 }
101 if (!fatal_ec) {
102 fatal_ec = ec;
103 }
104 throw rgw::io::Exception(ec.value(), std::system_category());
105 }
106 return bytes;
107 }
108
109 size_t recv_body(char* buf, size_t max) override {
110 auto& message = parser.get();
111 auto& body_remaining = message.body();
112 body_remaining.data = buf;
113 body_remaining.size = max;
114
115 while (body_remaining.size && !parser.is_done()) {
116 boost::system::error_code ec;
117 timeout.start();
118 http::async_read_some(stream, buffer, parser, yield[ec]);
119 timeout.cancel();
120 if (ec == http::error::need_buffer) {
121 break;
122 }
123 if (ec) {
124 ldout(cct, 4) << "failed to read body: " << ec.message() << dendl;
125 if (!fatal_ec) {
126 fatal_ec = ec;
127 }
128 throw rgw::io::Exception(ec.value(), std::system_category());
129 }
130 }
131 return max - body_remaining.size;
132 }
133};

146// log an http header value or '-' if it's missing
147struct log_header {
148 const http::fields& fields;
149 http::field field;
150 std::string_view quote;
151 log_header(const http::fields& fields, http::field field,
152 std::string_view quote = "")
153 : fields(fields), field(field), quote(quote) {}
154};
155std::ostream& operator<<(std::ostream& out, const log_header& h) {
156 auto p = h.fields.find(h.field);
157 if (p == h.fields.end()) {
158 return out << '-';
159 }
160 return out << h.quote << p->value() << h.quote;

188template <typename Stream>
189void handle_connection(boost::asio::io_context& context,
190 RGWProcessEnv& env, Stream& stream,
191 timeout_timer& timeout, size_t header_limit,
192 parse_buffer& buffer, bool is_ssl,
193 SharedMutex& pause_mutex,
194 rgw::dmclock::Scheduler *scheduler,
195 boost::system::error_code& ec,
196 yield_context yield)
197{
198 // don't impose a limit on the body, since we read it in pieces
199 static constexpr size_t body_limit = std::numeric_limits<size_t>::max();
200
201 auto cct = env.store->ctx();
202
203 // read messages from the stream until eof
204 for (;;) {
205 // configure the parser
206 rgw::asio::parser_type parser;
207 parser.header_limit(header_limit);
208 parser.body_limit(body_limit);
209 timeout.start();
210 // parse the header
211 http::async_read_header(stream, buffer, parser, yield[ec]);
212 timeout.cancel();
213 if (ec == boost::asio::error::connection_reset ||
214 ec == boost::asio::error::bad_descriptor ||
215 ec == boost::asio::error::operation_aborted ||
216#ifdef WITH_RADOSGW_BEAST_OPENSSL
217 ec == ssl::error::stream_truncated ||
218#endif
219 ec == http::error::end_of_stream) {
220 ldout(cct, 20) << "failed to read header: " << ec.message() << dendl;
221 return;
222 }
223 auto& message = parser.get();
224 if (ec) {
225 ldout(cct, 1) << "failed to read header: " << ec.message() << dendl;
226 http::response<http::empty_body> response;
227 response.result(http::status::bad_request);
228 response.version(message.version() == 10 ? 10 : 11);
229 response.prepare_payload();
230 timeout.start();
231 http::async_write(stream, response, yield[ec]);
232 timeout.cancel();
233 if (ec) {
234 ldout(cct, 5) << "failed to write response: " << ec.message() << dendl;
235 }
236 ldout(cct, 1) << "====== req done http_status=400 ======" << dendl;
237 return;
238 }
239
240 bool expect_continue = (message[http::field::expect] == "100-continue");
241
242 {
243 auto lock = pause_mutex.async_lock_shared(yield[ec]);
244 if (ec == boost::asio::error::operation_aborted) {
245 return;
246 } else if (ec) {
247 ldout(cct, 1) << "failed to lock: " << ec.message() << dendl;
248 return;
249 }
250
251 // process the request
252 RGWRequest req{env.store->get_new_req_id()};
253
254 auto& socket = stream.lowest_layer();
255 const auto& remote_endpoint = socket.remote_endpoint(ec);
256 if (ec) {
257 ldout(cct, 1) << "failed to connect client: " << ec.message() << dendl;
258 return;
259 }
260
261 StreamIO real_client{cct, stream, timeout, parser, yield, buffer,
262 is_ssl, socket.local_endpoint(),
263 remote_endpoint};
264
265 auto real_client_io = rgw::io::add_reordering(
266 rgw::io::add_buffering(cct,
267 rgw::io::add_chunking(
268 rgw::io::add_conlen_controlling(
269 &real_client))));
270 RGWRestfulIO client(cct, &real_client_io);
271 optional_yield y = null_yield;
272 if (cct->_conf->rgw_beast_enable_async) {
273 y = optional_yield{context, yield};
274 }
275 int http_ret = 0;
276 string user = "-";
277 const auto started = ceph::coarse_real_clock::now();
278 ceph::coarse_real_clock::duration latency{};
279 process_request(env.store, env.rest, &req, env.uri_prefix,
280 *env.auth_registry, &client, env.olog, y,
281 scheduler, &user, &latency,
282 env.ratelimiting->get_active(),
283 &http_ret);
284
285 if (cct->_conf->subsys.should_gather(dout_subsys, 1)) {
286 // access log line elements begin per Apache Combined Log Format with additions following
287 ldout(cct, 1) << "beast: " << std::hex << &req << std::dec << ": "
288 << remote_endpoint.address() << " - " << user << " [" << log_apache_time{started} << "] \""
289 << message.method_string() << ' ' << message.target() << ' '
290 << http_version{message.version()} << "\" " << http_ret << ' '
291 << client.get_bytes_sent() + client.get_bytes_received() << ' '
292 << log_header{message, http::field::referer, "\""} << ' '
293 << log_header{message, http::field::user_agent, "\""} << ' '
294 << log_header{message, http::field::range} << " latency="
295 << latency << dendl;
296 }
297
298 // process_request() can't distinguish between connection errors and
299 // http/s3 errors, so check StreamIO for fatal connection errors
300 ec = real_client.get_fatal_error_code();
301 if (ec) {
302 return;
303 }
304
305 if (real_client.sent_100_continue()) {
306 expect_continue = false;
307 }
308 }
309
310 if (!parser.keep_alive()) {
311 return;
312 }
313
314 // if we failed before reading the entire message, discard any remaining
315 // bytes before reading the next
316 while (!expect_continue && !parser.is_done()) {
317 static std::array<char, 1024> discard_buffer;
318
319 auto& body = parser.get().body();
320 body.size = discard_buffer.size();
321 body.data = discard_buffer.data();
322
323 timeout.start();
324 http::async_read_some(stream, buffer, parser, yield[ec]);
325 timeout.cancel();
326 if (ec == http::error::need_buffer) {
327 continue;
328 }
329 if (ec == boost::asio::error::connection_reset) {
330 return;
331 }
332 if (ec) {
333 ldout(cct, 5) << "failed to discard unread message: "
334 << ec.message() << dendl;
335 return;
336 }
337 }
338 }
339}

361 List connections;
362 std::mutex mutex;
363
364 void remove(Connection& c) {
365 std::lock_guard lock{mutex};
366 if (c.is_linked()) {
367 connections.erase(List::s_iterator_to(c));
368 }
369 }
370 public:
371 class Guard {
372 ConnectionList *list;
373 Connection *conn;
374 public:
375 Guard(ConnectionList *list, Connection *conn) : list(list), conn(conn) {}
376 ~Guard() { list->remove(*conn); }
377 };
378 [[nodiscard]] Guard add(Connection& conn) {
379 std::lock_guard lock{mutex};
380 connections.push_back(conn);
381 return Guard{this, &conn};
382 }

384 std::lock_guard lock{mutex};
385 for (auto& conn : connections) {
386 conn.socket.close(ec);
387 }
388 connections.clear();
389 }

469unsigned short parse_port(const char *input, boost::system::error_code& ec)
470{
471 char *end = nullptr;
472 auto port = std::strtoul(input, &end, 10);
473 if (port > std::numeric_limits<unsigned short>::max()) {
474 ec.assign(ERANGE, boost::system::system_category());
475 } else if (port == 0 && end == input) {
476 ec.assign(EINVAL, boost::system::system_category());
477 }
478 return port;
479}

481tcp::endpoint parse_endpoint(boost::asio::string_view input,
482 unsigned short default_port,
483 boost::system::error_code& ec)
484{
485 tcp::endpoint endpoint;
486
487 if (input.empty()) {
488 ec = boost::asio::error::invalid_argument;
489 return endpoint;
490 }
491
492 if (input[0] == '[') { // ipv6
493 const size_t addr_begin = 1;
494 const size_t addr_end = input.find(']');
495 if (addr_end == input.npos) { // no matching ]
496 ec = boost::asio::error::invalid_argument;
497 return endpoint;
498 }
499 if (addr_end + 1 < input.size()) {
500 // :port must must follow [ipv6]
501 if (input[addr_end + 1] != ':') {
502 ec = boost::asio::error::invalid_argument;
503 return endpoint;
504 } else {
505 auto port_str = input.substr(addr_end + 2);
506 endpoint.port(parse_port(port_str.data(), ec));
507 }
508 } else {
509 endpoint.port(default_port);
510 }
511 auto addr = input.substr(addr_begin, addr_end - addr_begin);
512 endpoint.address(boost::asio::ip::make_address_v6(addr, ec));
513 } else { // ipv4
514 auto colon = input.find(':');
515 if (colon != input.npos) {
516 auto port_str = input.substr(colon + 1);
517 endpoint.port(parse_port(port_str.data(), ec));
518 if (ec) {
519 return endpoint;
520 }
521 } else {
522 endpoint.port(default_port);
523 }
524 auto addr = input.substr(0, colon);
525 endpoint.address(boost::asio::ip::make_address_v4(addr, ec));
526 }
527 return endpoint;
528}

530static int drop_privileges(CephContext *ctx)
531{
532 uid_t uid = ctx->get_set_uid();
533 gid_t gid = ctx->get_set_gid();
534 std::string uid_string = ctx->get_set_uid_string();
535 std::string gid_string = ctx->get_set_gid_string();
536 if (gid && setgid(gid) != 0) {
537 int err = errno;
538 ldout(ctx, -1) << "unable to setgid " << gid << ": " << cpp_strerror(err) << dendl;
539 return -err;
540 }
541 if (uid && setuid(uid) != 0) {
542 int err = errno;
543 ldout(ctx, -1) << "unable to setuid " << uid << ": " << cpp_strerror(err) << dendl;
544 return -err;
545 }
546 if (uid && gid) {
547 ldout(ctx, 0) << "set uid:gid to " << uid << ":" << gid
548 << " (" << uid_string << ":" << gid_string << ")" << dendl;
549 }
550 return 0;
551}

553int AsioFrontend::init()
554{
555 boost::system::error_code ec;
556 auto& config = conf->get_config_map();
557
558// Setting global timeout
559 auto timeout = config.find("request_timeout_ms");
560 if (timeout != config.end()) {
561 auto timeout_number = ceph::parse<uint64_t>(timeout->second);
562 if (timeout_number) {
563 request_timeout = std::chrono::milliseconds(*timeout_number);
564 } else {
565 lderr(ctx()) << "WARNING: invalid value for request_timeout_ms: "
566 << timeout->second << " setting it to the default value: "
567 << REQUEST_TIMEOUT << dendl;
568 }
569 }
570
571 auto max_header_size = config.find("max_header_size");
572 if (max_header_size != config.end()) {
573 auto limit = ceph::parse<uint64_t>(max_header_size->second);
574 if (!limit) {
575 lderr(ctx()) << "WARNING: invalid value for max_header_size: "
576 << max_header_size->second << ", using the default value: "
577 << header_limit << dendl;
578 } else if (*limit > parse_buffer_size) { // can't exceed parse buffer size
579 header_limit = parse_buffer_size;
580 lderr(ctx()) << "WARNING: max_header_size " << max_header_size->second
581 << " capped at maximum value " << header_limit << dendl;
582 } else {
583 header_limit = *limit;
584 }
585 }
586
587#ifdef WITH_RADOSGW_BEAST_OPENSSL
588 int r = init_ssl();
589 if (r < 0) {
590 return r;
591 }
592#endif
593
594 // parse endpoints
595 auto ports = config.equal_range("port");
596 for (auto i = ports.first; i != ports.second; ++i) {
597 auto port = parse_port(i->second.c_str(), ec);
598 if (ec) {
599 lderr(ctx()) << "failed to parse port=" << i->second << dendl;
600 return -ec.value();
601 }
602 listeners.emplace_back(context);
603 listeners.back().endpoint.port(port);
604
605 listeners.emplace_back(context);
606 listeners.back().endpoint = tcp::endpoint(tcp::v6(), port);
607 }
608
609 auto endpoints = config.equal_range("endpoint");
610 for (auto i = endpoints.first; i != endpoints.second; ++i) {
611 auto endpoint = parse_endpoint(i->second, 80, ec);
612 if (ec) {
613 lderr(ctx()) << "failed to parse endpoint=" << i->second << dendl;
614 return -ec.value();
615 }
616 listeners.emplace_back(context);
617 listeners.back().endpoint = endpoint;
618 }
619 // parse tcp nodelay
620 auto nodelay = config.find("tcp_nodelay");
621 if (nodelay != config.end()) {
622 for (auto& l : listeners) {
623 l.use_nodelay = (nodelay->second == "1");
624 }
625 }
626
627
628 bool socket_bound = false;
629 // start listeners
630 for (auto& l : listeners) {
631 l.acceptor.open(l.endpoint.protocol(), ec);
632 if (ec) {
633 if (ec == boost::asio::error::address_family_not_supported) {
634 ldout(ctx(), 0) << "WARNING: cannot open socket for endpoint=" << l.endpoint
635 << ", " << ec.message() << dendl;
636 continue;
637 }
638
639 lderr(ctx()) << "failed to open socket: " << ec.message() << dendl;
640 return -ec.value();
641 }
642
643 if (l.endpoint.protocol() == tcp::v6()) {
644 l.acceptor.set_option(boost::asio::ip::v6_only(true), ec);
645 if (ec) {
646 lderr(ctx()) << "failed to set v6_only socket option: "
647 << ec.message() << dendl;
648 return -ec.value();
649 }
650 }
651
652 l.acceptor.set_option(tcp::acceptor::reuse_address(true));
653 l.acceptor.bind(l.endpoint, ec);
654 if (ec) {
655 lderr(ctx()) << "failed to bind address " << l.endpoint
656 << ": " << ec.message() << dendl;
657 return -ec.value();
658 }
659
660 auto it = config.find("max_connection_backlog");
661 auto max_connection_backlog = boost::asio::socket_base::max_listen_connections;
662 if (it != config.end()) {
663 string err;
664 max_connection_backlog = strict_strtol(it->second.c_str(), 10, &err);
665 if (!err.empty()) {
666 ldout(ctx(), 0) << "WARNING: invalid value for max_connection_backlog=" << it->second << dendl;
667 max_connection_backlog = boost::asio::socket_base::max_listen_connections;
668 }
669 }
670 l.acceptor.listen(max_connection_backlog);
671 l.acceptor.async_accept(l.socket,
672 [this, &l] (boost::system::error_code ec) {
673 accept(l, ec);
674 });
675
676 ldout(ctx(), 4) << "frontend listening on " << l.endpoint << dendl;
677 socket_bound = true;
678 }
679 if (!socket_bound) {
680 lderr(ctx()) << "Unable to listen at any endpoints" << dendl;
681 return -EINVAL;
682 }
683
684 return drop_privileges(ctx());
685}

709string ExpandMetaVar::process_str(const string& in)
710{
711 if (meta_map.empty()) {
712 return in;
713 }
714
715 auto pos = in.find('$');
716 if (pos == std::string::npos) {
717 return in;
718 }
719
720 string out;
721 decltype(pos) last_pos = 0;
722
723 while (pos != std::string::npos) {
724 if (pos > last_pos) {
725 out += in.substr(last_pos, pos - last_pos);
726 }
727
728 string var;
729 const char *valid_chars = "abcdefghijklmnopqrstuvwxyz_";
730
731 size_t endpos = 0;
732 if (in[pos+1] == '{') {
733 // ...${foo_bar}...
734 endpos = in.find_first_not_of(valid_chars, pos + 2);
735 if (endpos != std::string::npos &&
736 in[endpos] == '}') {
737 var = in.substr(pos + 2, endpos - pos - 2);
738 endpos++;
739 }
740 } else {
741 // ...$foo...
742 endpos = in.find_first_not_of(valid_chars, pos + 1);
743 if (endpos != std::string::npos)
744 var = in.substr(pos + 1, endpos - pos - 1);
745 else
746 var = in.substr(pos + 1);
747 }
748 string var_source = in.substr(pos, endpos - pos);
749 last_pos = endpos;
750
751 auto iter = meta_map.find(var);
752 if (iter != meta_map.end()) {
753 out += iter->second;
754 } else {
755 out += var_source;
756 }
757 pos = in.find('$', last_pos);
758 }
759 if (last_pos != std::string::npos) {
760 out += in.substr(last_pos);
761 }
762
763 return out;
764}

844int AsioFrontend::init_ssl()
845{
846 boost::system::error_code ec;
847 auto& config = conf->get_config_map();
848
849 // ssl configuration
850 std::optional<string> cert = conf->get_val("ssl_certificate");
851 if (cert) {
852 // only initialize the ssl context if it's going to be used
853 ssl_context = boost::in_place(ssl::context::tls);
854 }
855
856 std::optional<string> key = conf->get_val("ssl_private_key");
857 bool have_cert = false;
858
859 if (key && !cert) {
860 lderr(ctx()) << "no ssl_certificate configured for ssl_private_key" << dendl;
861 return -EINVAL;
862 }
863
864 std::optional<string> options = conf->get_val("ssl_options");
865 if (options) {
866 if (!cert) {
867 lderr(ctx()) << "no ssl_certificate configured for ssl_options" << dendl;
868 return -EINVAL;
869 }
870 } else if (cert) {
871 options = "no_sslv2:no_sslv3:no_tlsv1:no_tlsv1_1";
872 }
873
874 if (options) {
875 for (auto &option : ceph::split(*options, ":")) {
876 if (option == "default_workarounds") {
877 ssl_context->set_options(ssl::context::default_workarounds);
878 } else if (option == "no_compression") {
879 ssl_context->set_options(ssl::context::no_compression);
880 } else if (option == "no_sslv2") {
881 ssl_context->set_options(ssl::context::no_sslv2);
882 } else if (option == "no_sslv3") {
883 ssl_context->set_options(ssl::context::no_sslv3);
884 } else if (option == "no_tlsv1") {
885 ssl_context->set_options(ssl::context::no_tlsv1);
886 } else if (option == "no_tlsv1_1") {
887 ssl_context->set_options(ssl::context::no_tlsv1_1);
888 } else if (option == "no_tlsv1_2") {
889 ssl_context->set_options(ssl::context::no_tlsv1_2);
890 } else if (option == "single_dh_use") {
891 ssl_context->set_options(ssl::context::single_dh_use);
892 } else {
893 lderr(ctx()) << "ignoring unknown ssl option '" << option << "'" << dendl;
894 }
895 }
896 }
897
898 std::optional<string> ciphers = conf->get_val("ssl_ciphers");
899 if (ciphers) {
900 if (!cert) {
901 lderr(ctx()) << "no ssl_certificate configured for ssl_ciphers" << dendl;
902 return -EINVAL;
903 }
904
905 int r = SSL_CTX_set_cipher_list(ssl_context->native_handle(),
906 ciphers->c_str());
907 if (r == 0) {
908 lderr(ctx()) << "no cipher could be selected from ssl_ciphers: "
909 << *ciphers << dendl;
910 return -EINVAL;
911 }
912 }
913
914 auto ports = config.equal_range("ssl_port");
915 auto endpoints = config.equal_range("ssl_endpoint");
916
917 /*
918 * don't try to config certificate if frontend isn't configured for ssl
919 */
920 if (ports.first == ports.second &&
921 endpoints.first == endpoints.second) {
922 return 0;
923 }
924
925 bool key_is_cert = false;
926
927 if (cert) {
928 if (!key) {
929 key = cert;
930 key_is_cert = true;
931 }
932
933 ExpandMetaVar emv(env.store->get_zone());
934
935 cert = emv.process_str(*cert);
936 key = emv.process_str(*key);
937
938 int r = ssl_set_private_key(*key, key_is_cert);
939 bool have_private_key = (r >= 0);
940 if (r < 0) {
941 if (!key_is_cert) {
942 r = ssl_set_private_key(*cert, true);
943 have_private_key = (r >= 0);
944 }
945 }
946
947 if (have_private_key) {
948 int r = ssl_set_certificate_chain(*cert);
949 have_cert = (r >= 0);
950 }
951 }
952
953 // parse ssl endpoints
954 for (auto i = ports.first; i != ports.second; ++i) {
955 if (!have_cert) {
956 lderr(ctx()) << "no ssl_certificate configured for ssl_port" << dendl;
957 return -EINVAL;
958 }
959 auto port = parse_port(i->second.c_str(), ec);
960 if (ec) {
961 lderr(ctx()) << "failed to parse ssl_port=" << i->second << dendl;
962 return -ec.value();
963 }
964 listeners.emplace_back(context);
965 listeners.back().endpoint.port(port);
966 listeners.back().use_ssl = true;
967
968 listeners.emplace_back(context);
969 listeners.back().endpoint = tcp::endpoint(tcp::v6(), port);
970 listeners.back().use_ssl = true;
971 }
972
973 for (auto i = endpoints.first; i != endpoints.second; ++i) {
974 if (!have_cert) {
975 lderr(ctx()) << "no ssl_certificate configured for ssl_endpoint" << dendl;
976 return -EINVAL;
977 }
978 auto endpoint = parse_endpoint(i->second, 443, ec);
979 if (ec) {
980 lderr(ctx()) << "failed to parse ssl_endpoint=" << i->second << dendl;
981 return -ec.value();
982 }
983 listeners.emplace_back(context);
984 listeners.back().endpoint = endpoint;
985 listeners.back().use_ssl = true;
986 }
987 return 0;
988}

991void AsioFrontend::accept(Listener& l, boost::system::error_code ec)
992{
993 if (!l.acceptor.is_open()) {
994 return;
995 } else if (ec == boost::asio::error::operation_aborted) {
996 return;
997 } else if (ec) {
998 ldout(ctx(), 1) << "accept failed: " << ec.message() << dendl;
999 return;
1000 }
1001 auto stream = std::move(l.socket);
1002 stream.set_option(tcp::no_delay(l.use_nodelay), ec);
1003 l.acceptor.async_accept(l.socket,
1004 [this, &l] (boost::system::error_code ec) {
1005 accept(l, ec);
1006 });
1007
1008 // spawn a coroutine to handle the connection
1009#ifdef WITH_RADOSGW_BEAST_OPENSSL
1010 if (l.use_ssl) {
1011 spawn::spawn(context,
1012 [this, s=std::move(stream)] (yield_context yield) mutable {
1013 auto conn = boost::intrusive_ptr{new Connection(std::move(s))};
1014 auto c = connections.add(*conn);
1015 // wrap the tcp stream in an ssl stream
1016 boost::asio::ssl::stream<tcp_socket&> stream{conn->socket, *ssl_context};
1017 auto timeout = timeout_timer{context.get_executor(), request_timeout, conn};
1018 // do ssl handshake
1019 boost::system::error_code ec;
1020 timeout.start();
1021 auto bytes = stream.async_handshake(ssl::stream_base::server,
1022 conn->buffer.data(), yield[ec]);
1023 timeout.cancel();
1024 if (ec) {
1025 ldout(ctx(), 1) << "ssl handshake failed: " << ec.message() << dendl;
1026 return;
1027 }
1028 conn->buffer.consume(bytes);
1029 handle_connection(context, env, stream, timeout, header_limit,
1030 conn->buffer, true, pause_mutex, scheduler.get(),
1031 ec, yield);
1032 if (!ec) {
1033 // ssl shutdown (ignoring errors)
1034 stream.async_shutdown(yield[ec]);
1035 }
1036 conn->socket.shutdown(tcp::socket::shutdown_both, ec);
1037 }, make_stack_allocator());
1038 } else {
1039#else
1040 {
1041#endif // WITH_RADOSGW_BEAST_OPENSSL
1042 spawn::spawn(context,
1043 [this, s=std::move(stream)] (yield_context yield) mutable {
1044 auto conn = boost::intrusive_ptr{new Connection(std::move(s))};
1045 auto c = connections.add(*conn);
1046 auto timeout = timeout_timer{context.get_executor(), request_timeout, conn};
1047 boost::system::error_code ec;
1048 handle_connection(context, env, conn->socket, timeout, header_limit,
1049 conn->buffer, false, pause_mutex, scheduler.get(),
1050 ec, yield);
1051 conn->socket.shutdown(tcp_socket::shutdown_both, ec);
1052 }, make_stack_allocator());
1053 }
1054}

1056int AsioFrontend::run()
1057{
1058 auto cct = ctx();
1059 const int thread_count = cct->_conf->rgw_thread_pool_size;
1060 threads.reserve(thread_count);
1061
1062 ldout(cct, 4) << "frontend spawning " << thread_count << " threads" << dendl;
1063
1064 // the worker threads call io_context::run(), which will return when there's
1065 // no work left. hold a work guard to keep these threads going until join()
1066 work.emplace(boost::asio::make_work_guard(context));
1067
1068 for (int i = 0; i < thread_count; i++) {
1069 threads.emplace_back([=]() noexcept {
1070 // request warnings on synchronous librados calls in this thread
1071 is_asio_thread = true;
1072 // Have uncaught exceptions kill the process and give a
1073 // stacktrace, not be swallowed.
1074 context.run();
1075 });
1076 }
1077 return 0;
1078}
