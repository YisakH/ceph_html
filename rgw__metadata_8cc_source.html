175int RGWMetadataLog::list_entries(const DoutPrefixProvider *dpp, void *handle,
176 int max_entries,
177 list<cls_log_entry>& entries,
178 string *last_marker,
179 bool *truncated) {
180 LogListCtx *ctx = static_cast<LogListCtx *>(handle);
181
182 if (!max_entries) {
183 *truncated = false;
184 return 0;
185 }
186
187 std::string next_marker;
188 int ret = svc.cls->timelog.list(dpp, ctx->cur_oid, ctx->from_time, ctx->end_time,
189 max_entries, entries, ctx->marker,
190 &next_marker, truncated, null_yield);
191 if ((ret < 0) && (ret != -ENOENT))
192 return ret;
193
194 ctx->marker = std::move(next_marker);
195 if (last_marker) {
196 *last_marker = ctx->marker;
197 }
198
199 if (ret == -ENOENT)
200 *truncated = false;
201
202 return 0;
203}

429int RGWMetadataHandlerPut_SObj::put_pre(const DoutPrefixProvider *dpp)
430{
431 int ret = get(&old_obj, dpp);
432 if (ret < 0 && ret != -ENOENT) {
433 return ret;
434 }
435 exists = (ret != -ENOENT);
436
437 oo.reset(old_obj);
438
439 auto old_ver = (!old_obj ? obj_version() : old_obj->get_version());
440 auto old_mtime = (!old_obj ? ceph::real_time() : old_obj->get_mtime());
441
442 // are we actually going to perform this put, or is it too old?
443 if (!handler->check_versions(exists, old_ver, old_mtime,
444 objv_tracker.write_version, obj->get_mtime(),
445 apply_type)) {
446 return STATUS_NO_APPLY;
447 }
448
449 objv_tracker.read_version = old_ver; /* maintain the obj version we just read */
450
451 return 0;
452}

650int RGWMetadataManager::get(string& metadata_key, Formatter *f, optional_yield y, const DoutPrefixProvider *dpp)
651{
652 RGWMetadataHandler *handler;
653 string entry;
654 int ret = find_handler(metadata_key, &handler, entry);
655 if (ret < 0) {
656 return ret;
657 }
658
659 RGWMetadataObject *obj;
660
661 ret = handler->get(entry, &obj, y, dpp);
662 if (ret < 0) {
663 return ret;
664 }
665
666 f->open_object_section("metadata_info");
667 encode_json("key", metadata_key, f);
668 encode_json("ver", obj->get_version(), f);
669 real_time mtime = obj->get_mtime();
670 if (!real_clock::is_zero(mtime)) {
671 utime_t ut(mtime);
672 encode_json("mtime", ut, f);
673 }
674 encode_json("data", *obj, f);
675 f->close_section();
676
677 delete obj;
678
679 return 0;

682int RGWMetadataManager::put(string& metadata_key, bufferlist& bl,
683 optional_yield y,
684 const DoutPrefixProvider *dpp,
685 RGWMDLogSyncType sync_type,
686 bool from_remote_zone,
687 obj_version *existing_version)
688{
689 RGWMetadataHandler *handler;
690 string entry;
691
692 int ret = find_handler(metadata_key, &handler, entry);
693 if (ret < 0) {
694 return ret;
695 }
696
697 JSONParser parser;
698 if (!parser.parse(bl.c_str(), bl.length())) {
699 return -EINVAL;
700 }
701
702 RGWObjVersionTracker objv_tracker;
703
704 obj_version *objv = &objv_tracker.write_version;
705
706 utime_t mtime;
707
708 try {
709 JSONDecoder::decode_json("key", metadata_key, &parser);
710 JSONDecoder::decode_json("ver", *objv, &parser);
711 JSONDecoder::decode_json("mtime", mtime, &parser);
712 } catch (JSONDecoder::err& e) {
713 return -EINVAL;
714 }
715
716 JSONObj *jo = parser.find_obj("data");
717 if (!jo) {
718 return -EINVAL;
719 }
720
721 RGWMetadataObject *obj = handler->get_meta_obj(jo, *objv, mtime.to_real_time());
722 if (!obj) {
723 return -EINVAL;
724 }
725
726 ret = handler->put(entry, obj, objv_tracker, y, dpp, sync_type, from_remote_zone);
727 if (existing_version) {
728 *existing_version = objv_tracker.read_version;
