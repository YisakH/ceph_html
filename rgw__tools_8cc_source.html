40int rgw_init_ioctx(const DoutPrefixProvider *dpp,
41 librados::Rados *rados, const rgw_pool& pool,
42 librados::IoCtx& ioctx, bool create,
43 bool mostly_omap)
44{
45 int r = rados->ioctx_create(pool.name.c_str(), ioctx);
46 if (r == -ENOENT && create) {
47 r = rados->pool_create(pool.name.c_str());
48 if (r == -ERANGE) {
49 ldpp_dout(dpp, 0)
50 << __func__
51 << " ERROR: librados::Rados::pool_create returned " << cpp_strerror(-r)
52 << " (this can be due to a pool or placement group misconfiguration, e.g."
53 << " pg_num < pgp_num or mon_max_pg_per_osd exceeded)"
54 << dendl;
55 }
56 if (r < 0 && r != -EEXIST) {
57 return r;
58 }
59
60 r = rados->ioctx_create(pool.name.c_str(), ioctx);
61 if (r < 0) {
62 return r;
63 }
64
65 r = ioctx.application_enable(pg_pool_t::APPLICATION_NAME_RGW, false);
66 if (r < 0 && r != -EOPNOTSUPP) {
67 return r;
68 }
69
70 if (mostly_omap) {
71 // set pg_autoscale_bias
72 bufferlist inbl;
73 float bias = g_conf().get_val<double>("rgw_rados_pool_autoscale_bias");
74 int r = rados->mon_command(
75 "{\"prefix\": \"osd pool set\", \"pool\": \"" +
76 pool.name + "\", \"var\": \"pg_autoscale_bias\", \"val\": \"" +
77 stringify(bias) + "\"}",
78 inbl, NULL, NULL);
79 if (r < 0) {
80 ldpp_dout(dpp, 10) << __func__ << " warning: failed to set pg_autoscale_bias on "
81 << pool.name << dendl;
82 }
83 // set recovery_priority
84 int p = g_conf().get_val<uint64_t>("rgw_rados_pool_recovery_priority");
85 r = rados->mon_command(
86 "{\"prefix\": \"osd pool set\", \"pool\": \"" +
87 pool.name + "\", \"var\": \"recovery_priority\": \"" +
88 stringify(p) + "\"}",
89 inbl, NULL, NULL);
90 if (r < 0) {
91 ldpp_dout(dpp, 10) << __func__ << " warning: failed to set recovery_priority on "
92 << pool.name << dendl;
93 }
94 }
95 } else if (r < 0) {
96 return r;
97 }
98 if (!pool.ns.empty()) {
99 ioctx.set_namespace(pool.ns);
100 }
101 return 0;
102}

131int rgw_parse_list_of_flags(struct rgw_name_to_flag *mapping,
132 const string& str, uint32_t *perm)
133{
134 list<string> strs;
135 get_str_list(str, strs);
136 list<string>::iterator iter;
137 uint32_t v = 0;
138 for (iter = strs.begin(); iter != strs.end(); ++iter) {
139 string& s = *iter;
140 for (int i = 0; mapping[i].type_name; i++) {
141 if (s.compare(mapping[i].type_name) == 0)
142 v |= mapping[i].flag;
143 }
144 }
145
146 *perm = v;
147 return 0;
148}

187int rgw_get_system_obj(RGWSysObjectCtx& obj_ctx, const rgw_pool& pool, const string& key, bufferlist& bl,
188 RGWObjVersionTracker *objv_tracker, real_time *pmtime, optional_yield y, const DoutPrefixProvider *dpp, map<string, bufferlist> *pattrs,
189 rgw_cache_entry_info *cache_info,
190 boost::optional<obj_version> refresh_version, bool raw_attrs)
191{
192 bufferlist::iterator iter;
193 int request_len = READ_CHUNK_LEN;
194 rgw_raw_obj obj(pool, key);
195
196 obj_version original_readv;
197 if (objv_tracker && !objv_tracker->read_version.empty()) {
198 original_readv = objv_tracker->read_version;
199 }
200
201 do {
202 auto sysobj = obj_ctx.get_obj(obj);
203 auto rop = sysobj.rop();
204
205 int ret = rop.set_attrs(pattrs)
206 .set_last_mod(pmtime)
207 .set_objv_tracker(objv_tracker)
208 .set_raw_attrs(raw_attrs)
209 .stat(y, dpp);
210 if (ret < 0)
211 return ret;
212
213 ret = rop.set_cache_info(cache_info)
214 .set_refresh_version(refresh_version)
215 .read(dpp, &bl, y);
216 if (ret == -ECANCELED) {
217 /* raced, restart */
218 if (!original_readv.empty()) {
219 /* we were asked to read a specific obj_version, failed */
220 return ret;
221 }
222 if (objv_tracker) {
223 objv_tracker->read_version.clear();
224 }
225 sysobj.invalidate();
226 continue;
227 }
228 if (ret < 0)
229 return ret;
230
231 if (ret < request_len)
232 break;
233 bl.clear();
234 request_len *= 2;
235 } while (true);
236
237 return 0;
238}

254int rgw_rados_operate(const DoutPrefixProvider *dpp, librados::IoCtx& ioctx, const std::string& oid,
255 librados::ObjectReadOperation *op, bufferlist* pbl,
256 optional_yield y, int flags)
257{
258 // given a yield_context, call async_operate() to yield the coroutine instead
259 // of blocking
260 if (y) {
261 auto& context = y.get_io_context();
262 auto& yield = y.get_yield_context();
263 boost::system::error_code ec;
264 auto bl = librados::async_operate(
265 context, ioctx, oid, op, flags, yield[ec]);
266 if (pbl) {
267 *pbl = std::move(bl);
268 }
269 return -ec.value();
270 }
271 // work on asio threads should be asynchronous, so warn when they block
272 if (is_asio_thread) {
273 ldpp_dout(dpp, 20) << "WARNING: blocking librados call" << dendl;
274 }
275 return ioctx.operate(oid, op, nullptr, flags);
276}

316void parse_mime_map_line(const char *start, const char *end)
317{
318 char line[end - start + 1];
319 strncpy(line, start, end - start);
320 line[end - start] = '\0';
321 char *l = line;
322#define DELIMS " \t\n\r"
323
324 while (isspace(*l))
325 l++;
326
327 char *mime = strsep(&l, DELIMS);
328 if (!mime)
329 return;
330
331 char *ext;
332 do {
333 ext = strsep(&l, DELIMS);
334 if (ext && *ext) {
335 (*ext_mime_map)[ext] = mime;
336 }
337 } while (ext);
338}

341void parse_mime_map(const char *buf)
342{
343 const char *start = buf, *end = buf;
344 while (*end) {
345 while (*end && *end != '\n') {
346 end++;
347 }
348 parse_mime_map_line(start, end);
349 end++;
350 start = end;
351 }
352}

354static int ext_mime_map_init(const DoutPrefixProvider *dpp, CephContext *cct, const char *ext_map)
355{
356 int fd = open(ext_map, O_RDONLY);
357 char *buf = NULL;
358 int ret;
359 if (fd < 0) {
360 ret = -errno;
361 ldpp_dout(dpp, 0) << __func__ << " failed to open file=" << ext_map
362 << " : " << cpp_strerror(-ret) << dendl;
363 return ret;
364 }
365
366 struct stat st;
367 ret = fstat(fd, &st);
368 if (ret < 0) {
369 ret = -errno;
370 ldpp_dout(dpp, 0) << __func__ << " failed to stat file=" << ext_map
371 << " : " << cpp_strerror(-ret) << dendl;
372 goto done;
373 }
374
375 buf = (char *)malloc(st.st_size + 1);
376 if (!buf) {
377 ret = -ENOMEM;
378 ldpp_dout(dpp, 0) << __func__ << " failed to allocate buf" << dendl;
379 goto done;
380 }
381
382 ret = safe_read(fd, buf, st.st_size + 1);
383 if (ret != st.st_size) {
384 // huh? file size has changed?
385 ldpp_dout(dpp, 0) << __func__ << " raced! will retry.." << dendl;
386 free(buf);
387 close(fd);
388 return ext_mime_map_init(dpp, cct, ext_map);
389 }
390 buf[st.st_size] = '\0';
391
392 parse_mime_map(buf);
393 ret = 0;
394done:
395 free(buf);
396 close(fd);
397 return ret;
398}

474int RGWDataAccess::Object::put(bufferlist& data,
475 map<string, bufferlist>& attrs,
476 const DoutPrefixProvider *dpp,
477 optional_yield y)
478{
479 rgw::sal::Store* store = sd->store;
480 CephContext *cct = store->ctx();
481
482 string tag;
483 append_rand_alpha(cct, tag, tag, 32);
484
485 RGWBucketInfo& bucket_info = bucket->bucket_info;
486
487 rgw::BlockingAioThrottle aio(store->ctx()->_conf->rgw_put_obj_min_window_size);
488
489 RGWObjectCtx obj_ctx(store);
490 std::unique_ptr<rgw::sal::Bucket> b;
491 store->get_bucket(NULL, bucket_info, &b);
492 std::unique_ptr<rgw::sal::Object> obj = b->get_object(key);
493
494 auto& owner = bucket->policy.get_owner();
495
496 string req_id = store->zone_unique_id(store->get_new_req_id());
497
498 std::unique_ptr<rgw::sal::Writer> processor;
499 processor = store->get_atomic_writer(dpp, y, std::move(obj),
500 owner.get_id(), obj_ctx,
501 nullptr, olh_epoch, req_id);
502
503 int ret = processor->prepare(y);
504 if (ret < 0)
505 return ret;
506
507 rgw::sal::DataProcessor *filter = processor.get();
508
509 CompressorRef plugin;
510 boost::optional<RGWPutObj_Compress> compressor;
511
512 const auto& compression_type = store->get_zone()->get_params().get_compression_type(bucket_info.placement_rule);
513 if (compression_type != "none") {
514 plugin = Compressor::create(store->ctx(), compression_type);
515 if (!plugin) {
516 ldpp_dout(dpp, 1) << "Cannot load plugin for compression type "
517 << compression_type << dendl;
518 } else {
519 compressor.emplace(store->ctx(), plugin, filter);
520 filter = &*compressor;
521 }
522 }
523
524 off_t ofs = 0;
525 auto obj_size = data.length();
526
527 RGWMD5Etag etag_calc;
528
529 do {
530 size_t read_len = std::min(data.length(), (unsigned int)cct->_conf->rgw_max_chunk_size);
531
532 bufferlist bl;
533
534 data.splice(0, read_len, &bl);
535 etag_calc.update(bl);
536
537 ret = filter->process(std::move(bl), ofs);
538 if (ret < 0)
539 return ret;
540
541 ofs += read_len;
542 } while (data.length() > 0);
543
544 ret = filter->process({}, ofs);
545 if (ret < 0) {
546 return ret;
547 }
548 bool has_etag_attr = false;
549 auto iter = attrs.find(RGW_ATTR_ETAG);
550 if (iter != attrs.end()) {
551 bufferlist& bl = iter->second;
552 etag = bl.to_str();
553 has_etag_attr = true;
554 }
555
556 if (!aclbl) {
557 RGWAccessControlPolicy_S3 policy(cct);
558
559 policy.create_canned(bucket->policy.get_owner(), bucket->policy.get_owner(), string()); /* default private policy */
560
561 policy.encode(aclbl.emplace());
562 }
563
564 if (etag.empty()) {
565 etag_calc.finish(&etag);
566 }
567
568 if (!has_etag_attr) {
569 bufferlist etagbl;
570 etagbl.append(etag);
571 attrs[RGW_ATTR_ETAG] = etagbl;
572 }
573 attrs[RGW_ATTR_ACL] = *aclbl;
574
575 string *puser_data = nullptr;
576 if (user_data) {
577 puser_data = &(*user_data);
578 }
579
580 return processor->complete(obj_size, etag,
581 &mtime, mtime,
582 attrs, delete_at,
583 nullptr, nullptr,
584 puser_data,
585 nullptr, nullptr, y);
586}
