30int HeadObjectProcessor::process(bufferlist&& data, uint64_t logical_offset)
31{
32 const bool flush = (data.length() == 0);
33
34 // capture the first chunk for special handling
35 if (data_offset < head_chunk_size || data_offset == 0) {
36 if (flush) {
37 // flush partial chunk
38 return process_first_chunk(std::move(head_data), &processor);
39 }
40
41 auto remaining = head_chunk_size - data_offset;
42 auto count = std::min<uint64_t>(data.length(), remaining);
43 data.splice(0, count, &head_data);
44 data_offset += count;
45
46 if (data_offset == head_chunk_size) {
47 // process the first complete chunk
48 ceph_assert(head_data.length() == head_chunk_size);
49 int r = process_first_chunk(std::move(head_data), &processor);
50 if (r < 0) {
51 return r;
52 }
53 }
54 if (data.length() == 0) { // avoid flushing stripe processor
55 return 0;
56 }
57 }
58 ceph_assert(processor); // process_first_chunk() must initialize
59
60 // send everything else through the processor
61 auto write_offset = data_offset;
62 data_offset += data.length();
63 return processor->process(std::move(data), write_offset);
64}

67static int process_completed(const AioResultList& completed, RawObjSet *written)
68{
69 std::optional<int> error;
70 for (auto& r : completed) {
71 if (r.result >= 0) {
72 written->insert(r.obj.get_ref().obj);
73 } else if (!error) { // record first error code
74 error = r.result;
75 }
76 }
77 return error.value_or(0);
78}

98int RadosWriter::process(bufferlist&& bl, uint64_t offset)
99{
100 bufferlist data = std::move(bl);
101 const uint64_t cost = data.length();
102 if (cost == 0) { // no empty writes, use aio directly for creates
103 return 0;
104 }
105 librados::ObjectWriteOperation op;
106 add_write_hint(op);
107 if (offset == 0) {
108 op.write_full(data);
109 } else {
110 op.write(offset, data);
111 }
112 constexpr uint64_t id = 0; // unused
113 auto c = aio->get(stripe_obj, Aio::librados_op(std::move(op), y), cost, id);
114 return process_completed(c, &written);
115}

117int RadosWriter::write_exclusive(const bufferlist& data)
118{
119 const uint64_t cost = data.length();
120
121 librados::ObjectWriteOperation op;
122 op.create(true); // exclusive create
123 add_write_hint(op);
124 op.write_full(data);
125
126 constexpr uint64_t id = 0; // unused
127 auto c = aio->get(stripe_obj, Aio::librados_op(std::move(op), y), cost, id);
128 auto d = aio->drain();
129 c.splice(c.end(), d);
130 return process_completed(c, &written);
131}

138RadosWriter::~RadosWriter()
139{
140 // wait on any outstanding aio completions
141 process_completed(aio->drain(), &written);
142
143 bool need_to_remove_head = false;
144 std::optional<rgw_raw_obj> raw_head;
145 if (!rgw::sal::Object::empty(head_obj.get())) {
146 raw_head.emplace();
147 rgw::sal::RadosObject* obj = dynamic_cast<rgw::sal::RadosObject*>(head_obj.get());
148 obj->get_raw_obj(&*raw_head);
149 }
150
163 for (const auto& obj : written) {
164 if (raw_head && obj == *raw_head) {
165 ldpp_dout(dpp, 5) << "NOTE: we should not process the head object (" << obj << ") here" << dendl;
166 need_to_remove_head = true;
167 continue;
168 }
169
170 int r = store->delete_raw_obj(dpp, obj);
171 if (r < 0 && r != -ENOENT) {
172 ldpp_dout(dpp, 0) << "WARNING: failed to remove obj (" << obj << "), leaked" << dendl;
173 }
174 }
175
176 if (need_to_remove_head) {
177 std::string version_id;
178 ldpp_dout(dpp, 5) << "NOTE: we are going to process the head obj (" << *raw_head << ")" << dendl;
179 int r = head_obj->delete_object(dpp, &obj_ctx, null_yield);
180 if (r < 0 && r != -ENOENT) {
181 ldpp_dout(dpp, 0) << "WARNING: failed to remove obj (" << *raw_head << "), leaked" << dendl;
182 }
183 }

188int ManifestObjectProcessor::next(uint64_t offset, uint64_t *pstripe_size)
189{
190 // advance the manifest

194 }
195

197

209 *pstripe_size = manifest_gen.cur_stripe_max_size();

223int AtomicObjectProcessor::prepare(optional_yield y)
224{
225 uint64_t max_head_chunk_size;
226 uint64_t head_max_size;
227 uint64_t chunk_size = 0;
228 uint64_t alignment;
229
230 int r = dynamic_cast<rgw::sal::RadosObject*>(head_obj.get())->get_max_chunk_size(
231 dpp, head_obj->get_bucket()->get_placement_rule(),
232 &max_head_chunk_size, &alignment);
233 if (r < 0) {
234 return r;
235 }
236
237 bool same_pool = true;
238 if (head_obj->get_bucket()->get_placement_rule() != tail_placement_rule) {
239 if (!head_obj->placement_rules_match(head_obj->get_bucket()->get_placement_rule(), tail_placement_rule)) {
240 same_pool = false;
241 r = dynamic_cast<rgw::sal::RadosObject*>(head_obj.get())->get_max_chunk_size(dpp, tail_placement_rule, &chunk_size);
242 if (r < 0) {
243 return r;
244 }
245 head_max_size = 0;
246 }
247 }
248
249 if (same_pool) {
250 RGWZonePlacementInfo placement_info;
251 if (!store->svc()->zone->get_zone_params().get_placement(head_obj->get_bucket()->get_placement_rule().name, &placement_info) || placement_info.inline_data) {
252 head_max_size = max_head_chunk_size;
253 } else {
254 head_max_size = 0;
255 }
256 chunk_size = max_head_chunk_size;
257 }
258
259 uint64_t stripe_size;
260 const uint64_t default_stripe_size = store->ctx()->_conf->rgw_obj_stripe_size;
261
262 dynamic_cast<rgw::sal::RadosObject*>(head_obj.get())->get_max_aligned_size(
263 default_stripe_size, alignment, &stripe_size);
264
265 manifest.set_trivial_rule(head_max_size, stripe_size);
266
267 rgw_obj obj = head_obj->get_obj();
268
269 r = manifest_gen.create_begin(store->ctx(), &manifest,
270 head_obj->get_bucket()->get_placement_rule(),
271 &tail_placement_rule,
272 obj.bucket, obj);
273 if (r < 0) {
274 return r;
275 }
276
277 rgw_raw_obj stripe_obj = manifest_gen.get_cur_obj(store);
278
279 r = writer.set_stripe_obj(stripe_obj);
280 if (r < 0) {
281 return r;
282 }
283
284 set_head_chunk_size(head_max_size);
285 // initialize the processors
286 chunk = ChunkProcessor(&writer, chunk_size);
287 stripe = StripeProcessor(&chunk, this, head_max_size);
288 return 0;
289}

291int AtomicObjectProcessor::complete(size_t accounted_size,
292 const std::string& etag,
293 ceph::real_time *mtime,
294 ceph::real_time set_mtime,
295 rgw::sal::Attrs& attrs,
296 ceph::real_time delete_at,
297 const char *if_match,
298 const char *if_nomatch,
299 const std::string *user_data,
300 rgw_zone_set *zones_trace,
301 bool *pcanceled, optional_yield y)
302{
303 int r = writer.drain();
304 if (r < 0) {
305 return r;
306 }
307 const uint64_t actual_size = get_actual_size();
308 r = manifest_gen.create_next(actual_size);
309 if (r < 0) {
310 return r;
311 }
312
313 head_obj->set_atomic(&obj_ctx);
314
315 RGWRados::Object op_target(store->getRados(),
316 head_obj->get_bucket()->get_info(),
317 obj_ctx, head_obj->get_obj());
318 RGWRados::Object::Write obj_op(&op_target);
319
320 /* some object types shouldn't be versioned, e.g., multipart parts */
321 op_target.set_versioning_disabled(!head_obj->get_bucket()->versioning_enabled());
322 obj_op.meta.data = &first_chunk;
323 obj_op.meta.manifest = &manifest;
324 obj_op.meta.ptag = &unique_tag; /* use req_id as operation tag */
325 obj_op.meta.if_match = if_match;
326 obj_op.meta.if_nomatch = if_nomatch;
327 obj_op.meta.mtime = mtime;
328 obj_op.meta.set_mtime = set_mtime;
329 obj_op.meta.owner = owner;
330 obj_op.meta.flags = PUT_OBJ_CREATE;
331 obj_op.meta.olh_epoch = olh_epoch;
332 obj_op.meta.delete_at = delete_at;
333 obj_op.meta.user_data = user_data;
334 obj_op.meta.zones_trace = zones_trace;
335 obj_op.meta.modify_tail = true;
336
337 r = obj_op.write_meta(dpp, actual_size, accounted_size, attrs, y);
338 if (r < 0) {
339 if (r == -ETIMEDOUT) {
340 // The head object write may eventually succeed, clear the set of objects for deletion. if it
341 // doesn't ever succeed, we'll orphan any tail objects as if we'd crashed before that write
342 writer.clear_written();
343 }
344 return r;
345 }
346 if (!obj_op.meta.canceled) {
347 // on success, clear the set of objects for deletion
348 writer.clear_written();
349 }
350 if (pcanceled) {
351 *pcanceled = obj_op.meta.canceled;
352 }
353 return 0;
354}

357int MultipartObjectProcessor::process_first_chunk(bufferlist&& data,
358 DataProcessor **processor)
359{
360 // write the first chunk of the head object as part of an exclusive create,
361 // then drain to wait for the result in case of EEXIST
362 int r = writer.write_exclusive(data);
363 if (r == -EEXIST) {
364 // randomize the oid prefix and reprepare the head/manifest
365 std::string oid_rand = gen_rand_alphanumeric(store->ctx(), 32);
366
367 mp.init(target_obj->get_name(), upload_id, oid_rand);
368 manifest.set_prefix(target_obj->get_name() + "." + oid_rand);
369
370 r = prepare_head();
371 if (r < 0) {
372 return r;
373 }
374 // resubmit the write op on the new head object
375 r = writer.write_exclusive(data);
376 }
377 if (r < 0) {
378 return r;
379 }
380 *processor = &stripe;
381 return 0;
382}

384int MultipartObjectProcessor::prepare_head()
385{
386 const uint64_t default_stripe_size = store->ctx()->_conf->rgw_obj_stripe_size;
387 uint64_t chunk_size;
388 uint64_t stripe_size;
389 uint64_t alignment;
390
391 int r = dynamic_cast<rgw::sal::RadosObject*>(target_obj.get())->get_max_chunk_size(dpp,
392 tail_placement_rule, &chunk_size, &alignment);
393 if (r < 0) {
394 ldpp_dout(dpp, 0) << "ERROR: unexpected: get_max_chunk_size(): placement_rule=" << tail_placement_rule.to_str() << " obj=" << target_obj << " returned r=" << r << dendl;
395 return r;
396 }
397 dynamic_cast<rgw::sal::RadosObject*>(target_obj.get())->get_max_aligned_size(
398 default_stripe_size, alignment, &stripe_size);
399
400 manifest.set_multipart_part_rule(stripe_size, part_num);
401
402 r = manifest_gen.create_begin(store->ctx(), &manifest,
403 head_obj->get_bucket()->get_placement_rule(),
404 &tail_placement_rule,
405 target_obj->get_bucket()->get_key(),
406 target_obj->get_obj());
407 if (r < 0) {
408 return r;
409 }
410
411 rgw_raw_obj stripe_obj = manifest_gen.get_cur_obj(store);
412 dynamic_cast<rgw::sal::RadosObject*>(head_obj.get())->raw_obj_to_obj(stripe_obj);
413 head_obj->set_hash_source(target_obj->get_name());
414
415 r = writer.set_stripe_obj(stripe_obj);
416 if (r < 0) {
417 return r;
418 }
419 stripe_size = manifest_gen.cur_stripe_max_size();
420 set_head_chunk_size(stripe_size);
421
422 chunk = ChunkProcessor(&writer, chunk_size);
423 stripe = StripeProcessor(&chunk, this, stripe_size);
424 return 0;
425}

514 }

516 *pcanceled = obj_op.meta.canceled;
517 }
518 return 0;

531int AppendObjectProcessor::prepare(optional_yield y)
532{
533 RGWObjState *astate;
534 int r = head_obj->get_obj_state(dpp, &obj_ctx, &astate, y);
535 if (r < 0) {
536 return r;
537 }
538 cur_size = astate->size;
539 *cur_accounted_size = astate->accounted_size;
540 if (!astate->exists) {
541 if (position != 0) {
542 ldpp_dout(dpp, 5) << "ERROR: Append position should be zero" << dendl;
543 return -ERR_POSITION_NOT_EQUAL_TO_LENGTH;
544 } else {
545 cur_part_num = 1;
546 //set the prefix
547 char buf[33];
548 gen_rand_alphanumeric(store->ctx(), buf, sizeof(buf) - 1);
549 string oid_prefix = head_obj->get_name();
550 oid_prefix.append(".");
551 oid_prefix.append(buf);
552 oid_prefix.append("_");
553 manifest.set_prefix(oid_prefix);
554 }
555 } else {
556 // check whether the object appendable
557 map<string, bufferlist>::iterator iter = astate->attrset.find(RGW_ATTR_APPEND_PART_NUM);
558 if (iter == astate->attrset.end()) {
559 ldpp_dout(dpp, 5) << "ERROR: The object is not appendable" << dendl;
560 return -ERR_OBJECT_NOT_APPENDABLE;
561 }
562 if (position != *cur_accounted_size) {
563 ldpp_dout(dpp, 5) << "ERROR: Append position should be equal to the obj size" << dendl;
564 return -ERR_POSITION_NOT_EQUAL_TO_LENGTH;
565 }
566 try {
567 using ceph::decode;
568 decode(cur_part_num, iter->second);
569 } catch (buffer::error& err) {
570 ldpp_dout(dpp, 5) << "ERROR: failed to decode part num" << dendl;
571 return -EIO;
572 }
573 cur_part_num++;
574 //get the current obj etag
575 iter = astate->attrset.find(RGW_ATTR_ETAG);
576 if (iter != astate->attrset.end()) {
577 string s = rgw_string_unquote(iter->second.c_str());
578 size_t pos = s.find("-");
579 cur_etag = s.substr(0, pos);
580 }
581
582 iter = astate->attrset.find(RGW_ATTR_STORAGE_CLASS);
583 if (iter != astate->attrset.end()) {
584 tail_placement_rule.storage_class = iter->second.to_str();
585 }
586 cur_manifest = &(*astate->manifest);
587 manifest.set_prefix(cur_manifest->get_prefix());
588 astate->keep_tail = true;
589 }
590 manifest.set_multipart_part_rule(store->ctx()->_conf->rgw_obj_stripe_size, cur_part_num);
591
592 rgw_obj obj = head_obj->get_obj();
593
594 r = manifest_gen.create_begin(store->ctx(), &manifest, head_obj->get_bucket()->get_placement_rule(), &tail_placement_rule, obj.bucket, obj);
595 if (r < 0) {
596 return r;
597 }
598 rgw_raw_obj stripe_obj = manifest_gen.get_cur_obj(store);
599
600 uint64_t chunk_size = 0;
601 r = store->get_raw_chunk_size(dpp, stripe_obj, &chunk_size);
602 if (r < 0) {
603 return r;
604 }
605 r = writer.set_stripe_obj(std::move(stripe_obj));
606 if (r < 0) {
607 return r;
608 }
609
610 uint64_t stripe_size = manifest_gen.cur_stripe_max_size();
611
612 uint64_t max_head_size = std::min(chunk_size, stripe_size);
613 set_head_chunk_size(max_head_size);
614
615 // initialize the processors
616 chunk = ChunkProcessor(&writer, chunk_size);
617 stripe = StripeProcessor(&chunk, this, stripe_size);
618
619 return 0;
620}

622int AppendObjectProcessor::complete(size_t accounted_size, const string &etag, ceph::real_time *mtime,
623 ceph::real_time set_mtime, rgw::sal::Attrs& attrs,
624 ceph::real_time delete_at, const char *if_match, const char *if_nomatch,
625 const string *user_data, rgw_zone_set *zones_trace, bool *pcanceled,
626 optional_yield y)
627{
628 int r = writer.drain();
629 if (r < 0)
630 return r;
631 const uint64_t actual_size = get_actual_size();
632 r = manifest_gen.create_next(actual_size);
633 if (r < 0) {
634 return r;
635 }
636 head_obj->set_atomic(&obj_ctx);
637 RGWRados::Object op_target(store->getRados(),
638 head_obj->get_bucket()->get_info(),
639 obj_ctx, head_obj->get_obj());
640 RGWRados::Object::Write obj_op(&op_target);
641 //For Append obj, disable versioning
642 op_target.set_versioning_disabled(true);
643 if (cur_manifest) {
644 cur_manifest->append(dpp, manifest, store->get_zone());
645 obj_op.meta.manifest = cur_manifest;
646 } else {
647 obj_op.meta.manifest = &manifest;
648 }
649 obj_op.meta.ptag = &unique_tag; /* use req_id as operation tag */
650 obj_op.meta.mtime = mtime;
651 obj_op.meta.set_mtime = set_mtime;
652 obj_op.meta.owner = owner;
653 obj_op.meta.flags = PUT_OBJ_CREATE;
654 obj_op.meta.delete_at = delete_at;
655 obj_op.meta.user_data = user_data;
656 obj_op.meta.zones_trace = zones_trace;
657 obj_op.meta.modify_tail = true;
658 obj_op.meta.appendable = true;
659 //Add the append part number
660 bufferlist cur_part_num_bl;
661 using ceph::encode;
662 encode(cur_part_num, cur_part_num_bl);
663 attrs[RGW_ATTR_APPEND_PART_NUM] = cur_part_num_bl;
664 //calculate the etag
665 if (!cur_etag.empty()) {
666 MD5 hash;
667 // Allow use of MD5 digest in FIPS mode for non-cryptographic purposes
668 hash.SetFlags(EVP_MD_CTX_FLAG_NON_FIPS_ALLOW);
669 char petag[CEPH_CRYPTO_MD5_DIGESTSIZE];
