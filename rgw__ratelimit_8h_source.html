8class RateLimiterEntry {
9 /*
10 fixed_point_rgw_ratelimit is important to preserve the precision of the token calculation
11 for example: a user have a limit of single op per minute, the user will consume its single token and then will send another request, 1s after it.
12 in that case, without this method, the user will get 0 tokens although it should get 0.016 tokens.
13 using this method it will add 16 tokens to the user, and the user will have 16 tokens, each time rgw will do comparison rgw will divide by fixed_point_rgw_ratelimit, so the user will be blocked anyway until it has enough tokens.
14 */
15 static constexpr int64_t fixed_point_rgw_ratelimit = 1000;
16 // counters are tracked in multiples of fixed_point_rgw_ratelimit
17 struct counters {
18 int64_t ops = 0;
19 int64_t bytes = 0;
20 };
21 counters read;
22 counters write;
23 ceph::timespan ts;
24 bool first_run = true;
25 std::mutex ts_lock;
26 // Those functions are returning the integer value of the tokens
27 int64_t read_ops () const
28 {
29 return read.ops / fixed_point_rgw_ratelimit;
30 }
31 int64_t write_ops() const
32 {
33 return write.ops / fixed_point_rgw_ratelimit;
34 }
35 int64_t read_bytes() const
36 {
37 return read.bytes / fixed_point_rgw_ratelimit;
38 }
39 int64_t write_bytes() const
40 {
41 return write.bytes / fixed_point_rgw_ratelimit;
42 }
43 bool should_rate_limit_read(int64_t ops_limit, int64_t bw_limit) {
44 //check if tenants did not reach their bw or ops limits and that the limits are not 0 (which is unlimited)
45 if(((read_ops() - 1 < 0) && (ops_limit > 0)) ||
46 (read_bytes() < 0 && bw_limit > 0))
47 {
48 return true;
49 }
50 // we don't want to reduce ops' tokens if we've rejected it.
51 read.ops -= fixed_point_rgw_ratelimit;
52 return false;
53 }
54 bool should_rate_limit_write(int64_t ops_limit, int64_t bw_limit)
55 {
56 //check if tenants did not reach their bw or ops limits and that the limits are not 0 (which is unlimited)
57 if(((write_ops() - 1 < 0) && (ops_limit > 0)) ||
58 (write_bytes() < 0 && bw_limit > 0))
59 {
60 return true;
61 }
62
63 // we don't want to reduce ops' tokens if we've rejected it.
64 write.ops -= fixed_point_rgw_ratelimit;
65 return false;
66 }
67 /* The purpose of this function is to minimum time before overriding the stored timestamp
68 This function is necessary to force the increase tokens add at least 1 token when it updates the last stored timestamp.
69 That way the user/bucket will not lose tokens because of rounding
70 */
71 bool minimum_time_reached(ceph::timespan curr_timestamp) const
72 {
73 using namespace std::chrono;
74 constexpr auto min_duration = duration_cast<ceph::timespan>(seconds(60)) / fixed_point_rgw_ratelimit;
75 const auto delta = curr_timestamp - ts;
76 if (delta < min_duration)
77 {
78 return false;
79 }
80 return true;
81 }

83 void increase_tokens(ceph::timespan curr_timestamp,
84 const RGWRateLimitInfo* info)
85 {
86 constexpr int fixed_point = fixed_point_rgw_ratelimit;
87 if (first_run)
88 {
89 write.ops = info->max_write_ops * fixed_point;
90 write.bytes = info->max_write_bytes * fixed_point;
91 read.ops = info->max_read_ops * fixed_point;
92 read.bytes = info->max_read_bytes * fixed_point;
93 ts = curr_timestamp;
94 first_run = false;
95 return;
96 }
97 else if(curr_timestamp > ts && minimum_time_reached(curr_timestamp))
98 {
99 const int64_t time_in_ms = std::chrono::duration_cast<std::chrono::milliseconds>(curr_timestamp - ts).count() / 60.0 / std::milli::den * fixed_point; // / 60 to make it work with 1 min token bucket
100 ts = curr_timestamp;
101 const int64_t write_ops = info->max_write_ops * time_in_ms;
102 const int64_t write_bw = info->max_write_bytes * time_in_ms;
103 const int64_t read_ops = info->max_read_ops * time_in_ms;
104 const int64_t read_bw = info->max_read_bytes * time_in_ms;
105 read.ops = std::min(info->max_read_ops * fixed_point, read_ops + read.ops);
106 read.bytes = std::min(info->max_read_bytes * fixed_point, read_bw + read.bytes);
107 write.ops = std::min(info->max_write_ops * fixed_point, write_ops + write.ops);
108 write.bytes = std::min(info->max_write_bytes * fixed_point, write_bw + write.bytes);
109 }
110 }

112 public:
113 bool should_rate_limit(bool is_read, const RGWRateLimitInfo* ratelimit_info, ceph::timespan curr_timestamp)
114 {
115 std::unique_lock lock(ts_lock);
116 increase_tokens(curr_timestamp, ratelimit_info);
117 if (is_read)
118 {
119 return should_rate_limit_read(ratelimit_info->max_read_ops, ratelimit_info->max_read_bytes);
120 }
121 return should_rate_limit_write(ratelimit_info->max_write_ops, ratelimit_info->max_write_bytes);
122 }
123 void decrease_bytes(bool is_read, int64_t amount, const RGWRateLimitInfo* info) {
124 std::unique_lock lock(ts_lock);
125 // we don't want the tenant to be with higher debt than 120 seconds(2 min) of its limit
126 if (is_read)
127 {
128 read.bytes = std::max(read.bytes - amount * fixed_point_rgw_ratelimit,info->max_read_bytes * fixed_point_rgw_ratelimit * -2);
129 } else {
130 write.bytes = std::max(write.bytes - amount * fixed_point_rgw_ratelimit,info->max_write_bytes * fixed_point_rgw_ratelimit * -2);
131 }
132 }
133 void giveback_tokens(bool is_read)
134 {
135 std::unique_lock lock(ts_lock);
136 if (is_read)
137 {
138 read.ops += fixed_point_rgw_ratelimit;
139 } else {
140 write.ops += fixed_point_rgw_ratelimit;
141 }
142 }

148 std::shared_mutex insert_lock;
149 std::atomic_bool& replacing;
150 std::condition_variable& cv;
151 typedef std::unordered_map<std::string, RateLimiterEntry> hash_map;
152 hash_map ratelimit_entries{map_size};
153 static bool is_read_op(const std::string_view method) {
154 if (method == "GET" || method == "HEAD")
155 {
156 return true;
157 }
158 return false;
159 }
160
161 // find or create an entry, and return its iterator
162 auto& find_or_create(const std::string& key) {
163 std::shared_lock rlock(insert_lock);
164 if (ratelimit_entries.size() > 0.9 * map_size && replacing == false)
165 {
166 replacing = true;
167 cv.notify_all();
168 }
169 auto ret = ratelimit_entries.find(key);
170 rlock.unlock();
171 if (ret == ratelimit_entries.end())
172 {
173 std::unique_lock wlock(insert_lock);
174 ret = ratelimit_entries.emplace(std::piecewise_construct,
175 std::forward_as_tuple(key),
176 std::forward_as_tuple()).first;
177 }
178 return ret->second;
179 }

198 {
199 return false;

205 }

210 it.giveback_tokens(is_read);
211 }
212 void decrease_bytes(const char *method, const std::string& key, const int64_t amount, const RGWRateLimitInfo* info) {
213 if (key.empty() || key.length() == 1 || !info->enabled)
214 {
215 return;
216 }

218 if ((is_read && !info->max_read_bytes) || (!is_read && !info->max_write_bytes))
219 {
220 return;
221 }
222 auto& it = find_or_create(key);
223 it.decrease_bytes(is_read, amount, info);
224 }

229// This class purpose is to hold 2 RateLimiter instances, one active and one passive.
230// once the active has reached the watermark for clearing it will call the replace_active() thread using cv
231// The replace_active will clear the previous RateLimiter after all requests to it has been done (use_count() > 1)
232// In the meanwhile new requests will come into the newer active
233class ActiveRateLimiter : public DoutPrefix {
234 std::atomic_uint8_t stopped = {false};
235 std::condition_variable cv;
236 std::mutex cv_m;
237 std::thread runner;
238 std::atomic_bool replacing = false;
239 std::atomic_uint8_t current_active = 0;
240 std::shared_ptr<RateLimiter> ratelimit[2];
241 void replace_active() {
242 using namespace std::chrono_literals;
243 std::unique_lock<std::mutex> lk(cv_m);
244 while (!stopped) {
245 cv.wait(lk);
246 current_active = current_active ^ 1;
247 ldpp_dout(this, 20) << "replacing active ratelimit data structure" << dendl;
248 while (!stopped && ratelimit[(current_active ^ 1)].use_count() > 1 ) {
249 if (cv.wait_for(lk, 1min) != std::cv_status::timeout && stopped)
250 {
251 return;
252 }
253 }
254 if (stopped)
255 {
256 return;
257 }
258 ldpp_dout(this, 20) << "clearing passive ratelimit data structure" << dendl;
259 ratelimit[(current_active ^ 1)]->clear();
260 replacing = false;
261 }
262 }
263 public:
264 ActiveRateLimiter(const ActiveRateLimiter&) = delete;
265 ActiveRateLimiter& operator =(const ActiveRateLimiter&) = delete;
266 ActiveRateLimiter(ActiveRateLimiter&&) = delete;
267 ActiveRateLimiter& operator =(ActiveRateLimiter&&) = delete;
268 ActiveRateLimiter() = delete;
269 ActiveRateLimiter(CephContext* cct) :
270 DoutPrefix(cct, ceph_subsys_rgw, "rate limiter: ")
271 {
272 ratelimit[0] = std::make_shared<RateLimiter>(replacing, cv);
273 ratelimit[1] = std::make_shared<RateLimiter>(replacing, cv);
274 }
275 ~ActiveRateLimiter() {
276 ldpp_dout(this, 20) << "stopping ratelimit_gc thread" << dendl;
277 cv_m.lock();
278 stopped = true;
279 cv_m.unlock();
280 cv.notify_all();
281 runner.join();
282 }
283 std::shared_ptr<RateLimiter> get_active() {
284 return ratelimit[current_active];
285 }
286 void start() {
287 ldpp_dout(this, 20) << "starting ratelimit_gc thread" << dendl;
288 runner = std::thread(&ActiveRateLimiter::replace_active, this);
289 const auto rc = ceph_pthread_setname(runner.native_handle(), "ratelimit_gc");
290 ceph_assert(rc==0);
291 }
