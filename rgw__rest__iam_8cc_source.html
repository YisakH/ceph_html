21void RGWHandler_REST_IAM::rgw_iam_parse_input()
22{
23 std::string post_body = bl_post_body.to_str();
24 if (post_body.size() > 0) {
25 ldpp_dout(s, 10) << "Content of POST: " << post_body << dendl;
26
27 if (post_body.find("Action") != string::npos) {
28 boost::char_separator<char> sep("&");
29 boost::tokenizer<boost::char_separator<char>> tokens(post_body, sep);
30 for (const auto& t : tokens) {
31 auto pos = t.find("=");
32 if (pos != string::npos) {
33 s->info.args.append(t.substr(0,pos),
34 url_decode(t.substr(pos+1, t.size() -1)));
35 }
36 }
37 }
38 }
39 auto payload_hash = rgw::auth::s3::calc_v4_payload_hash(post_body);
40 s->info.args.append("PayloadHash", payload_hash);
41}

113int RGWHandler_REST_IAM::init_from_header(struct req_state* s,
114 int default_formatter,
115 bool configurable_format)
116{
117 string req;
118 string first;
119
120 s->prot_flags = RGW_REST_IAM;
121
122 const char *p, *req_name;
123 if (req_name = s->relative_uri.c_str(); *req_name == '?') {
124 p = req_name;
125 } else {
126 p = s->info.request_params.c_str();
127 }
128
129 s->info.args.set(p);
130 s->info.args.parse(s);
131
132 /* must be called after the args parsing */
133 if (int ret = allocate_formatter(s, default_formatter, configurable_format); ret < 0)
134 return ret;
135
136 if (*req_name != '/')
137 return 0;
138
139 req_name++;
140
141 if (!*req_name)
142 return 0;
143
144 req = req_name;
145 int pos = req.find('/');
146 if (pos >= 0) {
147 first = req.substr(0, pos);
148 } else {
149 first = req;
150 }
151
152 return 0;
153}
