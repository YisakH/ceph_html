55 int DBUser::create_bucket(const DoutPrefixProvider *dpp,
56 const rgw_bucket& b,
57 const string& zonegroup_id,
58 rgw_placement_rule& placement_rule,
59 string& swift_ver_location,
60 const RGWQuotaInfo * pquota_info,
61 const RGWAccessControlPolicy& policy,
62 Attrs& attrs,
63 RGWBucketInfo& info,
64 obj_version& ep_objv,
65 bool exclusive,
66 bool obj_lock_enabled,
67 bool *existed,
68 req_info& req_info,
69 std::unique_ptr<Bucket>* bucket_out,
70 optional_yield y)
71 {
72 int ret;
73 bufferlist in_data;
74 RGWBucketInfo master_info;
75 rgw_bucket *pmaster_bucket = nullptr;
76 uint32_t *pmaster_num_shards = nullptr;
77 real_time creation_time;
78 std::unique_ptr<Bucket> bucket;
79 obj_version objv, *pobjv = NULL;
80
81 /* If it exists, look it up; otherwise create it */
82 ret = store->get_bucket(dpp, this, b, &bucket, y);
83 if (ret < 0 && ret != -ENOENT)
84 return ret;
85
86 if (ret != -ENOENT) {
87 RGWAccessControlPolicy old_policy(store->ctx());
88 *existed = true;
89 if (swift_ver_location.empty()) {
90 swift_ver_location = bucket->get_info().swift_ver_location;
91 }
92 placement_rule.inherit_from(bucket->get_info().placement_rule);
93
94 // don't allow changes to the acl policy
95 /* int r = rgw_op_get_bucket_policy_from_attr(dpp, this, this, bucket->get_attrs(),
96 &old_policy, y);
97 if (r >= 0 && old_policy != policy) {
98 bucket_out->swap(bucket);
99 return -EEXIST;
100 }*/
101 } else {
102 bucket = std::make_unique<DBBucket>(store, b, this);
103 *existed = false;
104 bucket->set_attrs(attrs);
105 // XXX: For now single default zone and STANDARD storage class
106 // supported.
107 placement_rule.name = "default";
108 placement_rule.storage_class = "STANDARD";
109 }
110
111 /*
112 * XXX: If not master zone, fwd the request to master zone.
113 * For now DBStore has single zone.
114 */
115 std::string zid = zonegroup_id;
116 /* if (zid.empty()) {
117 zid = svc()->zone->get_zonegroup().get_id();
118 } */
119
120 if (*existed) {
121 rgw_placement_rule selected_placement_rule;
122 /* XXX: Handle this when zone is implemented
123 ret = svc()->zone->select_bucket_placement(this.get_info(),
124 zid, placement_rule,
125 &selected_placement_rule, nullptr, y);
126 if (selected_placement_rule != info.placement_rule) {
127 ret = -EEXIST;
128 bucket_out->swap(bucket);
129 return ret;
130 } */
131 } else {
132
133 /* XXX: We may not need to send all these params. Cleanup the unused ones */
134 ret = store->getDB()->create_bucket(dpp, this->get_info(), bucket->get_key(),
135 zid, placement_rule, swift_ver_location, pquota_info,
136 attrs, info, pobjv, &ep_objv, creation_time,
137 pmaster_bucket, pmaster_num_shards, y, exclusive);
138 if (ret == -EEXIST) {
139 *existed = true;
140 ret = 0;
141 } else if (ret != 0) {
142 return ret;
143 }
144 }
145
146 bucket->set_version(ep_objv);
147 bucket->get_info() = info;
148
149 bucket_out->swap(bucket);
150
151 return ret;
152 }

227 int DBBucket::remove_bucket(const DoutPrefixProvider *dpp, bool delete_children, bool forward_to_master, req_info* req_info, optional_yield y)
228 {
229 int ret;
230
231 ret = load_bucket(dpp, y);
232 if (ret < 0)
233 return ret;
234
235 /* XXX: handle delete_children */
236
237 if (!delete_children) {
238 /* Check if there are any objects */
239 rgw::sal::Bucket::ListParams params;
240 params.list_versions = true;
241 params.allow_unordered = true;
242
243 rgw::sal::Bucket::ListResults results;
244
245 results.objs.clear();
246
247 ret = list(dpp, params, 2, results, null_yield);
248
249 if (ret < 0) {
250 ldpp_dout(dpp, 20) << __func__ << ": Bucket list objects returned " <<
251 ret << dendl;
252 return ret;
253 }
254
255 if (!results.objs.empty()) {
256 ret = -ENOTEMPTY;
257 ldpp_dout(dpp, -1) << __func__ << ": Bucket Not Empty.. returning " <<
258 ret << dendl;
259 return ret;
260 }
261 }
262
263 ret = store->getDB()->remove_bucket(dpp, info);
264
265 return ret;
266 }

446 int DBBucket::list(const DoutPrefixProvider *dpp, ListParams& params, int max, ListResults& results, optional_yield y)
447 {
448 int ret = 0;
449
450 results.objs.clear();
451
452 DB::Bucket target(store->getDB(), get_info());
453 DB::Bucket::List list_op(&target);
454
455 list_op.params.prefix = params.prefix;
456 list_op.params.delim = params.delim;
457 list_op.params.marker = params.marker;
458 list_op.params.ns = params.ns;
459 list_op.params.end_marker = params.end_marker;
460 list_op.params.ns = params.ns;
461 list_op.params.enforce_ns = params.enforce_ns;
462 list_op.params.access_list_filter = params.access_list_filter;
463 list_op.params.force_check_filter = params.force_check_filter;
464 list_op.params.list_versions = params.list_versions;
465 list_op.params.allow_unordered = params.allow_unordered;
466
467 results.objs.clear();
468 ret = list_op.list_objects(dpp, max, &results.objs, &results.common_prefixes, &results.is_truncated);
469 if (ret >= 0) {
470 results.next_marker = list_op.get_next_marker();
471 params.marker = results.next_marker;
472 }
473
474 return ret;
475 }

945 int DBMultipartUpload::list_parts(const DoutPrefixProvider *dpp, CephContext *cct,
946 int num_parts, int marker,
947 int *next_marker, bool *truncated,
948 bool assume_unsorted)
949 {
950 std::list<RGWUploadPartInfo> parts_map;
951
952 std::unique_ptr<rgw::sal::Object> obj = get_meta_obj();
953
954 parts.clear();
955 int ret;
956
957 DB::Object op_target(store->getDB(),
958 obj->get_bucket()->get_info(), obj->get_obj());
959 ret = op_target.get_mp_parts_list(dpp, parts_map);
960 if (ret < 0) {
961 return ret;
962 }
963
964 int last_num = 0;
965
966 while (!parts_map.empty()) {
967 std::unique_ptr<DBMultipartPart> part = std::make_unique<DBMultipartPart>();
968 RGWUploadPartInfo &pinfo = parts_map.front();
969 part->set_info(pinfo);
970 if ((int)pinfo.num > marker) {
971 last_num = pinfo.num;
972 parts[pinfo.num] = std::move(part);
973 }
974 parts_map.pop_front();
975 }
976
977 /* rebuild a map with only num_parts entries */
978 std::map<uint32_t, std::unique_ptr<MultipartPart>> new_parts;
979 std::map<uint32_t, std::unique_ptr<MultipartPart>>::iterator piter;
980 int i;
981 for (i = 0, piter = parts.begin();
982 i < num_parts && piter != parts.end();
983 ++i, ++piter) {
984 last_num = piter->first;
985 new_parts[piter->first] = std::move(piter->second);
986 }
987
988 if (truncated) {
989 *truncated = (piter != parts.end());
990 }
991
992 parts.swap(new_parts);
993
994 if (next_marker) {
995 *next_marker = last_num;
996 }
997
998 return 0;
999 }

1001 int DBMultipartUpload::complete(const DoutPrefixProvider *dpp,
1002 optional_yield y, CephContext* cct,
1003 map<int, string>& part_etags,
1004 list<rgw_obj_index_key>& remove_objs,
1005 uint64_t& accounted_size, bool& compressed,
1006 RGWCompressionInfo& cs_info, off_t& ofs,
1007 std::string& tag, ACLOwner& owner,
1008 uint64_t olh_epoch,
1009 rgw::sal::Object* target_obj,
1010 RGWObjectCtx* obj_ctx)
1011 {
1012 char final_etag[CEPH_CRYPTO_MD5_DIGESTSIZE];
1013 char final_etag_str[CEPH_CRYPTO_MD5_DIGESTSIZE * 2 + 16];
1014 std::string etag;
1015 bufferlist etag_bl;
1016 MD5 hash;
1017 bool truncated;
1018 int ret;
1019
1020 int total_parts = 0;
1021 int handled_parts = 0;
1022 int max_parts = 1000;
1023 int marker = 0;
1024 uint64_t min_part_size = cct->_conf->rgw_multipart_min_part_size;
1025 auto etags_iter = part_etags.begin();
1026 rgw::sal::Attrs attrs = target_obj->get_attrs();
1027
1028 ofs = 0;
1029 accounted_size = 0;
1030 do {
1031 ret = list_parts(dpp, cct, max_parts, marker, &marker, &truncated);
1032 if (ret == -ENOENT) {
1033 ret = -ERR_NO_SUCH_UPLOAD;
1034 }
1035 if (ret < 0)
1036 return ret;
1037
1038 total_parts += parts.size();
1039 if (!truncated && total_parts != (int)part_etags.size()) {
1040 ldpp_dout(dpp, 0) << "NOTICE: total parts mismatch: have: " << total_parts
1041 << " expected: " << part_etags.size() << dendl;
1042 ret = -ERR_INVALID_PART;
1043 return ret;
1044 }
1045
1046 for (auto obj_iter = parts.begin(); etags_iter != part_etags.end() && obj_iter != parts.end(); ++etags_iter, ++obj_iter, ++handled_parts) {
1047 DBMultipartPart* part = dynamic_cast<rgw::sal::DBMultipartPart*>(obj_iter->second.get());
1048 uint64_t part_size = part->get_size();
1049 if (handled_parts < (int)part_etags.size() - 1 &&
1050 part_size < min_part_size) {
1051 ret = -ERR_TOO_SMALL;
1052 return ret;
1053 }
1054
1055 char petag[CEPH_CRYPTO_MD5_DIGESTSIZE];
1056 if (etags_iter->first != (int)obj_iter->first) {
1057 ldpp_dout(dpp, 0) << "NOTICE: parts num mismatch: next requested: "
1058 << etags_iter->first << " next uploaded: "
1059 << obj_iter->first << dendl;
1060 ret = -ERR_INVALID_PART;
1061 return ret;
1062 }
1063 string part_etag = rgw_string_unquote(etags_iter->second);
1064 if (part_etag.compare(part->get_etag()) != 0) {
1065 ldpp_dout(dpp, 0) << "NOTICE: etag mismatch: part: " << etags_iter->first
1066 << " etag: " << etags_iter->second << dendl;
1067 ret = -ERR_INVALID_PART;
1068 return ret;
1069 }
1070
1071 hex_to_buf(part->get_etag().c_str(), petag,
1072 CEPH_CRYPTO_MD5_DIGESTSIZE);
1073 hash.Update((const unsigned char *)petag, sizeof(petag));
1074
1075 RGWUploadPartInfo& obj_part = part->get_info();
1076
1077 ofs += obj_part.size;
1078 accounted_size += obj_part.accounted_size;
1079 }
1080 } while (truncated);
1081 hash.Final((unsigned char *)final_etag);
1082
1083 buf_to_hex((unsigned char *)final_etag, sizeof(final_etag), final_etag_str);
1084 snprintf(&final_etag_str[CEPH_CRYPTO_MD5_DIGESTSIZE * 2],
1085 sizeof(final_etag_str) - CEPH_CRYPTO_MD5_DIGESTSIZE * 2,
1086 "-%lld", (long long)part_etags.size());
1087 etag = final_etag_str;
1088 ldpp_dout(dpp, 10) << "calculated etag: " << etag << dendl;
1089
1090 etag_bl.append(etag);
1091
1092 attrs[RGW_ATTR_ETAG] = etag_bl;
1093
1094 /* XXX: handle compression ? */
1095
1096 /* Rename all the object data entries with original object name (i.e
1097 * from 'head_obj.name + "." + upload_id' to head_obj.name) */
1098
1099 /* Original head object */
1100 DB::Object op_target(store->getDB(),
1101 target_obj->get_bucket()->get_info(),
1102 target_obj->get_obj());
1103 DB::Object::Write obj_op(&op_target);
1104 obj_op.prepare(NULL);
1105
1106 /* Meta object */
1107 std::unique_ptr<rgw::sal::Object> meta_obj = get_meta_obj();
1108 DB::Object meta_op_target(store->getDB(),
1109 meta_obj->get_bucket()->get_info(),
1110 meta_obj->get_obj());
1111 DB::Object::Write mp_op(&meta_op_target);
1112 mp_op.update_mp_parts(dpp, target_obj->get_obj().key);
1113
1114 obj_op.meta.owner = owner.get_id();
1115 obj_op.meta.flags = PUT_OBJ_CREATE;
1116 obj_op.meta.modify_tail = true;
1117 obj_op.meta.completeMultipart = true;
1118
1119 ret = obj_op.write_meta(dpp, ofs, accounted_size, attrs);
1120 if (ret < 0)
1121 return ret;
1122
1123 /* No need to delete Meta obj here. It is deleted from sal */
1124 return ret;
1125 }

1127 int DBMultipartUpload::get_info(const DoutPrefixProvider *dpp, optional_yield y, RGWObjectCtx* obj_ctx, rgw_placement_rule** rule, rgw::sal::Attrs* attrs)
1128 {
1129 if (!rule && !attrs) {
1130 return 0;
1131 }
1132
1133 if (rule) {
1134 if (!placement.empty()) {
1135 *rule = &placement;
1136 if (!attrs) {
1137 /* Don't need attrs, done */
1138 return 0;
1139 }
1140 } else {
1141 *rule = nullptr;
1142 }
1143 }
1144
1145 /* We need either attributes or placement, so we need a read */
1146 std::unique_ptr<rgw::sal::Object> meta_obj;
1147 meta_obj = get_meta_obj();
1148 meta_obj->set_in_extra_data(true);
1149
1150 multipart_upload_info upload_info;
1151 bufferlist headbl;
1152
1153 /* Read the obj head which contains the multipart_upload_info */

1155 int ret = read_op->prepare(y, dpp);
1156 if (ret < 0) {
1157 if (ret == -ENOENT) {
1158 return -ERR_NO_SUCH_UPLOAD;
1159 }
1160 return ret;
1161 }
1162
1163 if (attrs) {
1164 /* Attrs are filled in by prepare */
1165 *attrs = meta_obj->get_attrs();
1166 if (!rule || *rule != nullptr) {
1167 /* placement was cached; don't actually read */
1168 return 0;
1169 }
1170 }
1171
1172 /* Now read the placement from the head */
1173 ret = read_op->read(0, store->getDB()->get_max_head_size(), headbl, y, dpp);
1174 if (ret < 0) {
1175 if (ret == -ENOENT) {
1176 return -ERR_NO_SUCH_UPLOAD;
1177 }
1178 return ret;
1179 }
1180
1181 if (headbl.length() <= 0) {
1182 return -ERR_NO_SUCH_UPLOAD;
1183 }
1184
1185 /* Decode multipart_upload_info */
1186 auto hiter = headbl.cbegin();
1187 try {
1188 decode(upload_info, hiter);
1189 } catch (buffer::error& err) {
1190 ldpp_dout(dpp, 0) << "ERROR: failed to decode multipart upload info" << dendl;
1191 return -EIO;
1192 }
1193 placement = upload_info.dest_placement;
1194 *rule = &placement;
1195
1196 return 0;
1197 }

1241 int DBMultipartWriter::process(bufferlist&& data, uint64_t offset)
1242 {
1243 /* XXX: same as AtomicWriter..consolidate code */
1244 total_data_size += data.length();
1245
1246 /* XXX: Optimize all bufferlist copies in this function */
1247
1248 /* copy head_data into meta. But for multipart we do not
1249 * need to write head_data */
1250 uint64_t max_chunk_size = store->getDB()->get_max_chunk_size();
1251 int excess_size = 0;
1252
1253 /* Accumulate tail_data till max_chunk_size or flush op */
1254 bufferlist tail_data;
1255
1256 if (data.length() != 0) {
1257 parent_op.meta.data = &head_data; /* Null data ?? */
1258
1259 /* handle tail )parts.
1260 * First accumulate and write data into dbstore in its chunk_size
1261 * parts
1262 */
1263 if (!tail_part_size) { /* new tail part */
1264 tail_part_offset = offset;
1265 }
1266 data.begin(0).copy(data.length(), tail_data);
1267 tail_part_size += tail_data.length();
1268 tail_part_data.append(tail_data);
1269
1270 if (tail_part_size < max_chunk_size) {
1271 return 0;
1272 } else {
1273 int write_ofs = 0;
1274 while (tail_part_size >= max_chunk_size) {
1275 excess_size = tail_part_size - max_chunk_size;
1276 bufferlist tmp;
1277 tail_part_data.begin(write_ofs).copy(max_chunk_size, tmp);
1278 /* write tail objects data */
1279 int ret = parent_op.write_data(dpp, tmp, tail_part_offset);
1280
1281 if (ret < 0) {
1282 return ret;
1283 }
1284
1285 tail_part_size -= max_chunk_size;
1286 write_ofs += max_chunk_size;
1287 tail_part_offset += max_chunk_size;
1288 }
1289 /* reset tail parts or update if excess data */
1290 if (excess_size > 0) { /* wrote max_chunk_size data */
1291 tail_part_size = excess_size;
1292 bufferlist tmp;
1293 tail_part_data.begin(write_ofs).copy(excess_size, tmp);
1294 tail_part_data = tmp;
1295 } else {
1296 tail_part_size = 0;
1297 tail_part_data.clear();
1298 tail_part_offset = 0;
1299 }
1300 }
1301 } else {
1302 if (tail_part_size == 0) {
1303 return 0; /* nothing more to write */
1304 }
1305
1306 /* flush watever tail data is present */
1307 int ret = parent_op.write_data(dpp, tail_part_data, tail_part_offset);
1308 if (ret < 0) {
1309 return ret;
1310 }
1311 tail_part_size = 0;
1312 tail_part_data.clear();
1313 tail_part_offset = 0;
1314 }
1315
1316 return 0;
1317 }

1384 int DBAtomicWriter::process(bufferlist&& data, uint64_t offset)
1385 {
1386 total_data_size += data.length();
1387
1388 /* XXX: Optimize all bufferlist copies in this function */
1389
1390 /* copy head_data into meta. */
1391 uint64_t head_size = store->getDB()->get_max_head_size();
1392 unsigned head_len = 0;
1393 uint64_t max_chunk_size = store->getDB()->get_max_chunk_size();
1394 int excess_size = 0;
1395
1396 /* Accumulate tail_data till max_chunk_size or flush op */
1397 bufferlist tail_data;
1398
1399 if (data.length() != 0) {
1400 if (offset < head_size) {
1401 /* XXX: handle case (if exists) where offset > 0 & < head_size */
1402 head_len = std::min((uint64_t)data.length(),
1403 head_size - offset);
1404 bufferlist tmp;
1405 data.begin(0).copy(head_len, tmp);
1406 head_data.append(tmp);
1407
1408 parent_op.meta.data = &head_data;
1409 if (head_len == data.length()) {
1410 return 0;
1411 }
1412
1413 /* Move offset by copy_len */
1414 offset = head_len;
1415 }
1416
1417 /* handle tail parts.
1418 * First accumulate and write data into dbstore in its chunk_size
1419 * parts
1420 */
1421 if (!tail_part_size) { /* new tail part */
1422 tail_part_offset = offset;
1423 }
1424 data.begin(head_len).copy(data.length() - head_len, tail_data);
1425 tail_part_size += tail_data.length();
1426 tail_part_data.append(tail_data);
1427
1428 if (tail_part_size < max_chunk_size) {
1429 return 0;
1430 } else {
1431 int write_ofs = 0;
1432 while (tail_part_size >= max_chunk_size) {
1433 excess_size = tail_part_size - max_chunk_size;
1434 bufferlist tmp;
1435 tail_part_data.begin(write_ofs).copy(max_chunk_size, tmp);
1436 /* write tail objects data */
1437 int ret = parent_op.write_data(dpp, tmp, tail_part_offset);
1438
1439 if (ret < 0) {
1440 return ret;
1441 }
1442
1443 tail_part_size -= max_chunk_size;
1444 write_ofs += max_chunk_size;
1445 tail_part_offset += max_chunk_size;
1446 }
1447 /* reset tail parts or update if excess data */
1448 if (excess_size > 0) { /* wrote max_chunk_size data */
1449 tail_part_size = excess_size;
1450 bufferlist tmp;
1451 tail_part_data.begin(write_ofs).copy(excess_size, tmp);
1452 tail_part_data = tmp;
1453 } else {
1454 tail_part_size = 0;
1455 tail_part_data.clear();
1456 tail_part_offset = 0;
1457 }
1458 }
1459 } else {
1460 if (tail_part_size == 0) {
1461 return 0; /* nothing more to write */
1462 }
1463
1464 /* flush watever tail data is present */
1465 int ret = parent_op.write_data(dpp, tail_part_data, tail_part_offset);
1466 if (ret < 0) {
1467 return ret;
1468 }
1469 tail_part_size = 0;
1470 tail_part_data.clear();
1471 tail_part_offset = 0;
1472 }
1473
1474 return 0;
1475 }
