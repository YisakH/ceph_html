215static void set_date_header(const real_time *t, map<string, string>& headers, bool high_precision_time, const string& header_name)
216{
217 if (!t) {
218 return;
219 }
220 stringstream s;
221 utime_t tm = utime_t(*t);
222 if (high_precision_time) {
223 tm.gmtime_nsec(s);
224 } else {
225 tm.gmtime(s);
226 }
227 headers[header_name] = s.str();
228}

230template <class T>
231static void set_header(T val, map<string, string>& headers, const string& header_name)
232{
233 stringstream s;
234 s << val;
235 headers[header_name] = s.str();
236}

260int RGWRESTConn::get_obj(const DoutPrefixProvider *dpp, const rgw::sal::Object* obj, const get_obj_params& in_params, bool send, RGWRESTStreamRWRequest **req)
261{
262 string url;
263 int ret = get_url(url);
264 if (ret < 0)
265 return ret;
266
267 param_vec_t params;
268 populate_params(params, &in_params.uid, self_zone_group);
269 if (in_params.prepend_metadata) {
270 params.push_back(param_pair_t(RGW_SYS_PARAM_PREFIX "prepend-metadata", "true"));
271 }
272 if (in_params.rgwx_stat) {
273 params.push_back(param_pair_t(RGW_SYS_PARAM_PREFIX "stat", "true"));
274 }
275 if (in_params.sync_manifest) {
276 params.push_back(param_pair_t(RGW_SYS_PARAM_PREFIX "sync-manifest", ""));
277 }
278 if (in_params.skip_decrypt) {
279 params.push_back(param_pair_t(RGW_SYS_PARAM_PREFIX "skip-decrypt", ""));
280 }
281 if (!obj->get_instance().empty()) {
282 const string& instance = obj->get_instance();
283 params.push_back(param_pair_t("versionId", instance));
284 }
285 if (in_params.get_op) {
286 *req = new RGWRESTStreamReadRequest(cct, url, in_params.cb, NULL, &params, api_name, host_style);
287 } else {
288 *req = new RGWRESTStreamHeadRequest(cct, url, in_params.cb, NULL, &params, api_name);
289 }
290 map<string, string> extra_headers;
291 if (in_params.info) {
292 const auto& orig_map = in_params.info->env->get_map();
293
294 /* add original headers that start with HTTP_X_AMZ_ */
295 static constexpr char SEARCH_AMZ_PREFIX[] = "HTTP_X_AMZ_";
296 for (auto iter= orig_map.lower_bound(SEARCH_AMZ_PREFIX); iter != orig_map.end(); ++iter) {
297 const string& name = iter->first;
298 if (name == "HTTP_X_AMZ_DATE") /* don't forward date from original request */
299 continue;
300 if (name.compare(0, strlen(SEARCH_AMZ_PREFIX), SEARCH_AMZ_PREFIX) != 0)
301 break;
302 extra_headers[iter->first] = iter->second;
303 }
304 }
305
306 set_date_header(in_params.mod_ptr, extra_headers, in_params.high_precision_time, "HTTP_IF_MODIFIED_SINCE");
307 set_date_header(in_params.unmod_ptr, extra_headers, in_params.high_precision_time, "HTTP_IF_UNMODIFIED_SINCE");
308 if (!in_params.etag.empty()) {
309 set_header(in_params.etag, extra_headers, "HTTP_IF_MATCH");
310 }
311 if (in_params.mod_zone_id != 0) {
312 set_header(in_params.mod_zone_id, extra_headers, "HTTP_DEST_ZONE_SHORT_ID");
313 }
314 if (in_params.mod_pg_ver != 0) {
315 set_header(in_params.mod_pg_ver, extra_headers, "HTTP_DEST_PG_VER");
316 }
317 if (in_params.range_is_set) {
318 char buf[64];
319 snprintf(buf, sizeof(buf), "bytes=%lld-%lld", (long long)in_params.range_start, (long long)in_params.range_end);
320 set_header(buf, extra_headers, "RANGE");
321 }
322
323 int r = (*req)->send_prepare(dpp, key, extra_headers, obj->get_obj());
324 if (r < 0) {
325 goto done_err;
326 }
327
328 if (!send) {
329 return 0;
330 }
331
332 r = (*req)->send(nullptr);
333 if (r < 0) {
334 goto done_err;
335 }
336 return 0;
337done_err:
338 delete *req;
339 *req = nullptr;
340 return r;
341}

357int RGWRESTConn::get_resource(const DoutPrefixProvider *dpp,
358 const string& resource,
359 param_vec_t *extra_params,
360 map<string, string> *extra_headers,
361 bufferlist& bl,
362 bufferlist *send_data,
363 RGWHTTPManager *mgr,
364 optional_yield y)
365{
366 string url;
367 int ret = get_url(url);
368 if (ret < 0)
369 return ret;
370
371 param_vec_t params;
372
373 if (extra_params) {
374 params.insert(params.end(), extra_params->begin(), extra_params->end());
375 }
376
377 populate_params(params, nullptr, self_zone_group);
378
379 RGWStreamIntoBufferlist cb(bl);
380
381 RGWRESTStreamReadRequest req(cct, url, &cb, NULL, &params, api_name, host_style);
382
383 map<string, string> headers;
384 if (extra_headers) {
385 headers.insert(extra_headers->begin(), extra_headers->end());
386 }
387
388 ret = req.send_request(dpp, &key, headers, resource, mgr, send_data);
389 if (ret < 0) {
390 ldpp_dout(dpp, 5) << __func__ << ": send_request() resource=" << resource << " returned ret=" << ret << dendl;
391 return ret;
392 }
393
394 return req.complete_request(y);
395}
