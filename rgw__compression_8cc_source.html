10int rgw_compression_info_from_attr(const bufferlist& attr,
11 bool& need_decompress,
12 RGWCompressionInfo& cs_info)
13{
14 auto bliter = attr.cbegin();
15 try {
16 decode(cs_info, bliter);
17 } catch (buffer::error& err) {
18 return -EIO;
19 }
20 if (cs_info.blocks.size() == 0) {
21 return -EIO;
22 }
23 if (cs_info.compression_type != "none")
24 need_decompress = true;
25 else
26 need_decompress = false;
27 return 0;
28}

44int RGWPutObj_Compress::process(bufferlist&& in, uint64_t logical_offset)
45{
46 bufferlist out;
47 if (in.length() > 0) {
48 // compression stuff
49 if ((logical_offset > 0 && compressed) || // if previous part was compressed
50 (logical_offset == 0)) { // or it's the first part
51 ldout(cct, 10) << "Compression for rgw is enabled, compress part " << in.length() << dendl;
52 int cr = compressor->compress(in, out, compressor_message);
53 if (cr < 0) {
54 if (logical_offset > 0) {
55 lderr(cct) << "Compression failed with exit code " << cr
56 << " for next part, compression process failed" << dendl;
57 return -EIO;
58 }
59 compressed = false;
60 ldout(cct, 5) << "Compression failed with exit code " << cr
61 << " for first part, storing uncompressed" << dendl;
62 out = std::move(in);
63 } else {
64 compressed = true;
65
66 compression_block newbl;
67 size_t bs = blocks.size();
68 newbl.old_ofs = logical_offset;
69 newbl.new_ofs = bs > 0 ? blocks[bs-1].len + blocks[bs-1].new_ofs : 0;
70 newbl.len = out.length();
71 blocks.push_back(newbl);
72 }
73 } else {
74 compressed = false;
75 out = std::move(in);
76 }
77 // end of compression stuff
78 }
79 return Pipe::process(std::move(out), logical_offset);
80}

99int RGWGetObj_Decompress::handle_data(bufferlist& bl, off_t bl_ofs, off_t bl_len)
100{
101 ldout(cct, 10) << "Compression for rgw is enabled, decompress part "
102 << "bl_ofs="<< bl_ofs << ", bl_len=" << bl_len << dendl;
103
104 if (!compressor.get()) {
105 // if compressor isn't available - error, because cannot return decompressed data?
106 lderr(cct) << "Cannot load compressor of type " << cs_info->compression_type << dendl;
107 return -EIO;
108 }
109 bufferlist out_bl, in_bl, temp_in_bl;
110 bl.begin(bl_ofs).copy(bl_len, temp_in_bl);
111 bl_ofs = 0;
112 int r = 0;
113 if (waiting.length() != 0) {
114 in_bl.append(waiting);
115 in_bl.append(temp_in_bl);
116 waiting.clear();
117 } else {
118 in_bl = std::move(temp_in_bl);
119 }
120 bl_len = in_bl.length();
121
122 auto iter_in_bl = in_bl.cbegin();
123 while (first_block <= last_block) {
124 bufferlist tmp;
125 off_t ofs_in_bl = first_block->new_ofs - cur_ofs;
126 if (ofs_in_bl + (off_t)first_block->len > bl_len) {
127 // not complete block, put it to waiting
128 unsigned tail = bl_len - ofs_in_bl;
129 if (iter_in_bl.get_off() != ofs_in_bl) {
130 iter_in_bl.seek(ofs_in_bl);
131 }
132 iter_in_bl.copy(tail, waiting);
133 cur_ofs -= tail;
134 break;
135 }
136 if (iter_in_bl.get_off() != ofs_in_bl) {
137 iter_in_bl.seek(ofs_in_bl);
138 }
139 iter_in_bl.copy(first_block->len, tmp);
140 int cr = compressor->decompress(tmp, out_bl, cs_info->compressor_message);
141 if (cr < 0) {
142 lderr(cct) << "Decompression failed with exit code " << cr << dendl;
143 return cr;
144 }
145 ++first_block;
146 while (out_bl.length() - q_ofs >=
147 static_cast<off_t>(cct->_conf->rgw_max_chunk_size)) {
148 off_t ch_len = std::min<off_t>(cct->_conf->rgw_max_chunk_size, q_len);
149 q_len -= ch_len;
150 r = next->handle_data(out_bl, q_ofs, ch_len);
151 if (r < 0) {
152 lsubdout(cct, rgw, 0) << "handle_data failed with exit code " << r << dendl;
153 return r;
154 }
155 out_bl.splice(0, q_ofs + ch_len);
156 q_ofs = 0;
157 }
158 }
159
160 cur_ofs += bl_len;
161 off_t ch_len = std::min<off_t>(out_bl.length() - q_ofs, q_len);
162 if (ch_len > 0) {
163 r = next->handle_data(out_bl, q_ofs, ch_len);
164 if (r < 0) {
165 lsubdout(cct, rgw, 0) << "handle_data failed with exit code " << r << dendl;
166 return r;
167 }
168 out_bl.splice(0, q_ofs + ch_len);
169 q_len -= ch_len;
170 q_ofs = 0;
171 }
172 return r;
173}

175int RGWGetObj_Decompress::fixup_range(off_t& ofs, off_t& end)
176{
177 if (partial_content) {
178 // if user set range, we need to calculate it in decompressed data
179 first_block = cs_info->blocks.begin(); last_block = cs_info->blocks.begin();
180 if (cs_info->blocks.size() > 1) {
181 vector<compression_block>::iterator fb, lb;
182 // not bad to use auto for lambda, I think
183 auto cmp_u = [] (off_t ofs, const compression_block& e) { return (uint64_t)ofs < e.old_ofs; };
184 auto cmp_l = [] (const compression_block& e, off_t ofs) { return e.old_ofs <= (uint64_t)ofs; };
185 fb = upper_bound(cs_info->blocks.begin()+1,
186 cs_info->blocks.end(),
187 ofs,
188 cmp_u);
189 first_block = fb - 1;
190 lb = lower_bound(fb,
191 cs_info->blocks.end(),
192 end,
193 cmp_l);
194 last_block = lb - 1;
195 }
196 } else {
197 first_block = cs_info->blocks.begin(); last_block = cs_info->blocks.end() - 1;
198 }
199
200 q_ofs = ofs - first_block->old_ofs;
201 q_len = end + 1 - ofs;
202
203 ofs = first_block->new_ofs;
204 end = last_block->new_ofs + last_block->len - 1;
205
206 cur_ofs = ofs;
207 waiting.clear();
208
209 return next->fixup_range(ofs, end);
210}
