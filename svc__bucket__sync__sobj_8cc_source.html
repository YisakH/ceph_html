74 const std::set<rgw_zone_id>& zones,

77 optional_yield y, const DoutPrefixProvider *dpp)

103int RGWSI_Bucket_Sync_SObj::resolve_policy_hints(RGWSI_Bucket_X_Ctx& ctx,
104 rgw_sync_bucket_entity& self_entity,
105 RGWBucketSyncPolicyHandlerRef& handler,
106 RGWBucketSyncPolicyHandlerRef& zone_policy_handler,
107 std::map<optional_zone_bucket, RGWBucketSyncPolicyHandlerRef>& temp_map,
108 optional_yield y,
109 const DoutPrefixProvider *dpp)
110{
111 set<rgw_zone_id> source_zones;
112 set<rgw_zone_id> target_zones;
113
114 zone_policy_handler->reflect(dpp, nullptr, nullptr,
115 nullptr, nullptr,
116 &source_zones,
117 &target_zones,
118 false); /* relaxed: also get all zones that we allow to sync to/from */
119
120 std::set<rgw_sync_bucket_entity> hint_entities;
121
122 get_hint_entities(ctx, source_zones, handler->get_source_hints(), &hint_entities, y, dpp);
123 get_hint_entities(ctx, target_zones, handler->get_target_hints(), &hint_entities, y, dpp);
124
125 std::set<rgw_sync_bucket_pipe> resolved_sources;
126 std::set<rgw_sync_bucket_pipe> resolved_dests;
127
128 for (auto& hint_entity : hint_entities) {
129 if (!hint_entity.zone ||
130 !hint_entity.bucket) {
131 continue; /* shouldn't really happen */
132 }
133
134 auto& zid = *hint_entity.zone;
135 auto& hint_bucket = *hint_entity.bucket;
136
137 RGWBucketSyncPolicyHandlerRef hint_bucket_handler;
138
139 auto iter = temp_map.find(optional_zone_bucket(zid, hint_bucket));
140 if (iter != temp_map.end()) {
141 hint_bucket_handler = iter->second;
142 } else {
143 int r = do_get_policy_handler(ctx, zid, hint_bucket, temp_map, &hint_bucket_handler, y, dpp);
144 if (r < 0) {
145 ldpp_dout(dpp, 20) << "could not get bucket sync policy handler for hint bucket=" << hint_bucket << " ... skipping" << dendl;
146 continue;
147 }
148 }
149
150 hint_bucket_handler->get_pipes(&resolved_dests,
151 &resolved_sources,
152 self_entity); /* flipping resolved dests and sources as these are
153 relative to the remote entity */
154 }
155
156 handler->set_resolved_hints(std::move(resolved_sources), std::move(resolved_dests));
157
158 return 0;
159}

161int RGWSI_Bucket_Sync_SObj::do_get_policy_handler(RGWSI_Bucket_X_Ctx& ctx,
162 std::optional<rgw_zone_id> zone,
163 std::optional<rgw_bucket> _bucket,
164 std::map<optional_zone_bucket, RGWBucketSyncPolicyHandlerRef>& temp_map,
165 RGWBucketSyncPolicyHandlerRef *handler,
166 optional_yield y,
167 const DoutPrefixProvider *dpp)
168{
169 if (!_bucket) {
170 *handler = svc.zone->get_sync_policy_handler(zone);
171 return 0;
172 }
173
174 auto& bucket = *_bucket;
175
176 string zone_key;
177 string bucket_key;
178
179 if (zone && *zone != svc.zone->zone_id()) {
180 zone_key = zone->id;
181 }
182
183 bucket_key = RGWSI_Bucket::get_bi_meta_key(bucket);
184
185 string cache_key("bi/" + zone_key + "/" + bucket_key);
186
187 if (auto e = sync_policy_cache->find(cache_key)) {
188 *handler = e->handler;
189 return 0;
190 }
191
192 bucket_sync_policy_cache_entry e;
193 rgw_cache_entry_info cache_info;
194
195 RGWBucketInfo bucket_info;
196 map<string, bufferlist> attrs;
197
198 int r = svc.bucket_sobj->read_bucket_instance_info(ctx.bi,
199 bucket_key,
200 &bucket_info,
201 nullptr,
202 &attrs,
203 y,
204 dpp,
205 &cache_info);
206 if (r < 0) {
207 if (r != -ENOENT) {
208 ldpp_dout(dpp, 0) << "ERROR: svc.bucket->read_bucket_instance_info(key=" << bucket_key << ") returned r=" << r << dendl;
209 }
210 return r;
211 }
212
213 auto zone_policy_handler = svc.zone->get_sync_policy_handler(zone);
214 if (!zone_policy_handler) {
215 ldpp_dout(dpp, 20) << "ERROR: could not find policy handler for zone=" << zone << dendl;
216 return -ENOENT;
217 }
218
219 e.handler.reset(zone_policy_handler->alloc_child(bucket_info, std::move(attrs)));
220
221 r = e.handler->init(dpp, y);
222 if (r < 0) {
223 ldpp_dout(dpp, 20) << "ERROR: failed to init bucket sync policy handler: r=" << r << dendl;
224 return r;
225 }
226
227 temp_map.emplace(optional_zone_bucket{zone, bucket}, e.handler);
228
229 rgw_sync_bucket_entity self_entity(zone.value_or(svc.zone->zone_id()), bucket);
230
231 r = resolve_policy_hints(ctx, self_entity,
232 e.handler,
233 zone_policy_handler,
234 temp_map, y, dpp);
235 if (r < 0) {
236 ldpp_dout(dpp, 20) << "ERROR: failed to resolve policy hints: bucket_key=" << bucket_key << ", r=" << r << dendl;
237 return r;
238 }
239
240 if (!sync_policy_cache->put(dpp, svc.cache, cache_key, &e, {&cache_info})) {
241 ldpp_dout(dpp, 20) << "couldn't put bucket_sync_policy cache entry, might have raced with data changes" << dendl;
242 }
243
244 *handler = e.handler;
245
246 return 0;
247}

260static bool diff_sets(std::set<rgw_bucket>& orig_set,
261 std::set<rgw_bucket>& new_set,
262 vector<rgw_bucket> *added,
263 vector<rgw_bucket> *removed)
264{
265 auto oiter = orig_set.begin();
266 auto niter = new_set.begin();
267
268 while (oiter != orig_set.end() &&
269 niter != new_set.end()) {
270 if (*oiter == *niter) {
271 ++oiter;
272 ++niter;
273 continue;
274 }
275 while (*oiter < *niter &&
276 oiter != orig_set.end()) {
277 removed->push_back(*oiter);
278 ++oiter;
279 }
280 while (*niter < *oiter
281 && niter != new_set.end()) {
282 added->push_back(*niter);
283 ++niter;
284 }
285 }
286 for (; oiter != orig_set.end(); ++oiter) {
287 removed->push_back(*oiter);
288 }
289 for (; niter != new_set.end(); ++niter) {
290 added->push_back(*niter);
291 }
292
293 return !(removed->empty() && added->empty());
294}

512template <typename C1, typename C2>
513int RGWSI_BS_SObj_HintIndexObj::update(const DoutPrefixProvider *dpp,
514 const rgw_bucket& entity,
515 const RGWBucketInfo& info_source,
516 C1 *add,
517 C2 *remove,
518 optional_yield y)
519{
520 int r = 0;
521
522 auto& info_source_ver = info_source.objv_tracker.read_version;
523
524#define MAX_RETRIES 25
525
526 for (int i = 0; i < MAX_RETRIES; ++i) {
527 if (!has_data) {
528 r = read(dpp, y);
529 if (r < 0) {
530 ldpp_dout(dpp, 0) << "ERROR: cannot update hint index: failed to read: r=" << r << dendl;
531 return r;
532 }
533 }
534
535 auto& instance = info.instances[entity];
536
537 update_entries(info_source.bucket,
538 info_source_ver,
539 add, remove,
540 &instance);
541
542 if (instance.empty()) {
543 info.instances.erase(entity);
544 }
545
546 r = flush(dpp, y);
547 if (r >= 0) {
548 return 0;
549 }
550
551 if (r != -ECANCELED) {
552 ldpp_dout(dpp, 0) << "ERROR: failed to flush hint index: obj=" << obj << " r=" << r << dendl;
553 return r;
554 }
555
556 invalidate();
557 }
558 ldpp_dout(dpp, 0) << "ERROR: failed to flush hint index: too many retries (obj=" << obj << "), likely a bug" << dendl;
559
560 return -EIO;
561}

583int RGWSI_BS_SObj_HintIndexObj::read(const DoutPrefixProvider *dpp, optional_yield y) {
584 RGWObjVersionTracker _ot;
585 bufferlist bl;
586 int r = sysobj.rop()
587 .set_objv_tracker(&_ot) /* forcing read of current version */
588 .read(dpp, &bl, y);
589 if (r < 0 && r != -ENOENT) {
590 ldpp_dout(dpp, 0) << "ERROR: failed reading data (obj=" << obj << "), r=" << r << dendl;
591 return r;
592 }
593
594 ot = _ot;
595
596 if (r >= 0) {
597 auto iter = bl.cbegin();
598 try {
599 decode(info, iter);
600 has_data = true;
601 } catch (buffer::error& err) {
602 ldpp_dout(dpp, 0) << "ERROR: " << __func__ << "(): failed to decode entries, ignoring" << dendl;
603 info.clear();
604 }
605 } else {
606 info.clear();
607 }
608
609 return 0;
610}

612int RGWSI_BS_SObj_HintIndexObj::flush(const DoutPrefixProvider *dpp, optional_yield y) {
613 int r;
614
615 if (!info.empty()) {
616 bufferlist bl;
617 encode(info, bl);
618
619 r = sysobj.wop()
620 .set_objv_tracker(&ot) /* forcing read of current version */
621 .write(dpp, bl, y);
622
623 } else { /* remove */
624 r = sysobj.wop()
625 .set_objv_tracker(&ot)
626 .remove(dpp, y);
627 }
628
629 if (r < 0) {
630 return r;
631 }
632
633 return 0;

652template <typename C1, typename C2>
653int RGWSI_Bucket_Sync_SObj_HintIndexManager::update_hints(const DoutPrefixProvider *dpp,
654 const RGWBucketInfo& bucket_info,
655 C1& added_dests,
656 C2& removed_dests,
657 C1& added_sources,
658 C2& removed_sources,
659 optional_yield y)
660{
661 C1 self_entity = { bucket_info.bucket };
662
663 if (!added_dests.empty() ||
664 !removed_dests.empty()) {
665 /* update our dests */
666 RGWSI_BS_SObj_HintIndexObj index(svc.sysobj,
667 get_dests_obj(bucket_info.bucket));
668 int r = index.update(dpp, bucket_info.bucket,
669 bucket_info,
670 &added_dests,
671 &removed_dests,
672 y);
673 if (r < 0) {
674 ldpp_dout(dpp, 0) << "ERROR: failed to update targets index for bucket=" << bucket_info.bucket << " r=" << r << dendl;
675 return r;
676 }
677
678 /* update dest buckets */
679 for (auto& dest_bucket : added_dests) {
680 RGWSI_BS_SObj_HintIndexObj dep_index(svc.sysobj,
681 get_sources_obj(dest_bucket));
682 int r = dep_index.update(dpp, dest_bucket,
683 bucket_info,
684 &self_entity,
685 static_cast<C2 *>(nullptr),
686 y);
687 if (r < 0) {
688 ldpp_dout(dpp, 0) << "ERROR: failed to update targets index for bucket=" << dest_bucket << " r=" << r << dendl;
689 return r;
690 }
691 }
692 /* update removed dest buckets */
693 for (auto& dest_bucket : removed_dests) {
694 RGWSI_BS_SObj_HintIndexObj dep_index(svc.sysobj,
695 get_sources_obj(dest_bucket));
696 int r = dep_index.update(dpp, dest_bucket,
697 bucket_info,
698 static_cast<C1 *>(nullptr),
699 &self_entity,
700 y);
701 if (r < 0) {
702 ldpp_dout(dpp, 0) << "ERROR: failed to update targets index for bucket=" << dest_bucket << " r=" << r << dendl;
703 return r;
704 }
705 }
706 }
707
708 if (!added_sources.empty() ||
709 !removed_sources.empty()) {
710 RGWSI_BS_SObj_HintIndexObj index(svc.sysobj,
711 get_sources_obj(bucket_info.bucket));
712 /* update our sources */
713 int r = index.update(dpp, bucket_info.bucket,
714 bucket_info,
715 &added_sources,
716 &removed_sources,
717 y);
718 if (r < 0) {
719 ldpp_dout(dpp, 0) << "ERROR: failed to update targets index for bucket=" << bucket_info.bucket << " r=" << r << dendl;
720 return r;
721 }
722
723 /* update added sources buckets */
724 for (auto& source_bucket : added_sources) {
725 RGWSI_BS_SObj_HintIndexObj dep_index(svc.sysobj,
726 get_dests_obj(source_bucket));
727 int r = dep_index.update(dpp, source_bucket,
728 bucket_info,
729 &self_entity,
730 static_cast<C2 *>(nullptr),
731 y);
732 if (r < 0) {
733 ldpp_dout(dpp, 0) << "ERROR: failed to update targets index for bucket=" << source_bucket << " r=" << r << dendl;
734 return r;
735 }

792int RGWSI_Bucket_Sync_SObj::handle_bi_update(const DoutPrefixProvider *dpp,
793 RGWBucketInfo& bucket_info,
794 RGWBucketInfo *orig_bucket_info,
795 optional_yield y)
796{
797 std::set<rgw_bucket> orig_sources;
798 std::set<rgw_bucket> orig_dests;
799
800 if (orig_bucket_info &&
801 orig_bucket_info->sync_policy) {
802 orig_bucket_info->sync_policy->get_potential_related_buckets(bucket_info.bucket,
803 &orig_sources,
804 &orig_dests);
805 }
806
807 std::set<rgw_bucket> sources;
808 std::set<rgw_bucket> dests;
809 if (bucket_info.sync_policy) {
810 bucket_info.sync_policy->get_potential_related_buckets(bucket_info.bucket,
811 &sources,
812 &dests);
813 }
814
815 std::vector<rgw_bucket> removed_sources;
816 std::vector<rgw_bucket> added_sources;
817 bool found = diff_sets(orig_sources, sources, &added_sources, &removed_sources);
818 ldpp_dout(dpp, 20) << __func__ << "(): bucket=" << bucket_info.bucket << ": orig_sources=" << orig_sources << " new_sources=" << sources << dendl;
819 ldpp_dout(dpp, 20) << __func__ << "(): bucket=" << bucket_info.bucket << ": potential sources added=" << added_sources << " removed=" << removed_sources << dendl;
820
821 std::vector<rgw_bucket> removed_dests;
822 std::vector<rgw_bucket> added_dests;
823 found = found || diff_sets(orig_dests, dests, &added_dests, &removed_dests);
824
825 ldpp_dout(dpp, 20) << __func__ << "(): bucket=" << bucket_info.bucket << ": orig_dests=" << orig_dests << " new_dests=" << dests << dendl;
826 ldpp_dout(dpp, 20) << __func__ << "(): bucket=" << bucket_info.bucket << ": potential dests added=" << added_dests << " removed=" << removed_dests << dendl;
827
828 if (!found) {
829 return 0;
830 }
831
832 return hint_index_mgr->update_hints(dpp, bucket_info,
833 dests, /* set all dests, not just the ones that were added */
834 removed_dests,
835 sources, /* set all sources, not just that the ones that were added */
836 removed_sources,
837 y);
838}
