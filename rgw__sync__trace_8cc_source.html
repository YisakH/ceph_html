83RGWSyncTraceNodeRef RGWSyncTraceManager::add_node(const RGWSyncTraceNodeRef& parent,
84 const std::string& type,
85 const std::string& id)
86{
87 shunique_lock wl(lock, ceph::acquire_unique);
88 auto handle = alloc_handle();
89 RGWSyncTraceNodeRef& ref = nodes[handle];
90 ref.reset(new RGWSyncTraceNode(cct, handle, parent, type, id));
91 // return a separate shared_ptr that calls finish() on the node instead of
92 // deleting it. the lambda capture holds a reference to the original 'ref'
93 auto deleter = [ref, this] (RGWSyncTraceNode *node) { finish_node(node); };
94 return {ref.get(), deleter};
95}

97bool RGWSyncTraceNode::match(const string& search_term, bool search_history)
98{
99 try {
100 std::regex expr(search_term);
101 std::smatch m;
102
103 if (regex_search(prefix, m, expr)) {
104 return true;
105 }
106 if (regex_search(status, m,expr)) {
107 return true;
108 }
109 if (!search_history) {
110 return false;
111 }
112
113 for (auto h : history) {
114 if (regex_search(h, m, expr)) {
115 return true;
116 }
117 }
118 } catch (const std::regex_error& e) {
119 ldout(cct, 5) << "NOTICE: sync trace: bad expression: bad regex search term" << dendl;
120 }
121
122 return false;
123}

159static void dump_node(RGWSyncTraceNode *entry, bool show_history, Formatter *f)
160{
161 f->open_object_section("entry");
162 ::encode_json("status", entry->to_str(), f);

199int RGWSyncTraceManager::call(std::string_view command, const cmdmap_t& cmdmap,
200 const bufferlist&,
201 Formatter *f,
202 std::ostream& ss,
203 bufferlist& out) {
204
205 bool show_history = (command == "sync trace history");
206 bool show_short = (command == "sync trace active_short");
207 bool show_active = (command == "sync trace active") || show_short;
208
209 string search;
210
211 auto si = cmdmap.find("search");
212 if (si != cmdmap.end()) {
213 search = boost::get<string>(si->second);
214 }
215
216 shunique_lock rl(lock, ceph::acquire_shared);
217
218 f->open_object_section("result");
219 f->open_array_section("running");
220 for (auto n : nodes) {
221 auto& entry = n.second;
222
223 if (!search.empty() && !entry->match(search, show_history)) {
224 continue;
225 }
226 if (show_active && !entry->test_flags(RGW_SNS_FLAG_ACTIVE)) {
227 continue;
228 }
229 if (show_short) {
230 const string& name = entry->get_resource_name();
231 if (!name.empty()) {
232 ::encode_json("entry", name, f);
233 }
234 } else {
235 dump_node(entry.get(), show_history, f);
236 }
237 f->flush(out);
238 }
239 f->close_section();
240
241 f->open_array_section("complete");
242 for (auto& entry : complete_nodes) {
243 if (!search.empty() && !entry->match(search, show_history)) {
244 continue;
245 }
246 if (show_active && !entry->test_flags(RGW_SNS_FLAG_ACTIVE)) {
247 continue;
248 }
249 dump_node(entry.get(), show_history, f);
250 f->flush(out);
251 }
252 f->close_section();
253
254 f->close_section();
255
256 return 0;
257}

259void RGWSyncTraceManager::finish_node(RGWSyncTraceNode *node)
260{
261 RGWSyncTraceNodeRef old_node;
262
263 {
264 shunique_lock wl(lock, ceph::acquire_unique);
265 if (!node) {
266 return;
267 }
268 auto iter = nodes.find(node->handle);
269 if (iter == nodes.end()) {
270 /* not found, already finished */
271 return;
272 }
273
274 if (complete_nodes.full()) {
275 /* take a reference to the entry that is going to be evicted,
276 * can't let it get evicted under lock held, otherwise
277 * it's a deadlock as it will call finish_node()
278 */
279 old_node = complete_nodes.front();
280 }
281
282 complete_nodes.push_back(iter->second);
283 nodes.erase(iter);
284 }
285};
