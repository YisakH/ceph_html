27bool rgw_s3_key_filter::decode_xml(XMLObj* obj) {
28 XMLObjIter iter = obj->find("FilterRule");
29 XMLObj *o;
30
31 const auto throw_if_missing = true;
32 auto prefix_not_set = true;
33 auto suffix_not_set = true;
34 auto regex_not_set = true;
35 std::string name;
36
37 while ((o = iter.get_next())) {
38 RGWXMLDecoder::decode_xml("Name", name, o, throw_if_missing);
39 if (name == "prefix" && prefix_not_set) {
40 prefix_not_set = false;
41 RGWXMLDecoder::decode_xml("Value", prefix_rule, o, throw_if_missing);
42 } else if (name == "suffix" && suffix_not_set) {
43 suffix_not_set = false;
44 RGWXMLDecoder::decode_xml("Value", suffix_rule, o, throw_if_missing);
45 } else if (name == "regex" && regex_not_set) {
46 regex_not_set = false;
47 RGWXMLDecoder::decode_xml("Value", regex_rule, o, throw_if_missing);
48 } else {
49 throw RGWXMLDecoder::err("invalid/duplicate S3Key filter rule name: '" + name + "'");
50 }
51 }
52 return true;
53}

136bool match(const rgw_s3_key_filter& filter, const std::string& key) {
137 const auto key_size = key.size();
138 const auto prefix_size = filter.prefix_rule.size();
139 if (prefix_size != 0) {
140 // prefix rule exists
141 if (prefix_size > key_size) {
142 // if prefix is longer than key, we fail
143 return false;
144 }
145 if (!std::equal(filter.prefix_rule.begin(), filter.prefix_rule.end(), key.begin())) {
146 return false;
147 }
148 }
149 const auto suffix_size = filter.suffix_rule.size();
150 if (suffix_size != 0) {
151 // suffix rule exists
152 if (suffix_size > key_size) {
153 // if suffix is longer than key, we fail
154 return false;
155 }
156 if (!std::equal(filter.suffix_rule.begin(), filter.suffix_rule.end(), (key.end() - suffix_size))) {
157 return false;
158 }
159 }
160 if (!filter.regex_rule.empty()) {
161 // TODO add regex chaching in the filter
162 const std::regex base_regex(filter.regex_rule);
163 if (!std::regex_match(key, base_regex)) {
164 return false;
165 }
166 }
167 return true;
168}

170bool match(const rgw_s3_key_value_filter& filter, const KeyValueMap& kv) {
171 // all filter pairs must exist with the same value in the object's metadata/tags
172 // object metadata/tags may include items not in the filter
173 return std::includes(kv.begin(), kv.end(), filter.kv.begin(), filter.kv.end());
174}

176bool match(const rgw_s3_key_value_filter& filter, const KeyMultiValueMap& kv) {
177 // all filter pairs must exist with the same value in the object's metadata/tags
178 // object metadata/tags may include items not in the filter
179 for (auto& filter : filter.kv) {
180 auto result = kv.equal_range(filter.first);
181 if (std::any_of(result.first, result.second, [&filter](const pair<string,string>& p) { return p.second == filter.second;}))
182 continue;
183 else
184 return false;
185 }
186 return true;
187}

596{

598

606 rgw_pubsub_bucket_topics bucket_topics;

621 return ret;
622 }
623 return 0;
624 }
625
626 // write back the notifications without the deleted one
627 ret = write_topics(dpp, bucket_topics, &objv_tracker, y);
628 if (ret < 0) {
629 ldpp_dout(dpp, 1) << "ERROR: failed to write topics info: ret=" << ret << dendl;
630 return ret;
631 }
632
633 return 0;

636int RGWPubSub::Bucket::remove_notifications(const DoutPrefixProvider *dpp, optional_yield y)
637{
638 // get all topics on a bucket
639 rgw_pubsub_bucket_topics bucket_topics;
640 auto ret = get_topics(&bucket_topics);
641 if (ret < 0 && ret != -ENOENT) {
642 ldpp_dout(dpp, 1) << "ERROR: failed to get list of topics from bucket '" << bucket.name << "', ret=" << ret << dendl;
643 return ret ;
644 }
645
646 // remove all auto-genrated topics
647 for (const auto& topic : bucket_topics.topics) {

763int RGWPubSub::Sub::subscribe(const DoutPrefixProvider *dpp, const string& topic, const rgw_pubsub_sub_dest& dest, optional_yield y, const std::string& s3_id)
764{
765 RGWObjVersionTracker objv_tracker;
766 rgw_pubsub_topics topics;
767
768 int ret = ps->read_topics(&topics, &objv_tracker);
769 if (ret < 0) {
770 ldpp_dout(dpp, 1) << "ERROR: failed to read topics info: ret=" << ret << dendl;
771 return ret != -ENOENT ? ret : -EINVAL;
772 }
773
774 auto iter = topics.topics.find(topic);
775 if (iter == topics.topics.end()) {
776 ldpp_dout(dpp, 1) << "ERROR: cannot add subscription to topic: topic not found" << dendl;
777 return -EINVAL;
778 }
779
780 auto& t = iter->second;
781
782 rgw_pubsub_sub_config sub_conf;
783
784 sub_conf.user = rgw_user("", ps->tenant);
785 sub_conf.name = sub;
786 sub_conf.topic = topic;
787 sub_conf.dest = dest;
788 sub_conf.s3_id = s3_id;
789
790 t.subs.insert(sub);
791
792 ret = ps->write_topics(dpp, topics, &objv_tracker, y);
793 if (ret < 0) {
794 ldpp_dout(dpp, 1) << "ERROR: failed to write topics info: ret=" << ret << dendl;
795 return ret;
796 }
797
798 ret = write_sub(dpp, sub_conf, nullptr, y);
799 if (ret < 0) {
800 ldpp_dout(dpp, 1) << "ERROR: failed to write subscription info: ret=" << ret << dendl;
801 return ret;
802 }
803 return 0;
804}

806int RGWPubSub::Sub::unsubscribe(const DoutPrefixProvider *dpp, const string& _topic, optional_yield y)
807{
808 string topic = _topic;
809 RGWObjVersionTracker sobjv_tracker;
810
811 if (topic.empty()) {
812 rgw_pubsub_sub_config sub_conf;
813 int ret = read_sub(&sub_conf, &sobjv_tracker);
814 if (ret < 0) {
815 ldpp_dout(dpp, 1) << "ERROR: failed to read subscription info: ret=" << ret << dendl;
816 return ret;
817 }
818 topic = sub_conf.topic;
819 }
820
821 RGWObjVersionTracker objv_tracker;
822 rgw_pubsub_topics topics;
823
824 int ret = ps->read_topics(&topics, &objv_tracker);
825 if (ret < 0) {
826 // not an error - could be that topic was already deleted
827 ldpp_dout(dpp, 10) << "WARNING: failed to read topics info: ret=" << ret << dendl;
828 } else {
829 auto iter = topics.topics.find(topic);
830 if (iter != topics.topics.end()) {
831 auto& t = iter->second;
832
833 t.subs.erase(sub);
834
835 ret = ps->write_topics(dpp, topics, &objv_tracker, y);
836 if (ret < 0) {
837 ldpp_dout(dpp, 1) << "ERROR: failed to write topics info: ret=" << ret << dendl;
838 return ret;
839 }
840 }
841 }
842
843 ret = remove_sub(dpp, &sobjv_tracker, y);
844 if (ret < 0) {
845 ldpp_dout(dpp, 1) << "ERROR: failed to delete subscription info: ret=" << ret << dendl;
846 return ret;
847 }
848 return 0;

863template<typename EventType>
864int RGWPubSub::SubWithEvents<EventType>::list_events(const DoutPrefixProvider *dpp, const string& marker, int max_events)
865{
866 RGWRados *store = ps->store->getRados();
867 rgw_pubsub_sub_config sub_conf;
868 int ret = get_conf(&sub_conf);
869 if (ret < 0) {
870 ldpp_dout(dpp, 1) << "ERROR: failed to read sub config: ret=" << ret << dendl;
871 return ret;
872 }
873
874 RGWBucketInfo bucket_info;
875 string tenant;
876 ret = store->get_bucket_info(&store->svc, tenant, sub_conf.dest.bucket_name, bucket_info, nullptr, null_yield, nullptr);
877 if (ret == -ENOENT) {
878 list.is_truncated = false;
879 return 0;
880 }
881 if (ret < 0) {
882 ldpp_dout(dpp, 1) << "ERROR: failed to read bucket info for events bucket: bucket=" << sub_conf.dest.bucket_name << " ret=" << ret << dendl;
883 return ret;
884 }
885
886 RGWRados::Bucket target(store, bucket_info);
887 RGWRados::Bucket::List list_op(&target);
888
889 list_op.params.prefix = sub_conf.dest.oid_prefix;
890 list_op.params.marker = marker;
891
892 std::vector<rgw_bucket_dir_entry> objs;
893
894 ret = list_op.list_objects(dpp, max_events, &objs, nullptr, &list.is_truncated, null_yield);
895 if (ret < 0) {
896 ldpp_dout(dpp, 1) << "ERROR: failed to list bucket: bucket=" << sub_conf.dest.bucket_name << " ret=" << ret << dendl;
897 return ret;
898 }
899 if (list.is_truncated) {
900 list.next_marker = list_op.get_next_marker().name;
901 }
902
903 for (auto& obj : objs) {
904 bufferlist bl64;
905 bufferlist bl;
906 bl64.append(obj.meta.user_data);
907 try {
908 bl.decode_base64(bl64);
909 } catch (buffer::error& err) {
910 ldpp_dout(dpp, 1) << "ERROR: failed to event (not a valid base64)" << dendl;
911 continue;
912 }
913 EventType event;
914
915 auto iter = bl.cbegin();
916 try {
917 decode(event, iter);
918 } catch (buffer::error& err) {
919 ldpp_dout(dpp, 1) << "ERROR: failed to decode event" << dendl;
920 continue;
921 };
922
923 list.events.push_back(event);
924 }
925 return 0;
926}
